79 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
80 NonLinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, const Material& mat,
81 VolumeLoad volumeLoad = {}, const BoundaryPatch<GridView>* neumannBoundary = nullptr,
82 NeumannBoundaryLoad neumannBoundaryLoad = {})
83 : BasePowerFE(globalBasis, element),
84 VolumeType(volumeLoad),
85 TractionType(neumannBoundary, neumannBoundaryLoad),
86 mat_{mat} {
87 this->localView().bind(element);
88 auto& firstChild = this->localView().tree().child(0);
89 const auto& fe = firstChild.finiteElement();
90 geo_ = std::make_shared<const Geometry>(this->localView().element().geometry());
91 numberOfNodes_ = fe.size();
92 order_ = 2 * (fe.localBasis().order());
93 localBasis_ = Dune::CachedLocalBasis(fe.localBasis());
94 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order_); })
95 if (this->localView().element().impl().isTrimmed())
96 localBasis_.bind(this->localView().element().impl().getQuadratureRule(order_), Dune::bindDerivatives(0, 1));
97 else
98 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
99 Dune::bindDerivatives(0, 1));
100 else
101 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
102 Dune::bindDerivatives(0, 1));
103 }

145 template <typename ScalarType, int strainDim, bool voigt = true>
146 auto materialTangent(const Eigen::Vector<ScalarType, strainDim>& strain) const {
147 if constexpr (std::is_same_v<ScalarType, double>)
148 return mat_.template tangentModuli<strainType, voigt>(strain);

150 decltype(auto) matAD = mat_.template rebind<ScalarType>();

223 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
224 using namespace Dune::DerivativeDirections;
225 using namespace Dune;
226 const auto eps = strainFunction(par);
227 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
228 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
229 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
230 const auto C = materialTangent(EVoigt);
231 const auto stresses = getStress(EVoigt);
232 for (size_t i = 0; i < numberOfNodes_; ++i) {
233 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
234 for (size_t j = 0; j < numberOfNodes_; ++j) {
235 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
236 const auto kgIJ = eps.evaluateDerivative(gpIndex, wrt(coeff(i, j)), along(stresses), on(gridElement));
237 K.template block<myDim, myDim>(i * myDim, j * myDim) += (bopI.transpose() * C * bopJ + kgIJ) * intElement;
238 }
239 }
240 }
241
242 // Update due to displacement-dependent external loads, e.g., follower loads
243 VolumeType::calculateMatrix(par, K);
244 TractionType::calculateMatrix(par, K);
245 }

256 template <ResultType resType>
257 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
258 static_assert(resType == ResultType::PK2Stress, "The requested result type is NOT implemented.");
259
260 using namespace Dune::DerivativeDirections;
261 if constexpr (resType == ResultType::PK2Stress) {
262 const auto uFunction = displacementFunction(req);
263 const auto H = uFunction.evaluateDerivative(local, Dune::wrt(spatialAll), Dune::on(gridElement));
264 const auto E = (0.5 * (H.transpose() + H + H.transpose() * H)).eval();
265 const auto EVoigt = toVoigt(E);
266
267 return mat_.template stresses<StrainTags::greenLagrangian>(EVoigt);
268 }
269 }

279 template <typename ScalarType>
280 auto calculateScalarImpl(const FERequirementType& par,
281 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const
282 -> ScalarType {
283 using namespace Dune::DerivativeDirections;
284 using namespace Dune;
285 const auto uFunction = displacementFunction(par, dx);
286 const auto eps = strainFunction(par, dx);
287 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
288 ScalarType energy = 0.0;
289
290 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
291 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
292 const auto internalEnergy = getInternalEnergy(EVoigt);
293 energy += internalEnergy * geo_->integrationElement(gp.position()) * gp.weight();
294 }
295
296 // External forces volume forces over the domain
297 energy += VolumeType::calculateScalarImpl(par, dx);
298
299 // line or surface loads, i.e., neumann boundary
300 energy += TractionType::calculateScalarImpl(par, dx);
301 return energy;
302 }

304 template <typename ScalarType>
305 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
306 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
307 using namespace Dune::DerivativeDirections;
308 using namespace Dune;
309 const auto eps = strainFunction(par, dx);
310
311 // Internal forces
312 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
313 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
314 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
315 const auto stresses = getStress(EVoigt);
316 for (size_t i = 0; i < numberOfNodes_; ++i) {
317 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
318 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
319 }
320 }
321
322 // External forces volume forces over the domain
323 VolumeType::calculateVectorImpl(par, force, dx);
324
325 // External forces, boundary forces, i.e., at the Neumann boundary
326 TractionType::calculateVectorImpl(par, force, dx);
327 }
