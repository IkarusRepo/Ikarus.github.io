34 template <typename Derived>
35 auto orthonormalizeMatrixColumns(const Eigen::MatrixBase<Derived>& A) {
36 // Gram Schmidt Ortho
37 auto Q = A.eval();
38
39 Q.col(0).normalize();
40
41 for (int colIndex = 1; colIndex < Q.cols(); colIndex++) {
42 Q.col(colIndex) -= Q.leftCols(colIndex) * (Q.leftCols(colIndex).transpose() * A.col(colIndex));
43 Q.col(colIndex).normalize();
44 }
45
46 return Q;
47 }

440 template <typename FieldVectorT>
441 auto createRandomVector(typename FieldVectorT::value_type lower = -1, typename FieldVectorT::value_type upper = 1) {
442 std::random_device rd;
443 std::mt19937 mt(rd());
444 std::uniform_real_distribution<typename FieldVectorT::value_type> dist(lower, upper);
445 auto rand = [&dist, &mt]() { return dist(mt); };
446 FieldVectorT vec;
447 std::generate(vec.begin(), vec.end(), rand);
448 return vec;

480 template <typename Derived, size_t sizeOfCondensedIndices>
481 auto staticCondensation(const Eigen::MatrixBase<Derived>& E,
482 const std::array<size_t, sizeOfCondensedIndices>& indices) {
483 constexpr size_t colsFull = std::remove_cvref_t<Derived>::ColsAtCompileTime;
484 constexpr size_t rowsFull = std::remove_cvref_t<Derived>::RowsAtCompileTime;
485 static_assert(colsFull == rowsFull, "staticCondensation only allowed for square matrices");
486 std::array<size_t, rowsFull - sizeOfCondensedIndices> remainingIndices{};
487 std::ranges::set_difference(std::ranges::iota_view(size_t(0), size_t(colsFull)), indices, remainingIndices.begin());
488
489 auto K11 = E(remainingIndices, remainingIndices);
490 auto K12 = E(indices, remainingIndices);
491 auto K22 = E(indices, indices);
492
493 return (K11 - K12.transpose() * K22.inverse() * K12).eval();
494 }

508 template <typename Derived, size_t sizeOfRemovedCols>
509 auto removeCol(const Eigen::MatrixBase<Derived>& E, const std::array<size_t, sizeOfRemovedCols>& indices) {
510 constexpr size_t colsFull = std::remove_cvref_t<Derived>::ColsAtCompileTime;
511 constexpr size_t rowsFull = std::remove_cvref_t<Derived>::RowsAtCompileTime;
512 static_assert(colsFull == 1);
513
514 std::array<size_t, rowsFull - sizeOfRemovedCols> remainingIndices{};
515 std::ranges::set_difference(std::ranges::iota_view(size_t(0), size_t(rowsFull)), indices, remainingIndices.begin());
516
517 return (E(remainingIndices)).eval();
518 }

562 template <typename Material, typename Derived>
563 decltype(auto) enlargeIfReduced(const Eigen::MatrixBase<Derived>& E) {
564 if constexpr (!Concepts::EigenVector6<Derived> and Concepts::EigenVector<Derived>) {
565 static_assert(Material::isReduced, "You should only end up here, if your material is reduced");
566
567 auto freeindices = Material::MaterialImplType::freeVoigtIndices;
568 auto p_E = Eigen::Vector<typename Material::MaterialImplType::ScalarType, 6>::Zero().eval();
569 for (int ri = 0; auto i : freeindices)
570 p_E(i) = E(ri++);
571 return p_E;
572
573 } else if constexpr (Concepts::EigenMatrix66<
574 Derived> or Concepts::EigenMatrix33<Derived> or Concepts::EigenVector6<Derived>) {
575 return E.derived();
576 } else {
577 static_assert(Material::isReduced, "You should only end up here, if your material is reduced");
578
579 auto freeindices = Material::MaterialImpl::freeVoigtIndices;
580 auto p_E = Eigen::Matrix<typename Material::MaterialImpl::ScalarType, 6, 6>::Zero().eval();
581 for (int ri = 0; auto i : freeindices) {
582 for (int rj = 0; auto j : freeindices)
583 p_E(i, j) = E(ri, rj++);
584 ++ri;
585 }
586 return p_E;
