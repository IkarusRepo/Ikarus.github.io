148 template <template <typename, int, int> class RT>
149 requires(DisplacementBasedElement::template canProvideResultType<RT>())
150 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
151 if (isDisplacementBased())
152 return DisplacementBasedElement::template calculateAt<RT>(req, local);
153
154 using namespace Dune::Indices;
155 using namespace Dune::DerivativeDirections;
156
157 using RTWrapper = ResultWrapper<RT<typename Traits::ctype, Traits::mydim, Traits::worlddim>, ResultShape::Vector>;
158 RTWrapper resultWrapper{};
159
160 if constexpr (isSameResultType<RT, ResultType::linearStress>) {
161 const auto C = DisplacementBasedElement::materialTangentFunction(req);
162 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
163 auto uFunction = DisplacementBasedElement::displacementFunction(req);
164
165 auto calculateAtContribution = [&]<typename EAST>(const EAST& easFunction) {
166 typename EAST::DType D;
167 calculateDAndLMatrix(easFunction, req, D, L_);
168
169 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
170 const auto alpha = (-D.inverse() * L_ * disp).eval();
171 const auto M = easFunction.calcM(local);
172 const auto CEval = C(local);
173 auto easStress = (CEval * M * alpha).eval();
174 resultWrapper = RTWrapper{DisplacementBasedElement::template calculateAt<RT>(req, local).asVec() + easStress};
175 };
176 easVariant_(calculateAtContribution);
177 return resultWrapper;
178 }

204 template <typename ScalarType>
205 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
206 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
207 using namespace Dune;
208 using namespace Dune::DerivativeDirections;
209
210 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
211
212 const auto uFunction = DisplacementBasedElement::displacementFunction(par, dx);
213 auto strainFunction = DisplacementBasedElement::strainFunction(par, dx);
214 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
215
216 auto C = DisplacementBasedElement::materialTangentFunction(par);
217
218 auto calculateForceContribution = [&]<typename EAST>(const EAST& easFunction) {
219 typename EAST::DType D;
220 calculateDAndLMatrix(easFunction, par, D, L_);
221
222 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
223 const auto alpha = (-D.inverse() * L_ * disp).eval();
224 const auto geo = localView().element().geometry();
225
226 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
227 const auto M = easFunction.calcM(gp.position());
228 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
229 const auto CEval = C(gpIndex);
230 auto stresses = (CEval * M * alpha).eval();
231 for (size_t i = 0; i < numberOfNodes; ++i) {
232 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
233 force.template segment<Traits::worlddim>(Traits::worlddim * i) += bopI.transpose() * stresses * intElement;
234 }
235 }
236 };
237 easVariant_(calculateForceContribution);
238 }

240private:

244 template <int enhancedStrainSize>
245 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
246 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
247 Eigen::MatrixXd& LMat) const {
248 using namespace Dune;
249 using namespace Dune::DerivativeDirections;
250
251 auto strainFunction = DisplacementBasedElement::strainFunction(par);
252 const auto C = DisplacementBasedElement::materialTangentFunction(par);
253 const auto geo = localView().element().geometry();
254 const auto numberOfNodes = DisplacementBasedElement::numberOfNodes();
255 DMat.setZero();
256 LMat.setZero(enhancedStrainSize, localView().size());
257 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
258 const auto M = easFunction.calcM(gp.position());
259 const auto CEval = C(gpIndex);
260 const double detJ = geo.integrationElement(gp.position());
261 DMat += M.transpose() * CEval * M * detJ * gp.weight();
262 for (size_t i = 0U; i < numberOfNodes; ++i) {
263 const size_t I = Traits::worlddim * i;
264 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
265 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) +=
266 M.transpose() * CEval * Bi * detJ * gp.weight();
267 }
268 }
269 }
