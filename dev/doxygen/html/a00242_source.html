147 template <ResultType resType>
148 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
149 using namespace Dune::Indices;
150 using namespace Dune::DerivativeDirections;
151 using namespace Dune;
152
153 static_assert(resType == ResultType::linearStress, "The requested result type is NOT implemented.");
154
155 auto resultVector = DisplacementBasedElement::template calculateAt<resType>(req, local);
156
157 const auto C = DisplacementBasedElement::materialTangentFunction(req);
158 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
159 auto uFunction = DisplacementBasedElement::displacementFunction(req);
160
161 auto calculateAtContribution = [&]<typename EAST>(const EAST& easFunction) {
162 typename EAST::DType D;
163 calculateDAndLMatrix(easFunction, req, D, L_);
164
165 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
166 const auto alpha = (-D.inverse() * L_ * disp).eval();
167 const auto M = easFunction.calcM(local);
168 const auto CEval = C(local);
169 auto easStress = (CEval * M * alpha).eval();
170
171 resultVector.resize(3, 1);
172 resultVector = resultVector + easStress;
173 };
174 easVariant_(calculateAtContribution);
175 return resultVector;
176 }

201 template <typename ScalarType>
202 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
203 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
204 using namespace Dune;
205 using namespace Dune::DerivativeDirections;
206
207 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
208
209 const auto uFunction = DisplacementBasedElement::displacementFunction(par, dx);
210 auto strainFunction = DisplacementBasedElement::strainFunction(par, dx);
211 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
212
213 auto C = DisplacementBasedElement::materialTangentFunction(par);
214
215 auto calculateForceContribution = [&]<typename EAST>(const EAST& easFunction) {
216 typename EAST::DType D;
217 calculateDAndLMatrix(easFunction, par, D, L_);
218
219 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
220 const auto alpha = (-D.inverse() * L_ * disp).eval();
221 const auto geo = localView().element().geometry();
222
223 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
224 const auto M = easFunction.calcM(gp.position());
225 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
226 const auto CEval = C(gpIndex);
227 auto stresses = (CEval * M * alpha).eval();
228 for (size_t i = 0; i < numberOfNodes; ++i) {
229 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
230 force.template segment<Traits::worlddim>(Traits::worlddim * i) += bopI.transpose() * stresses * intElement;
231 }
232 }
233 };
234 easVariant_(calculateForceContribution);
235 }

241 template <int enhancedStrainSize>
242 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
243 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
244 Eigen::MatrixXd& LMat) const {
245 using namespace Dune;
246 using namespace Dune::DerivativeDirections;
247
248 auto strainFunction = DisplacementBasedElement::strainFunction(par);
249 const auto C = DisplacementBasedElement::materialTangentFunction(par);
250 const auto geo = localView().element().geometry();
251 const auto numberOfNodes = DisplacementBasedElement::numberOfNodes();
252 DMat.setZero();
253 LMat.setZero(enhancedStrainSize, localView().size());
254 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
255 const auto M = easFunction.calcM(gp.position());
256 const auto CEval = C(gpIndex);
257 const double detJ = geo.integrationElement(gp.position());
258 DMat += M.transpose() * CEval * M * detJ * gp.weight();
259 for (size_t i = 0U; i < numberOfNodes; ++i) {
260 const size_t I = Traits::worlddim * i;
261 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
262 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) +=
263 M.transpose() * CEval * Bi * detJ * gp.weight();
264 }
265 }
266 }
267};
