163 template <ResultType resType>
164 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
165 using namespace Dune::Indices;
166 using namespace Dune::DerivativeDirections;
167 using namespace Dune;
168
169 static_assert(resType == ResultType::linearStress, "The requested result type is NOT implemented.");
170
171 auto resultVector = DisplacementBasedElement::template calculateAt<resType>(req, local);
172
173 if (isDisplacementBased())
174 return resultVector;
175
176 const auto C = DisplacementBasedElement::materialTangentFunction(req);
177 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
178 auto uFunction = DisplacementBasedElement::displacementFunction(req);
179 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
180
181 std::visit(
182 [&]<typename EAST>(const EAST& easFunction) {
183 if constexpr (not std::is_same_v<std::monostate, EAST>) {
184 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
185 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
186 calculateDAndLMatrix(easFunction, req, D, L_);
187 const auto alpha = (-D.inverse() * L_ * disp).eval();
188 const auto M = easFunction.calcM(local);
189 const auto CEval = C(local);
190 auto easStress = (CEval * M * alpha).eval();
191
192 resultVector.resize(3, 1);
193 resultVector = resultVector + easStress;
194 }
195 },
196 easVariant_);
197 return resultVector;
198 }

205 void setEASType(int numberOfEASParameters) {
206 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
207 if (not((numberOfNodes == 4 and Traits::mydim == 2) or (numberOfNodes == 8 and Traits::mydim == 3)) and
208 (not isDisplacementBased()))
209 DUNE_THROW(Dune::NotImplemented, "EAS only supported for Q1 or H1 elements");
210
211 if constexpr (Traits::mydim == 2) {
212 switch (numberOfEASParameters) {
213 case 0:
214 easVariant_ = std::monostate();
215 break;
216 case 4:
217 easVariant_ = EAS::Q1E4(localView().element().geometry());
218 break;
219 case 5:
220 easVariant_ = EAS::Q1E5(localView().element().geometry());
221 break;
222 case 7:
223 easVariant_ = EAS::Q1E7(localView().element().geometry());
224 break;
225 default:
226 DUNE_THROW(Dune::NotImplemented, "The given EAS parameters are not available for the 2D case.");
227 break;
228 }
229 } else if constexpr (Traits::mydim == 3) {
230 switch (numberOfEASParameters) {
231 case 0:
232 easVariant_ = std::monostate();
233 break;
234 case 9:
235 easVariant_ = EAS::H1E9(localView().element().geometry());
236 break;
237 case 21:

239 break;

241 DUNE_THROW(Dune::NotImplemented, "The given EAS parameters are not available for the 3D case.");

257 template <typename ScalarType>
258 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
259 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
260 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
261 if (isDisplacementBased())
262 return;
263 using namespace Dune;
264 const auto uFunction = DisplacementBasedElement::displacementFunction(par, dx);
265 auto strainFunction = DisplacementBasedElement::strainFunction(par, dx);
266 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
267 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
268
269 using namespace Dune::DerivativeDirections;
270
271 auto C = DisplacementBasedElement::materialTangentFunction(par);
272 const auto geo = localView().element().geometry();
273
274 // Internal forces from enhanced strains
275 std::visit(
276 [&]<typename EAST>(const EAST& easFunction) {
277 if constexpr (not std::is_same_v<std::monostate, EAST>) {
278 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
279 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
280 calculateDAndLMatrix(easFunction, par, D, L_);
281
282 const auto alpha = (-D.inverse() * L_ * disp).eval();
283
284 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
285 const auto M = easFunction.calcM(gp.position());
286 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
287 const auto CEval = C(gpIndex);
288 auto stresses = (CEval * M * alpha).eval();
289 for (size_t i = 0; i < numberOfNodes; ++i) {
290 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
291 force.template segment<Traits::worlddim>(Traits::worlddim * i) +=
292 bopI.transpose() * stresses * intElement;
293 }
294 }
295 }
296 },
297 easVariant_);
298 }

300private:
301 using EASVariant = EAS::Variants<Geometry>::type;
302 EASVariant easVariant_;
303 mutable Eigen::MatrixXd L_;
304 template <int enhancedStrainSize>
305 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
306 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
307 Eigen::MatrixXd& LMat) const {
308 using namespace Dune;
309 using namespace Dune::DerivativeDirections;
310
311 auto strainFunction = DisplacementBasedElement::strainFunction(par);
312 const auto C = DisplacementBasedElement::materialTangentFunction(par);
313 const auto geo = localView().element().geometry();
314 const auto numberOfNodes = DisplacementBasedElement::numberOfNodes();
315 DMat.setZero();
316 LMat.setZero(enhancedStrainSize, localView().size());
317 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
318 const auto M = easFunction.calcM(gp.position());
319 const auto CEval = C(gpIndex);
320 const double detJ = geo.integrationElement(gp.position());
321 DMat += M.transpose() * CEval * M * detJ * gp.weight();
322 for (size_t i = 0U; i < numberOfNodes; ++i) {
323 const size_t I = Traits::worlddim * i;
324 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
325 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) +=
326 M.transpose() * CEval * Bi * detJ * gp.weight();
327 }
328 }
329 }
