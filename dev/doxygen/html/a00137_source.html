31template <typename Geometry>
32Eigen::Matrix3d calcTransformationMatrix2D(const Geometry& geometry) {
33 const auto& referenceElement = Dune::ReferenceElements<double, 2>::general(geometry.type());
34 const auto quadPos0 = referenceElement.position(0, 0);
35
36 const auto jacobianinvT0 = toEigen(geometry.jacobianInverseTransposed(quadPos0));
37 const auto detJ0 = geometry.integrationElement(quadPos0);
38
39 auto jaco = (jacobianinvT0).inverse().eval();
40 auto J11 = jaco(0, 0);
41 auto J12 = jaco(0, 1);
42 auto J21 = jaco(1, 0);
43 auto J22 = jaco(1, 1);
44
45 Eigen::Matrix3d T0;
46 // clang-format off
47 T0 << J11 * J11, J12 * J12, J11 * J12,
48 J21 * J21, J22 * J22, J21 * J22,
49 2.0 * J11 * J21, 2.0 * J12 * J22, J21 * J12 + J11 * J22;
50 // clang-format on
51 return T0.inverse() * detJ0;
52}

63template <typename Geometry>
64Eigen::Matrix<double, 6, 6> calcTransformationMatrix3D(const Geometry& geometry) {
65 const auto& referenceElement = Dune::ReferenceElements<double, 3>::general(geometry.type());
66 const auto quadPos0 = referenceElement.position(0, 0);
67
68 const auto jacobianinvT0 = toEigen(geometry.jacobianInverseTransposed(quadPos0));
69 const auto detJ0 = geometry.integrationElement(quadPos0);
70
71 auto jaco = (jacobianinvT0).inverse().eval();
72 auto J11 = jaco(0, 0);
73 auto J12 = jaco(0, 1);
74 auto J13 = jaco(0, 2);
75 auto J21 = jaco(1, 0);
76 auto J22 = jaco(1, 1);
77 auto J23 = jaco(1, 2);
78 auto J31 = jaco(2, 0);
79 auto J32 = jaco(2, 1);
80 auto J33 = jaco(2, 2);
81
82 Eigen::Matrix<double, 6, 6> T0;
83 // clang-format off
84 T0 << J11 * J11, J12 * J12, J13 * J13, J11 * J12, J11 * J13, J12 * J13,
85 J21 * J21, J22 * J22, J23 * J23, J21 * J22, J21 * J23, J22 * J23,
86 J31 * J31, J32 * J32, J33 * J33, J31 * J32, J31 * J33, J32 * J33,
87 2.0 * J11 * J21, 2.0 * J12 * J22, 2.0 * J13 * J23, J11 * J22 + J21 * J12, J11 * J23 + J21 * J13, J12 * J23 + J22 * J13,
88 2.0 * J11 * J31, 2.0 * J12 * J32, 2.0 * J13 * J33, J11 * J32 + J31 * J12, J11 * J33 + J31 * J13, J12 * J33 + J32 * J13,
89 2.0 * J31 * J21, 2.0 * J32 * J22, 2.0 * J33 * J23, J31 * J22 + J21 * J32, J31 * J23 + J21 * J33, J32 * J23 + J22 * J33;
90 // clang-format on
91
92 return T0.inverse() * detJ0;
93}

104template <typename Geometry>
105struct EASQ1E4
106{
107 static constexpr int strainSize = 3;
108 static constexpr int enhancedStrainSize = 4;
109 using MType = Eigen::Matrix<double, strainSize, enhancedStrainSize>;
110
111 EASQ1E4() = default;
112 explicit EASQ1E4(const Geometry& geometry_)
113 : geometry{std::make_shared<Geometry>(geometry_)},
114 T0InverseTransformed{calcTransformationMatrix2D(geometry_)} {}
115
116 auto calcM(const Dune::FieldVector<double, 2>& quadPos) const {
117 MType M;
118 M.setZero(strainSize, enhancedStrainSize);
119 const double xi = quadPos[0];
120 const double eta = quadPos[1];
121 M(0, 0) = 2 * xi - 1.0;
122 M(1, 1) = 2 * eta - 1.0;
123 M(2, 2) = 2 * xi - 1.0;
124 M(2, 3) = 2 * eta - 1.0;
125 const double detJ = geometry->integrationElement(quadPos);
126 M = T0InverseTransformed / detJ * M;
127 return M;
128 }
129
130 std::shared_ptr<Geometry> geometry;
131 Eigen::Matrix3d T0InverseTransformed;
132};

471 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
472 ResultTypeMap<double>& result) const {
473 using namespace Dune::Indices;
474 using namespace Dune::DerivativeDirections;
475 using namespace Dune;
476
477 DisplacementBasedElement::calculateAt(req, local, result);
478
479 if (isDisplacementBased())
480 return;
481
482 const auto& par = req.getFERequirements();
483 const auto C = DisplacementBasedElement::materialTangentFunction(req.getFERequirements());
484 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
485 auto uFunction = DisplacementBasedElement::displacementFunction(par);
486 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
487
488 std::visit(
489 [&]<typename EAST>(const EAST& easFunction) {
490 if constexpr (not std::is_same_v<std::monostate, EAST>) {
491 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
492 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
493 calculateDAndLMatrix(easFunction, req.getFERequirements(), D, L);
494 const auto alpha = (-D.inverse() * L * disp).eval();
495 const auto M = easFunction.calcM(local);
496 const auto CEval = C(local);
497 auto easStress = (CEval * M * alpha).eval();
498 typename ResultTypeMap<double>::ResultArray resultVector;
499 if (req.isResultRequested(ResultType::linearStress)) {
500 resultVector.resize(3, 1);
501 resultVector = result.getResult(ResultType::linearStress) + easStress;
502 result.insertOrAssignResult(ResultType::linearStress, resultVector);
503 } else
504 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
505 }
506 },
507 easVariant_);
508 }

567 template <typename ScalarType>
568 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
569 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
570 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
571 if (isDisplacementBased())
572 return;
573 using namespace Dune;
574 const auto uFunction = DisplacementBasedElement::displacementFunction(par, dx);
575 auto strainFunction = DisplacementBasedElement::strainFunction(par, dx);
576 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
577 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
578
579 using namespace Dune::DerivativeDirections;
580
581 auto C = DisplacementBasedElement::materialTangentFunction(par);
582 const auto geo = localView().element().geometry();
583
584 // Internal forces from enhanced strains
585 std::visit(
586 [&]<typename EAST>(const EAST& easFunction) {
587 if constexpr (not std::is_same_v<std::monostate, EAST>) {
588 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
589 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
590 calculateDAndLMatrix(easFunction, par, D, L);
591
592 const auto alpha = (-D.inverse() * L * disp).eval();
593
594 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
595 const auto M = easFunction.calcM(gp.position());
596 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
597 const auto CEval = C(gpIndex);
598 auto stresses = (CEval * M * alpha).eval();
599 for (size_t i = 0; i < numberOfNodes; ++i) {
600 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
601 force.template segment<Traits::worlddim>(Traits::worlddim * i) +=
602 bopI.transpose() * stresses * intElement;
603 }
604 }
605 }
606 },
607 easVariant_);
608 }

614 mutable Eigen::MatrixXd L;
615 template <int enhancedStrainSize>
616 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
617 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
618 Eigen::MatrixXd& LMat) const {
619 using namespace Dune;
620 using namespace Dune::DerivativeDirections;
621
622 auto strainFunction = DisplacementBasedElement::strainFunction(par);
623 const auto C = DisplacementBasedElement::materialTangentFunction(par);
624 const auto geo = localView().element().geometry();
625 const auto numberOfNodes = DisplacementBasedElement::numberOfNodes();
626 DMat.setZero();
627 LMat.setZero(enhancedStrainSize, localView().size());
628 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
629 const auto M = easFunction.calcM(gp.position());
630 const auto CEval = C(gpIndex);
631 const double detJ = geo.integrationElement(gp.position());
632 DMat += M.transpose() * CEval * M * detJ * gp.weight();
633 for (size_t i = 0U; i < numberOfNodes; ++i) {
634 const size_t I = Traits::worlddim * i;
635 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
636 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) +=
637 M.transpose() * CEval * Bi * detJ * gp.weight();
638 }
639 }
640 }
