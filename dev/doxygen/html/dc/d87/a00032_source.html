17 template <size_t size>
18 consteval auto createfreeVoigtIndices(const std::array<StressIndexPair, size> &fixed) {
19 std::array<size_t, 6 - size> res{};
20 std::array<size_t, size> voigtFixedIndices;
21 std::ranges::transform(fixed, voigtFixedIndices.begin(), [](auto pair) { return toVoigt(pair.row, pair.col); });
22 std::ranges::sort(voigtFixedIndices);
23 std::ranges::set_difference(std::ranges::iota_view(size_t(0), size_t(6)), voigtFixedIndices, res.begin());
24 std::ranges::sort(res);
25 return res;
26 }

28 template <size_t size>
29 consteval auto createFixedVoigtIndices(const std::array<StressIndexPair, size> &fixed) {
30 std::array<size_t, size> fixedIndices;
31 std::ranges::transform(fixed, fixedIndices.begin(), [](auto pair) { return toVoigt(pair.row, pair.col); });
32 std::ranges::sort(fixedIndices);
33 return fixedIndices;
34 }

36 template <size_t size>
37 constexpr size_t countDiagonalIndices(const std::array<StressIndexPair, size> &fixed) {
38 size_t count = 0;
39 for (auto v : fixed) {
40 if (v.col == v.row) ++count;
41 }
42 return count;

84 template <bool voigt, typename Derived>
85 auto stressesImpl(const Eigen::MatrixBase<Derived> &E) const {
86 const auto [nonOp, Esol] = reduceStress(E);
87 auto stressesRed = matImpl.template stresses<MaterialImpl::strainTag, true>(Esol);
88
89 if constexpr (voigt) {
90 return removeCol(stressesRed, fixedVoigtIndices);
91 } else
92 return fromVoigt(stressesRed, false);

95 template <bool voigt, typename Derived>
96 auto tangentModuliImpl(const Eigen::MatrixBase<Derived> &E) const {
97 const auto [nonOp, Esol] = reduceStress(E);
98 auto C = matImpl.template tangentModuli<MaterialImpl::strainTag, true>(Esol);
99 if constexpr (voigt)
100 return staticCondensation(C, fixedVoigtIndices);
101 else
102 return fromVoigt(C);

112 template <typename Derived>
113 decltype(auto) maybeFromVoigt(const Eigen::MatrixBase<Derived> &E) const {
114 if constexpr (Concepts::EigenVector<Derived>) { // receiving vector means Voigt notation
115 return fromVoigt(E.derived(), true);
116 } else

120 template <typename Derived>
121 void initUnknownStrains(Eigen::MatrixBase<Derived> &E) const {
122 for (size_t i = 0; i < fixedPairs.size(); ++i) {
123 ScalarType initialVal = E(fixedPairs[i].row, fixedPairs[i].col);
124 if constexpr (strainTag == StrainTags::deformationGradient or strainTag == StrainTags::rightCauchyGreenTensor) {
125 if (Dune::FloatCmp::eq(initialVal, ScalarType(0.0)) and (fixedPairs[i].row == fixedPairs[i].col))
126 initialVal = ScalarType(1.0);
127 }
128 if (fixedPairs[i].row != fixedPairs[i].col) initialVal = ScalarType(0.0);
129 E(fixedPairs[i].row, fixedPairs[i].col) = E(fixedPairs[i].col, fixedPairs[i].row) = initialVal;
130 }
131 }

133 template <typename Derived>
134 auto reduceStress(const Eigen::MatrixBase<Derived> &p_Eraw) const {
135 auto E = maybeFromVoigt(p_Eraw);
136 initUnknownStrains(E);
137
138 std::array<size_t, fixedDiagonalVoigtIndicesSize> fixedDiagonalVoigtIndices;
139 for (size_t ri = 0; auto i : fixedVoigtIndices) {
140 auto indexPair = fromVoigt(i);
141 if (indexPair[0] == indexPair[1]) fixedDiagonalVoigtIndices[ri++] = i;
142 }
143 auto f = [&](auto &) {
144 auto S = matImpl.template stresses<MaterialImpl::strainTag, true>(E);
145 return S(fixedDiagonalVoigtIndices).eval();
146 };
147 auto df = [&](auto &) {
148 auto moduli = (matImpl.template tangentModuli<MaterialImpl::strainTag, true>(E)).eval();
149 return (moduli(fixedDiagonalVoigtIndices, fixedDiagonalVoigtIndices) / MaterialImpl::derivativeFactor).eval();
150 };
151
152 auto Er = E(fixedDiagonalVoigtIndices, fixedDiagonalVoigtIndices).eval().template cast<ScalarType>();
153 auto nonOp = Ikarus::NonLinearOperator(functions(f, df), parameter(Er));
154 auto nr = Ikarus::makeNewtonRaphson(
155 nonOp, [&](auto &r, auto &A) { return (A.inverse() * r).eval(); },
156 [&](auto &/* Ex33 */, auto &Ecomps) {
157 for (int ri = 0; auto i : fixedDiagonalVoigtIndices) {
158 auto indexPair = fromVoigt(i);
159 E(indexPair[0], indexPair[1]) += Ecomps(ri++);
160 }
161 });
162 nr->setup({.tol = tol, .maxIter = 100});
163 if (!static_cast<bool>(nr->solve()))
164 DUNE_THROW(Dune::MathError, "The stress reduction of material " << nameImpl() << " was unsuccessful\n"
165 << "The strains are\n"
166 << E << "\n The stresses are\n"
167 << f(Er));
168 return std::make_pair(nonOp, E);
169 }
170
171 MaterialImpl matImpl;
172 double tol{};
173 };
