23 void registerDirichletValues(pybind11::handle scope, pybind11::class_<DirichletValues, options...> cls) {
24 using pybind11::operator""_a;
25
26 using Basis = typename DirichletValues::Basis;
27 using BackendType = typename DirichletValues::BackendType;
28 using FlagsType = typename DirichletValues::FlagsType;
29 using MultiIndex = typename Basis::MultiIndex;
30 using LocalView = typename Basis::LocalView;
31 using Intersection = typename Basis::GridView::Intersection;
32
33 pybind11::module scopedf = pybind11::module::import("dune.functions");
34 typedef Dune::Python::LocalViewWrapper<Basis> LocalViewWrapper;
35 auto includes = Dune::Python::IncludeFiles{"dune/python/functions/globalbasis.hh"};
36 auto lv = Dune::Python::insertClass<LocalViewWrapper>(
37 scopedf, "LocalView",
38 Dune::Python::GenerateTypeName("Dune::Python::LocalViewWrapper", Dune::MetaType<Basis>()), includes)
39 .first;
40
41 cls.def(pybind11::init([](const Basis& basis) { return new DirichletValues(basis); }),
42 pybind11::keep_alive<1, 2>());
43
44 // Eigen::Ref needed due to https://pybind11.readthedocs.io/en/stable/advanced/cast/eigen.html#pass-by-reference
45 cls.def("fixBoundaryDOFs",
46 [](DirichletValues& self, const std::function<void(Eigen::Ref<Eigen::VectorX<bool>>, int)>& f) {
47 auto lambda = [&](BackendType& vec, const MultiIndex& indexGlobal) {
48 // we explicitly only allow flat indices
49 f(vec.vector(), indexGlobal[0]);
50 };
51 self.fixBoundaryDOFs(lambda);
52 });
53
54 cls.def("fixBoundaryDOFsUsingLocalView",
55 [](DirichletValues& self,
56 const std::function<void(Eigen::Ref<Eigen::VectorX<bool>>, int, LocalViewWrapper&)>& f) {
57 auto lambda = [&](BackendType& vec, int localIndex, LocalView& lv) {
58 auto lvWrapper = LocalViewWrapper(lv.globalBasis());
59 // this can be simplified when
60 // https://gitlab.dune-project.org/staging/dune-functions/-/merge_requests/418 becomes available
61 pybind11::object obj = pybind11::cast(lv.element());
62 lvWrapper.bind(obj);
63 f(vec.vector(), localIndex, lvWrapper);
64 };
65 self.fixBoundaryDOFs(lambda);
66 });
67
68 cls.def("fixBoundaryDOFsUsingLocalViewAndIntersection",

71 f) {
72 auto lambda = [&](BackendType& vec, int localIndex, LocalView& lv, const Intersection& intersection) {
73 auto lvWrapper = LocalViewWrapper(lv.globalBasis());
74 // this can be simplified when
75 // https://gitlab.dune-project.org/staging/dune-functions/-/merge_requests/418 becomes available
76 pybind11::object obj = pybind11::cast(lv.element());
77 lvWrapper.bind(obj);
78 f(vec.vector(), localIndex, lvWrapper, intersection);
79 };
80 self.fixBoundaryDOFs(lambda);
81 });
82
83 cls.def("fixDOFs",
84 [](DirichletValues& self, const std::function<void(Eigen::Ref<Eigen::VectorX<bool>>, const Basis&)>& f) {
85 self.fixBoundaryDOFs(f);
