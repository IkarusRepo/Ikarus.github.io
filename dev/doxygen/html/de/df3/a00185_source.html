17 struct IterationBased {
18 template <typename NonLinearOperator>
19 void operator()(const NonLinearSolverInformation& solverInfo, Ikarus::SubsidiaryArgs& subsidiaryArgs,
20 const NonLinearOperator&) {
21 if (subsidiaryArgs.currentStep == 0) return;
22 if (targetIterations_ == 0)
23 DUNE_THROW(Dune::InvalidStateException,
24 "For IterationBased, targetIterations should not be equal to 0. Try calling "
25 "setTargetIterations(int) first.");
26 const auto previousIterations = solverInfo.iterations;
27 if (previousIterations > targetIterations_)
28 subsidiaryArgs.stepSize
29 = sqrt(static_cast<double>(targetIterations_) / previousIterations) * subsidiaryArgs.stepSize;
30 }
31 [[nodiscard]] int targetIterations() const { return targetIterations_; }
32 void setTargetIterations(int targetIterations) { targetIterations_ = targetIterations; }
33

35 int targetIterations_{0};
