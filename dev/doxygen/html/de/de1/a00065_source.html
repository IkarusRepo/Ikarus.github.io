24 template <typename FEContainer_, typename DirichletValuesType_>
25 class FlatAssemblerBase {
26 public:
27 using FEContainerRaw = std::remove_cvref_t<FEContainer_>;
28 using FERequirementType = typename FEContainerRaw::value_type::FERequirementType;
29 using GlobalIndex = typename FEContainerRaw::value_type::GlobalIndex;
30 using Basis = typename DirichletValuesType_::Basis;
31 using GridView = typename Basis::GridView;
32 using FEContainer = FEContainer_;
33 // If we get a reference we store the reference but if we get an r-value we store by value
34 using FEContainerType = std::conditional_t<std::is_reference_v<FEContainer>, const FEContainer, FEContainer>;
35 using DirichletValuesType = DirichletValuesType_;
36 FlatAssemblerBase(FEContainer &&fes, const DirichletValuesType &p_dirichletValues)
37 : feContainer{std::forward<FEContainer>(fes)}, dirichletValues{&p_dirichletValues} {
38 constraintsBelow_.reserve(dirichletValues->size());
39 size_t counter = 0;
40 for (auto iv : std::ranges::iota_view{decltype(dirichletValues->size())(0), dirichletValues->size()}) {
41 constraintsBelow_.emplace_back(counter);
42 if (dirichletValues->isConstrained(iv)) ++counter;
43 }
44 fixedDofs = dirichletValues->fixedDOFsize();
45 }
46
48 size_t reducedSize() { return dirichletValues->size() - fixedDofs; }
49
51 size_t size() { return dirichletValues->size(); }
52
56 Eigen::VectorXd createFullVector(Eigen::Ref<const Eigen::VectorXd> reducedVector);
57
59 auto &finiteElements() const { return feContainer; }
60
62 [[nodiscard]] size_t constraintsBelow(size_t i) const { return constraintsBelow_[i]; }
63
65 [[nodiscard]] bool isConstrained(size_t i) const { return dirichletValues->isConstrained(i); }
66
70 [[nodiscard]] size_t estimateOfConnectivity() const {
71 return dirichletValues->basis().gridView().size(GridView::dimension) * 8;
72 }
73
74 private:
75 FEContainerType feContainer;
76 DirichletValuesType const *dirichletValues;
77 std::vector<size_t> constraintsBelow_{};
78 size_t fixedDofs{};
79 };

102 private:
103 double &getScalarImpl(const FERequirementType &feRequirements) {
104 scal = 0.0;
105 for (auto &fe : this->finiteElements()) {
106 scal += fe.calculateScalar(feRequirements);

109 }
