34 template <typename Derived>
35 ScalarType storedEnergyImpl(const Eigen::MatrixBase<Derived>& E) const {
36 static_assert(Concepts::EigenMatrixOrVoigtNotation3<Derived>);
37 if constexpr (Concepts::EigenVector<Derived>) {
38 const ScalarType traceE = E.template segment<3>(0).sum();
39 const ScalarType squaredNorm
40 = E.template segment<3>(0).squaredNorm() + E.template segment<3>(3).squaredNorm() / ScalarType(2.0);
41 return materialParameter.lambda / ScalarType(2.0) * traceE * traceE + materialParameter.mu * squaredNorm;
42 } else {
43 const auto traceE = E.trace();
44 return materialParameter.lambda / ScalarType(2.0) * traceE * traceE + materialParameter.mu * E.squaredNorm();
45 }
46 }

48 template <bool voigt, typename Derived>
49 auto stressesImpl(const Eigen::MatrixBase<Derived>& EMB) const {
50 static_assert(Concepts::EigenMatrixOrVoigtNotation3<decltype(EMB.eval())>);
51 const auto& E = EMB.derived();
52 if constexpr (!voigt) {
53 if constexpr (Concepts::EigenVector<Derived>) {
54 static_assert(Concepts::EigenVector6<Derived>);
55 Eigen::Matrix<ScalarType, 3, 3> S;
56 const ScalarType traceE = E.template segment<3>(0).sum();
57 S.diagonal().array()
58 = materialParameter.lambda * traceE + 2 * materialParameter.mu * E.template segment<3>(0).array();
59 S(1, 2) = S(2, 1) = materialParameter.mu * E(3);
60 S(0, 2) = S(2, 0) = materialParameter.mu * E(4);
61 S(0, 1) = S(1, 0) = materialParameter.mu * E(5); // no two since E voigt has 2* on off-diagonal terms
62 return S;
63 } else {
64 static_assert(Concepts::EigenMatrix33<Derived>);
65 return (materialParameter.lambda * E.trace() * StrainMatrix::Identity() + 2 * materialParameter.mu * E)
66 .eval();
67 }
68 } else {
69 if constexpr (Concepts::EigenVector<Derived>) {
70 static_assert(Concepts::EigenVector6<Derived>);
71 Eigen::Matrix<ScalarType, 6, 1> S;
72 const ScalarType traceE = E.template segment<3>(0).sum();
73 S.template segment<3>(0).array() = traceE * materialParameter.lambda;
74 S.template segment<3>(0) += materialParameter.mu * 2 * E.template segment<3>(0);
75 S.template segment<3>(3)
76 = materialParameter.mu * E.template segment<3>(3); // no two since E voigt has 2* on off-diagonal terms
77 return S;
78 } else {
79 Eigen::Matrix<ScalarType, 6, 1> S;
80 S.template segment<3>(0).array() = E.trace() * materialParameter.lambda;
81 S.template segment<3>(0) += 2 * materialParameter.mu * E.diagonal();
82 S(3) = 2 * materialParameter.mu * E(1, 2);
83 S(4) = 2 * materialParameter.mu * E(0, 2);
84 S(5) = 2 * materialParameter.mu * E(0, 1);
85 return S;
86 }
87 }
88 }

90 template <bool voigt, typename Derived>
91 auto tangentModuliImpl(const Eigen::MatrixBase<Derived>&) const {
92 static_assert(Concepts::EigenMatrixOrVoigtNotation3<Derived>);
93 if constexpr (!voigt) {
94 Eigen::TensorFixedSize<ScalarType, Eigen::Sizes<3, 3, 3, 3>> moduli;
95 moduli = materialParameter.lambda * identityFourthOrder()
96 + 2 * materialParameter.mu * symmetricIdentityFourthOrder();
97 return moduli;
98 } else {
99 Eigen::Matrix<ScalarType, 6, 6> moduli;
100 moduli.setZero();
101 moduli.template block<3, 3>(0, 0).array() = materialParameter.lambda;
102 moduli.template block<3, 3>(0, 0).diagonal().array() += 2 * materialParameter.mu;
103 moduli.template block<3, 3>(3, 3).diagonal().array() = materialParameter.mu;
104 return moduli;
105 }
106 }
