60 struct NoPredictor {};
61 template <typename SolutionType = NoPredictor, typename SubsidiaryType>
62 requires std::is_same_v<SolutionType, NoPredictor> || std::is_convertible_v<
63 SolutionType, std::remove_cvref_t<typename NonLinearOperatorImpl::ValueType>>
64 Ikarus::NonLinearSolverInformation solve(SubsidiaryType& subsidiaryFunction, SubsidiaryArgs& subsidiaryArgs) {
65 this->notify(NonLinearSolverMessages::INIT);
66
68 Ikarus::NonLinearSolverInformation solverInformation;
69 solverInformation.success = true;
70 auto& x = nonLinearOperator().firstParameter(); // x = D (Displacements)
71 auto& lambda = nonLinearOperator().lastParameter();
72
76
77 auto lambdaDummy = lambda;
78 auto DDummy = x;
79 x.setZero();
80 lambda = 1.0;
81 nonLinearOperator().template update<0>();
82 const auto Fext0 = (-nonLinearOperator().value()).eval(); // dRdlambda(lambda)
83 lambda = lambdaDummy;
84 x = DDummy;
85
86 Eigen::MatrixX2<double> residual2d, sol2d;
87 nonLinearOperator().updateAll();
88 const auto& rx = nonLinearOperator().value();
89 const auto& Ax = nonLinearOperator().derivative();
90
91 Eigen::VectorXd deltaD;
92 deltaD.resizeLike(rx);
93 deltaD.setZero();
94
95 subsidiaryArgs.dfdDD.resizeLike(Fext0);
96
97 subsidiaryFunction(subsidiaryArgs);
98 auto rNorm = sqrt(rx.dot(rx));
99 decltype(rNorm) dNorm;
100 int iter{0};
101 if constexpr (isLinearSolver) linearSolver.analyzePattern(Ax);
102
104 while (rNorm > settings.tol && iter < settings.maxIter) {
105 this->notify(NonLinearSolverMessages::ITERATION_STARTED);
106
108 residual2d.resize(rx.rows(), 2);
109 residual2d << -rx, Fext0;
110 sol2d.resize(rx.rows(), 2);
111
112 if constexpr (isLinearSolver) {
113 linearSolver.factorize(Ax);
114 linearSolver.solve(sol2d, residual2d);
115 } else {
116 sol2d = linearSolver(residual2d, Ax);
117 }
118
119 subsidiaryFunction(subsidiaryArgs);
120
121 const double deltalambda = (-subsidiaryArgs.f - subsidiaryArgs.dfdDD.dot(sol2d.col(0)))
122 / (subsidiaryArgs.dfdDD.dot(sol2d.col(1)) + subsidiaryArgs.dfdDlambda);
123 deltaD = sol2d.col(0) + deltalambda * sol2d.col(1);
124
125 updateFunction(x, deltaD);
126 updateFunction(subsidiaryArgs.DD, deltaD);
127
128 lambda += deltalambda;
129 subsidiaryArgs.Dlambda += deltalambda;
130
131 dNorm = sqrt(deltaD.dot(deltaD) + deltalambda * deltalambda);
132 nonLinearOperator().updateAll();
133 rNorm = sqrt(rx.dot(rx) + subsidiaryArgs.f * subsidiaryArgs.f);
134
135 this->notify(NonLinearSolverMessages::SOLUTION_CHANGED, static_cast<double>(lambda));
136 this->notify(NonLinearSolverMessages::CORRECTIONNORM_UPDATED, static_cast<double>(dNorm));
137 this->notify(NonLinearSolverMessages::RESIDUALNORM_UPDATED, static_cast<double>(rNorm));
138 this->notify(NonLinearSolverMessages::ITERATION_ENDED);
139
140 ++iter;
141 }
142
143 if (iter == settings.maxIter) solverInformation.success = false;
144 solverInformation.iterations = iter;
145 solverInformation.residualNorm = rNorm;
146 solverInformation.correctionNorm = dNorm;
147 if (solverInformation.success) this->notify(NonLinearSolverMessages::FINISHED_SUCESSFULLY, iter);
148
149 return solverInformation;
150 }

152 auto& nonLinearOperator() { return nonLinearOperator_; }
