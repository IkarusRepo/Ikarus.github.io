24 template <typename NonLinearOperatorImpl, typename LinearSolver = SolverDefault,
25 typename UpdateFunctionType_ = UpdateDefault>
26 class NewtonRaphson : public IObservable<NonLinearSolverMessages> {
27 public:
28 static constexpr bool isLinearSolver
29 = Ikarus::Concepts::LinearSolverCheck<LinearSolver, typename NonLinearOperatorImpl::DerivativeType,
30 typename NonLinearOperatorImpl::ValueType>;
31
32 using ResultType = typename NonLinearOperatorImpl::template ParameterValue<0>;
33 using UpdateFunctionType = UpdateFunctionType_;
34
35 explicit NewtonRaphson(const NonLinearOperatorImpl& p_nonLinearOperator, LinearSolver&& p_linearSolver = {},
36 UpdateFunctionType_ p_updateFunction = {})
37 : nonLinearOperator_{p_nonLinearOperator},
38 linearSolver{std::move(p_linearSolver)},
39 updateFunction{p_updateFunction} {
40 if constexpr (std::is_same_v<typename NonLinearOperatorImpl::ValueType, Eigen::VectorXd>)
41 correction.setZero(nonLinearOperator().value().size());
42 }
43
44 using NonLinearOperator = NonLinearOperatorImpl;
45
46 void setup(const NewtonRaphsonSettings& p_settings) { settings = p_settings; }
47
48 struct NoPredictor {};
49 template <typename SolutionType = NoPredictor>
50 requires std::is_same_v<SolutionType, NoPredictor> || std::is_convertible_v<
51 SolutionType, std::remove_cvref_t<typename NonLinearOperatorImpl::ValueType>>
52 [[nodiscard(
53 "The solve method returns information of the solution process. You should store this information and check if "
54 "it was successful")]] Ikarus::NonLinearSolverInformation
55 solve(const SolutionType& dx_predictor = NoPredictor{}) {
56 this->notify(NonLinearSolverMessages::INIT);
57 Ikarus::NonLinearSolverInformation solverInformation;
58 solverInformation.success = true;
59 auto& x = nonLinearOperator().firstParameter();
60 if constexpr (not std::is_same_v<SolutionType, NoPredictor>) updateFunction(x, dx_predictor);
61 nonLinearOperator().updateAll();
62 const auto& rx = nonLinearOperator().value();
63 const auto& Ax = nonLinearOperator().derivative();
64 auto rNorm = norm(rx);
65 decltype(rNorm) dNorm;
66 int iter{0};
67 if constexpr (isLinearSolver) linearSolver.analyzePattern(Ax);
68 while (rNorm > settings.tol && iter < settings.maxIter) {
69 this->notify(NonLinearSolverMessages::ITERATION_STARTED);
70 if constexpr (isLinearSolver) {
71 linearSolver.factorize(Ax);
72 linearSolver.solve(correction, -rx);
73 dNorm = correction.norm();
74 updateFunction(x, correction);
75 } else {
76 correction = -linearSolver(rx, Ax);
77 dNorm = norm(correction);
78 updateFunction(x, correction);
79 }
80 this->notify(NonLinearSolverMessages::CORRECTIONNORM_UPDATED, static_cast<double>(dNorm));
81 this->notify(NonLinearSolverMessages::SOLUTION_CHANGED);
82 nonLinearOperator().updateAll();
83 rNorm = norm(rx);
84 this->notify(NonLinearSolverMessages::RESIDUALNORM_UPDATED, static_cast<double>(rNorm));
85 this->notify(NonLinearSolverMessages::ITERATION_ENDED);
86 ++iter;
87 }
88 if (iter == settings.maxIter) solverInformation.success = false;
89 solverInformation.iterations = iter;
90 solverInformation.residualNorm = static_cast<double>(rNorm);
91 solverInformation.correctionNorm = static_cast<double>(dNorm);
92 if (solverInformation.success) this->notify(NonLinearSolverMessages::FINISHED_SUCESSFULLY, iter);
93 return solverInformation;
94 }
95
96 auto& nonLinearOperator() { return nonLinearOperator_; }
97
98 private:
99 NonLinearOperatorImpl nonLinearOperator_;
100 typename NonLinearOperatorImpl::ValueType correction;
101 LinearSolver linearSolver;
102 UpdateFunctionType updateFunction;
103 NewtonRaphsonSettings settings;
104 };
