102 template <typename F>
103 void storeInhomogeneousBoundaryCondition(F&& f) {
104 auto derivativeLambda = [&](const auto& globalCoord, const double& lambda) {
105 autodiff::real lambdaDual = lambda;
106 lambdaDual[1] = 1; // Setting the derivative in lambda direction to 1

119 void evaluateInhomogeneousBoundaryCondition(Eigen::VectorXd& xIh, const double& lambda) {
120 inhomogeneousBoundaryVectorDummy.setZero();
121 xIh.resizeLike(inhomogeneousBoundaryVectorDummy);
122 xIh.setZero();
123 for (auto& f : dirichletFunctions) {
124 interpolate(
125 basis_, inhomogeneousBoundaryVectorDummy,
126 [&](const auto& globalCoord) { return f.value(globalCoord, lambda); }, dirichletFlagsBackend);
127 xIh += inhomogeneousBoundaryVectorDummy;
128 }
129 }
