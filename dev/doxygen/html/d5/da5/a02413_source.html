38 template <class ScalarType>
39 ScalarType energyHelper(const Eigen::Vector<ScalarType, 3>& epsV, const auto& Aconv, double E, double nu) {
40 const double lambda = E * nu / ((1.0 + nu) * (1.0 - 2.0 * nu));
41 const double mu = E / (2.0 * (1.0 + nu));
42 const double lambdbar = 2.0 * lambda * mu / (lambda + 2.0 * mu);
43 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3, 3>> moduli;
44 const auto AconvT = tensorView(Aconv, std::array<Eigen::Index, 2>({3, 3}));
45 moduli = lambdbar * dyadic(AconvT, AconvT).eval() + 2 * mu * symmetricFourthOrder<double>(Aconv, Aconv);
46
47 auto C = toVoigt(moduli);
48 auto C33 = C({0, 1, 5}, {0, 1, 5}).eval();
49 return 0.5 * epsV.dot(C33 * epsV);
50 }

218 template <typename ScalarType>
219 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
220 = std::nullopt) const -> ScalarType {
221 using namespace Dune::DerivativeDirections;
222 using namespace Dune;
223 const auto [uFunction, uNodes] = getDisplacementFunction(par, dx);
224 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
225 const auto geo = this->localView().element().geometry();
226 ScalarType energy = 0.0;
227 const auto uasMatrix = Dune::viewAsEigenMatrixAsDynFixed(uNodes);
228
229 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
230 const auto [X, Jd, Hd] = geo.impl().zeroFirstAndSecondDerivativeOfPosition(gp.position());
231 const auto J = toEigen(Jd);
232 const auto H = toEigen(Hd);
233 const Eigen::Matrix<double, 2, 2> A = J * J.transpose();
234 const Eigen::Matrix<ScalarType, 3, 2> gradu = toEigen(
235 uFunction.evaluateDerivative(gpIndex, wrt(spatialAll), Dune::on(DerivativeDirections::referenceElement)));
236 const Eigen::Matrix<ScalarType, 2, 3> j = J + gradu.transpose();
237
238 const auto& Ndd = localBasis.evaluateSecondDerivatives(gpIndex);
239 const auto h = H + Ndd.transpose().template cast<ScalarType>() * uasMatrix;
240 const Eigen::Vector3<ScalarType> a3 = (j.row(0).cross(j.row(1))).normalized();
241 Eigen::Vector<ScalarType, 3> bV = h * a3;
242 bV(2) *= 2; // Voigt notation requires the two here
243
244 Eigen::Matrix<double, 3, 3> G;
245 G.setZero();
246 G.block<2, 2>(0, 0) = A;
247 G(2, 2) = 1;
248 const Eigen::Matrix<double, 3, 3> GInv = G.inverse();
249
250 const auto epsV = toVoigt((0.5 * (j * j.transpose() - A)).eval()).eval();
251 const auto BV = toVoigt(toEigen(geo.impl().secondFundamentalForm(gp.position())));
252 const auto kappaV = (BV - bV).eval();
253 const ScalarType membraneEnergy = thickness_ * energyHelper(epsV, GInv, emod_, nu_);
254 const ScalarType bendingEnergy = Dune::power(thickness_, 3) / 12.0 * energyHelper(kappaV, GInv, emod_, nu_);
255 energy += (membraneEnergy + bendingEnergy) * geo.integrationElement(gp.position()) * gp.weight();
256 }
257
258 if (volumeLoad) {
259 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
260 const auto u = uFunction.evaluate(gpIndex);
261 const Eigen::Vector<double, Traits::worlddim> fExt = volumeLoad(toEigen(geo.global(gp.position())), lambda);
262 energy -= u.dot(fExt) * geo.integrationElement(gp.position()) * gp.weight();
263 }
264 }
265
266 // line or surface loads, i.e., neumann boundary
267 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
268
269 const auto& element = this->localView().element();
270 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
271 if (not neumannBoundary or not neumannBoundary->contains(intersection)) continue;
272
273 const auto& quadLine = Dune::QuadratureRules<double, Traits::mydim - 1>::rule(intersection.type(), order);
274
275 for (const auto& curQuad : quadLine) {
276 // Local position of the quadrature point
277 const Dune::FieldVector<double, Traits::mydim>& quadPos
278 = intersection.geometryInInside().global(curQuad.position());
279
280 const double intElement = intersection.geometry().integrationElement(curQuad.position());
281
282 // The value of the local function
283 const auto u = uFunction.evaluate(quadPos);
284
285 // Value of the Neumann data at the current position
286 const auto neumannValue
287 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
288 energy -= neumannValue.dot(u) * curQuad.weight() * intElement;
289 }
290 }
291
292 return energy;
293 }
