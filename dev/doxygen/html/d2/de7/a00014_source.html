10 [[deprecated(
11 "These are hard-coded function you should use the material library Ikarus::LinearElasticity")]] inline Eigen::
12 Matrix3d
13 planeStressLinearElasticMaterialTangent(double E, double nu) {
14 Eigen::Matrix3d C;
15 C.setZero();
16 C(0, 0) = C(1, 1) = 1;
17 C(0, 1) = C(1, 0) = nu;
18 C(2, 2) = (1 - nu) / 2;
19 C *= E / (1 - nu * nu);
20 return C;
21 }

23 [[deprecated(
24 "These are hard-coded function you should use the material library: Ikarus::LinearElasticity")]] inline Eigen::
25 Matrix<double, 6, 6>
26 linearElasticMaterialTangent3D(double E, double nu) {
27 Eigen::Matrix<double, 6, 6> C;
28 C.setZero();
29 C(0, 0) = C(1, 1) = C(2, 2) = 1 - nu;
30 C(0, 1) = C(1, 0) = C(2, 0) = C(0, 2) = C(1, 2) = C(2, 1) = nu;
31 C(3, 3) = C(4, 4) = C(5, 5) = (1 - 2 * nu) / 2;
32 C *= E / ((1 + nu) * (1 - 2 * nu));
33 return C;
34 }

96 template <typename ValuePair>
97 struct ConvertLameConstants {
98 constexpr inline double toLamesFirstParameter() requires(
99 !std::is_same_v<
100 ValuePair,
101 YoungsModulusAndLamesFirstParameter> and !std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter> and !std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
102 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
103 const auto& E = vp.emodul;
104 const auto& nu = vp.nu;
105 return Dune::FloatCmp::eq(nu, 0.5) ? std::numeric_limits<double>::infinity()
106 : E * nu / ((1.0 + nu) * (1.0 - 2.0 * nu));
107 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
108 const auto& E = vp.emodul;
109 const auto& mu = vp.mu;
110 return mu * (E - 2.0 * mu) / (3.0 * mu - E);
111 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
112 const auto& E = vp.emodul;
113 const auto& K = vp.K;
114 return 3.0 * K * (3.0 * K - E) / (9.0 * K - E);
115 } else
116 assert(false && "Your LameParameter request is not implemented");
117 }
118
119 constexpr inline double toBulkModulus() requires(
120 !std::is_same_v<
121 ValuePair, YoungsModulusAndBulkModulus> and !std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
122 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
123 const auto& E = vp.emodul;
124 const auto& nu = vp.nu;
125 return E / (3.0 * (1.0 - 2.0 * nu));
126 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
127 const auto& E = vp.emodul;
128 const auto& mu = vp.mu;
129 return E * mu / (3.0 * (3.0 * mu - E));
130 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
131 const auto& E = vp.emodul;
132 const auto& lambda = vp.lambda;
133 return (E + 3.0 * lambda + calcR(vp)) / 6.0;
134 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
135 const auto& lambda = vp.lambda;
136 const auto& mu = vp.mu;
137 return lambda + 2.0 * mu / 3.0;
138 } else
139 assert(false && "Your LameParameter request is not implemented");
140 }
141
142 constexpr inline double toShearModulus() requires(
143 !std::is_same_v<
144 ValuePair,
145 YoungsModulusAndShearModulus> and !std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
146 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
147 const auto& E = vp.emodul;
148 const auto& nu = vp.nu;
149 return E / (2.0 * (1.0 + nu));
150 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
151 const auto& E = vp.emodul;
152 const auto& K = vp.K;
153 return 3.0 * K * E / (9.0 * K - E);
154 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
155 const auto& E = vp.emodul;
156 const auto& lambda = vp.lambda;
157 return (E - 3.0 * lambda + calcR(vp)) / 4.0;
158 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
159 const auto& K = vp.K;
160 const auto& lambda = vp.lambda;
161 return 3.0 * (K - lambda) / 2.0;
162 } else
163 assert(false && "Your LameParameter request is not implemented");
164 }
165
166 constexpr inline double toPWaveModulus() {
167 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
168 const auto& E = vp.emodul;
169 const auto& nu = vp.nu;
170 return E * (1.0 - nu) / ((1.0 + nu) * (1.0 - 2.0 * nu));
171 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
172 const auto& E = vp.emodul;
173 const auto& mu = vp.mu;
174 return mu * (4.0 * mu - E) / (3.0 * mu - E);
175 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
176 const auto& E = vp.emodul;
177 const auto& K = vp.K;
178 return 3.0 * K * (3.0 * K + E) / (9.0 * K - E);
179 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
180 const auto& E = vp.emodul;
181 const auto& lambda = vp.lambda;
182 return (E - lambda + calcR(vp)) / 2.0;
183 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
184 const auto& K = vp.K;
185 const auto& lambda = vp.lambda;
186 return 3.0 * K - 2.0 * lambda;
187 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
188 const auto& lambda = vp.lambda;
189 const auto& mu = vp.mu;
190 return lambda + 2.0 * mu;
191 } else
192 assert(false && "Your LameParameter request is not implemented");
193 }
194
195 constexpr inline double toPoissonsRatio() requires(!std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
196 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
197 const auto& E = vp.emodul;
198 const auto& mu = vp.mu;
199 return E / (2.0 * mu) - 1.0;
200 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
201 const auto& E = vp.emodul;
202 const auto& K = vp.K;
203 return (3.0 * K - E) / (6.0 * K);
204 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
205 const auto& E = vp.emodul;
206 const auto& lambda = vp.lambda;
207 return 2.0 * lambda / (E + lambda + calcR(vp));
208 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
209 const auto& K = vp.K;
210 const auto& lambda = vp.lambda;
211 return lambda / (3 * K - lambda);
212 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
213 const auto& lambda = vp.lambda;
214 const auto& mu = vp.mu;
215 return lambda / (2.0 * (lambda + mu));
216 } else
217 assert(false && "Your LameParameter request is not implemented");
218 }
219
220 constexpr inline double toYoungsModulus() requires(
221 !std::is_same_v<
222 ValuePair,
223 YoungsModulusAndPoissonsRatio> and !std::is_same_v<ValuePair, YoungsModulusAndShearModulus> and !std::is_same_v<ValuePair, YoungsModulusAndBulkModulus> and !std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
224 if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
225 return 9.0 * vp.K * (vp.K - vp.lambda) / (3.0 * vp.K - vp.lambda);
226 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
227 const auto& lambda = vp.lambda;
228 const auto& mu = vp.mu;
229 return mu * (3.0 * lambda + 2.0 * mu) / (lambda + mu);
230 } else
231 assert(false && "Your LameParameter request is not implemented");
232 }

250 double calcR(const YoungsModulusAndLamesFirstParameter& vp_) {
251 const auto& E = vp_.emodul;
252 const auto& lambda = vp_.lambda;
253 return std::sqrt(E * E + 9 * lambda * lambda + 2 * E * lambda);
