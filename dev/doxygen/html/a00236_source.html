171template <typename SolutionVectorType_ = std::reference_wrapper<Eigen::VectorXd>,
172 typename ParameterType_ = std::reference_wrapper<double>>
173class FERequirements
174{
175public:
176 using SolutionVectorType = SolutionVectorType_;
177 using SolutionVectorTypeRaw = typename Impl::DeduceRawVectorType<std::remove_cvref_t<SolutionVectorType_>>::Type;
178 using ParameterType = ParameterType_;
179 using ParameterTypeRaw = typename ParameterType_::type;
180
190 template <FEAffordance Affordance>
191 FERequirements& addAffordance(Affordance&& affordance) {
192 if constexpr (std::is_same_v<Affordance, ScalarAffordances>)
193 affordances.scalarAffordances = affordance;
194 else if constexpr (std::is_same_v<Affordance, VectorAffordances>)
195 affordances.vectorAffordances = affordance;
196 else if constexpr (std::is_same_v<Affordance, MatrixAffordances>)
197 affordances.matrixAffordances = affordance;
198 else if constexpr (std::is_same_v<Affordance, AffordanceCollectionImpl>)
199 affordances = affordance;
200 return *this;
201 }
202
212 FERequirements& insertParameter(const FEParameter& key, ParameterTypeRaw& val) {
213 parameter.insert_or_assign(key, val);
214 return *this;
215 }
216
226 FERequirements& insertGlobalSolution(const FESolutions& key, SolutionVectorTypeRaw& sol) {
227 sols.insert_or_assign(key, sol);
228 return *this;
229 }
230
241 const SolutionVectorTypeRaw& getGlobalSolution(const FESolutions& key) const {
242 try {
243 if constexpr (std::is_same_v<SolutionVectorType, std::reference_wrapper<Eigen::VectorXd>>)
244 return sols.at(key).get();
245 else
246 return sols.at(key);
247 } catch (std::out_of_range& oor) {
248 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
249 " in getGlobalSolution with key" + toString(key));
250 abort();
251 }
252 }
253
264 const ParameterTypeRaw& getParameter(FEParameter&& key) const {
265 try {
266 return parameter.at(key).get();
267 } catch (std::out_of_range& oor) {
268 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
269 " in getParameter with key" + toString(key));
270 abort();
271 }
272 }
273
283 template <FEAffordance Affordance>
284 bool hasAffordance(Affordance&& affordance) const {
285 if constexpr (std::is_same_v<Affordance, ScalarAffordances>)
286 return affordances.scalarAffordances == affordance;
287 else if constexpr (std::is_same_v<Affordance, VectorAffordances>)
288 return affordances.vectorAffordances == affordance;
289 else if constexpr (std::is_same_v<Affordance, MatrixAffordances>)
290 return affordances.matrixAffordances == affordance;
291 else if constexpr (std::is_same_v<Affordance, AffordanceCollectionImpl>)
292 return affordances == affordance;
293 }
294
295private:
296 std::map<FESolutions, SolutionVectorType> sols;
297 std::map<FEParameter, ParameterType> parameter;
298 AffordanceCollectionImpl affordances;
299};
