171template <typename SV = std::reference_wrapper<Eigen::VectorXd>, typename PM = std::reference_wrapper<double>>
172class FERequirements
173{
174public:
175 using SolutionVectorType = SV;
176 using SolutionVectorTypeRaw = typename Impl::DeduceRawVectorType<std::remove_cvref_t<SV>>::Type;
177 using ParameterType = PM;
178 using ParameterTypeRaw = typename PM::type;

189 template <FEAffordance Affordance>
190 FERequirements& addAffordance(Affordance&& affordance) {
191 if constexpr (std::is_same_v<Affordance, ScalarAffordances>)
192 affordances_.scalarAffordances = affordance;
193 else if constexpr (std::is_same_v<Affordance, VectorAffordances>)
194 affordances_.vectorAffordances = affordance;
195 else if constexpr (std::is_same_v<Affordance, MatrixAffordances>)
196 affordances_.matrixAffordances = affordance;
197 else if constexpr (std::is_same_v<Affordance, AffordanceCollectionImpl>)
198 affordances_ = affordance;
199 return *this;
200 }
201
211 FERequirements& insertParameter(const FEParameter& key, ParameterTypeRaw& val) {
212 parameter_.insert_or_assign(key, val);
213 return *this;
214 }
215
225 FERequirements& insertGlobalSolution(const FESolutions& key, SolutionVectorTypeRaw& sol) {
226 sols_.insert_or_assign(key, sol);
227 return *this;
228 }
229
240 const SolutionVectorTypeRaw& getGlobalSolution(const FESolutions& key) const {
241 try {
242 if constexpr (std::is_same_v<SolutionVectorType, std::reference_wrapper<Eigen::VectorXd>>)
243 return sols_.at(key).get();
244 else
245 return sols_.at(key);
246 } catch (std::out_of_range& oor) {
247 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
248 " in getGlobalSolution with key" + toString(key));
249 abort();
250 }
251 }
252
263 const ParameterTypeRaw& getParameter(FEParameter&& key) const {
264 try {
265 return parameter_.at(key).get();
266 } catch (std::out_of_range& oor) {
267 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
268 " in getParameter with key" + toString(key));
269 abort();
270 }
271 }

282 template <FEAffordance Affordance>
283 bool hasAffordance(Affordance&& affordance) const {
284 if constexpr (std::is_same_v<Affordance, ScalarAffordances>)
285 return affordances_.scalarAffordances == affordance;
286 else if constexpr (std::is_same_v<Affordance, VectorAffordances>)
287 return affordances_.vectorAffordances == affordance;
288 else if constexpr (std::is_same_v<Affordance, MatrixAffordances>)
289 return affordances_.matrixAffordances == affordance;
290 else if constexpr (std::is_same_v<Affordance, AffordanceCollectionImpl>)
291 return affordances_ == affordance;
292 }
293
294private:
295 std::map<FESolutions, SolutionVectorType> sols_;
296 std::map<FEParameter, ParameterType> parameter_;
297 AffordanceCollectionImpl affordances_;
298};
