81 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
82 NonLinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, const Material& p_mat,
83 VolumeLoad p_volumeLoad = {}, const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
84 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
85 : BasePowerFE(globalBasis.flat(), element),
86 VolumeType(p_volumeLoad),
87 TractionType(p_neumannBoundary, p_neumannBoundaryLoad),
88 mat{p_mat} {
89 this->localView().bind(element);
90 auto& first_child = this->localView().tree().child(0);
91 const auto& fe = first_child.finiteElement();
92 geo_ = std::make_shared<const Geometry>(this->localView().element().geometry());
93 numberOfNodes_ = fe.size();
94 order_ = 2 * (fe.localBasis().order());
95 localBasis = Dune::CachedLocalBasis(fe.localBasis());
96 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order_); })
97 if (this->localView().element().impl().isTrimmed())
98 localBasis.bind(this->localView().element().impl().getQuadratureRule(order_), Dune::bindDerivatives(0, 1));
99 else
100 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
101 Dune::bindDerivatives(0, 1));
102 else
103 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
104 Dune::bindDerivatives(0, 1));
105 }

148 auto materialTangent(const Eigen::Vector<ScalarType, strainDim>& strain) const {

225 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
226 using namespace Dune::DerivativeDirections;
227 using namespace Dune;
228 const auto eps = strainFunction(par);
229 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
230 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
231 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
232 const auto C = materialTangent(EVoigt);
233 const auto stresses = getStress(EVoigt);
234 for (size_t i = 0; i < numberOfNodes_; ++i) {
235 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
236 for (size_t j = 0; j < numberOfNodes_; ++j) {
237 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
238 const auto kgIJ = eps.evaluateDerivative(gpIndex, wrt(coeff(i, j)), along(stresses), on(gridElement));
239 K.template block<myDim, myDim>(i * myDim, j * myDim) += (bopI.transpose() * C * bopJ + kgIJ) * intElement;
240 }
241 }
242 }
243
244 // Update due to displacement-dependent external loads, e.g., follower loads
245 VolumeType::calculateMatrix(par, K);
246 TractionType::calculateMatrix(par, K);
247 }

256 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
257 ResultTypeMap<double>& result) const {
258 using namespace Dune::DerivativeDirections;
259 using namespace Dune;
260
261 const auto uFunction = displacementFunction(req.getFERequirements());
262 const auto H = uFunction.evaluateDerivative(local, Dune::wrt(spatialAll), Dune::on(gridElement));
263 const auto E = (0.5 * (H.transpose() + H + H.transpose() * H)).eval();
264 const auto EVoigt = toVoigt(E);
265 auto PK2 = mat.template stresses<StrainTags::greenLagrangian>(EVoigt);
266
267 if (req.isResultRequested(ResultType::PK2Stress))
268 result.insertOrAssignResult(ResultType::PK2Stress, PK2);
269 else
270 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
271 }

281 template <typename ScalarType>
282 auto calculateScalarImpl(const FERequirementType& par,
283 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const
284 -> ScalarType {
285 using namespace Dune::DerivativeDirections;
286 using namespace Dune;
287 const auto uFunction = displacementFunction(par, dx);
288 const auto eps = strainFunction(par, dx);
289 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
290 ScalarType energy = 0.0;
291
292 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
293 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
294 const auto internalEnergy = getInternalEnergy(EVoigt);
295 energy += internalEnergy * geo_->integrationElement(gp.position()) * gp.weight();
296 }
297
298 // External forces volume forces over the domain
299 energy += VolumeType::calculateScalarImpl(par, dx);
300
301 // line or surface loads, i.e., neumann boundary
302 energy += TractionType::calculateScalarImpl(par, dx);
303 return energy;
304 }

306 template <typename ScalarType>
307 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
308 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
309 using namespace Dune::DerivativeDirections;
310 using namespace Dune;
311 const auto eps = strainFunction(par, dx);
312
313 // Internal forces
314 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
315 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
316 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
317 const auto stresses = getStress(EVoigt);
318 for (size_t i = 0; i < numberOfNodes_; ++i) {
319 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
320 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
321 }
322 }
323
324 // External forces volume forces over the domain
325 VolumeType::calculateVectorImpl(par, force, dx);
326
327 // External forces, boundary forces, i.e., at the Neumann boundary
328 TractionType::calculateVectorImpl(par, force, dx);
329 }
