230 auto computeMaterialAndStrains(const Dune::FieldVector<double, 2>& gpPos, int gpIndex, const Geometry& geo,
231 const auto& uFunction) const {
232 using ST = typename std::remove_cvref_t<decltype(uFunction)>::ctype;
233
234 KinematicVariables<ST> kin;
235 using namespace Dune;
236 using namespace Dune::DerivativeDirections;
237 const auto [X, Jd, Hd] = geo_->impl().zeroFirstAndSecondDerivativeOfPosition(gpPos);
238 kin.J = toEigen(Jd);
239 kin.H = toEigen(Hd);
240 const Eigen::Matrix<double, 2, 2> A = kin.J * kin.J.transpose();
241 Eigen::Matrix<double, 3, 3> G = Eigen::Matrix<double, 3, 3>::Zero();
242
243 G.block<2, 2>(0, 0) = A;
244 G(2, 2) = 1;
245 const Eigen::Matrix<double, 3, 3> GInv = G.inverse();
246 kin.C = materialTangent(GInv);
247
248 kin.epsV = membraneStrain_.value(gpPos, geo, uFunction);
249
250 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
251 const auto uasMatrix = Dune::viewAsEigenMatrixAsDynFixed(uFunction.coefficientsRef());
252 const auto hessianu = Ndd.transpose().template cast<ST>() * uasMatrix;
253 kin.h = kin.H + hessianu;
254 const Eigen::Matrix<ST, 3, 2> gradu = toEigen(uFunction.evaluateDerivative(
255 gpIndex, Dune::wrt(spatialAll), Dune::on(Dune::DerivativeDirections::referenceElement)));
256 kin.j = kin.J + gradu.transpose();
257 kin.a3N = (kin.j.row(0).cross(kin.j.row(1)));
258 kin.a3 = kin.a3N.normalized();
259 Eigen::Vector<ST, 3> bV = kin.h * kin.a3;
260 bV(2) *= 2; // Voigt notation requires the two here
261 const auto BV = toVoigt(toEigen(geo_->impl().secondFundamentalForm(gpPos)));
262 kin.kappaV = BV - bV;
263 return kin;
264 }

266 template <typename ST>
267 void calculateMatrixImpl(const FERequirementType& par, typename Traits::template MatrixType<ST> K,
268 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const {
269 K.setZero();
270 using namespace Dune::DerivativeDirections;
271 using namespace Dune;
272 const auto uFunction = displacementFunction(par, dx);
273 const auto& lambda = par.getParameter(FEParameter::loadfactor);
274
275 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
276 const auto intElement = geo_->integrationElement(gp.position()) * gp.weight();
277 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3] =
278 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
279 const Eigen::Vector<ST, membraneStrainSize> membraneForces = thickness_ * C * epsV;
280 const Eigen::Vector<ST, bendingStrainSize> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
281
282 const auto& Nd = localBasis_.evaluateJacobian(gpIndex);
283 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
284 for (size_t i = 0; i < numberOfNodes_; ++i) {
285 Eigen::Matrix<ST, membraneStrainSize, worldDim> bopIMembrane =
286 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, i);
287
288 Eigen::Matrix<ST, bendingStrainSize, worldDim> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
289 for (size_t j = i; j < numberOfNodes_; ++j) {
290 auto KBlock = K.template block<worldDim, worldDim>(worldDim * i, worldDim * j);
291 Eigen::Matrix<ST, membraneStrainSize, worldDim> bopJMembrane =
292 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, j);
293 Eigen::Matrix<ST, bendingStrainSize, worldDim> bopJBending = bopBending(jE, h, Nd, Ndd, j, a3N, a3);
294 KBlock += thickness_ * bopIMembrane.transpose() * C * bopJMembrane * intElement;
295 KBlock += Dune::power(thickness_, 3) / 12.0 * bopIBending.transpose() * C * bopJBending * intElement;
296
297 Eigen::Matrix<ST, worldDim, worldDim> kgMembraneIJ =
298 membraneStrain_.secondDerivative(gp.position(), Nd, *geo_, uFunction, localBasis_, membraneForces, i, j);
299 Eigen::Matrix<ST, worldDim, worldDim> kgBendingIJ = kgBending(jE, h, Nd, Ndd, a3N, a3, moments, i, j);
300 KBlock += kgMembraneIJ * intElement;
301 KBlock += kgBendingIJ * intElement;
302 }
303 }
304 }
305 K.template triangularView<Eigen::StrictlyLower>() = K.transpose();
306
307 // Update due to displacement-dependent external loads, e.g., follower loads
308 VolumeType::calculateMatrixImpl(par, K, dx);
309 TractionType::calculateMatrixImpl(par, K, dx);
310 }

312 template <typename ST>
313 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ST> force,
314 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const {
315 force.setZero();
316 using namespace Dune::DerivativeDirections;
317 using namespace Dune;
318 const auto uFunction = displacementFunction(par, dx);
319 const auto& lambda = par.getParameter(FEParameter::loadfactor);
320
321 // Internal forces
322 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
323 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3] =
324 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
325 const Eigen::Vector<ST, 3> membraneForces = thickness_ * C * epsV;
326 const Eigen::Vector<ST, 3> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
327
328 const auto& Nd = localBasis_.evaluateJacobian(gpIndex);
329 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
330 for (size_t i = 0; i < numberOfNodes_; ++i) {
331 Eigen::Matrix<ST, 3, 3> bopIMembrane =
332 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, i);
333 Eigen::Matrix<ST, 3, 3> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
334 force.template segment<3>(3 * i) +=
335 bopIMembrane.transpose() * membraneForces * geo_->integrationElement(gp.position()) * gp.weight();
336 force.template segment<3>(3 * i) +=
337 bopIBending.transpose() * moments * geo_->integrationElement(gp.position()) * gp.weight();
338 }
339 }
340
341 // External forces volume forces over the domain
342 VolumeType::calculateVectorImpl(par, force, dx);
343
344 // External forces, boundary forces, i.e., at the Neumann boundary
345 TractionType::calculateVectorImpl(par, force, dx);
346 }

348 template <typename ST>
349 auto calculateScalarImpl(const FERequirementType& par,
350 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const -> ST {
351 using namespace Dune::DerivativeDirections;
352 using namespace Dune;
353 const auto uFunction = displacementFunction(par, dx);
354 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
355 ST energy = 0.0;
356
357 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
358 const auto [C, epsV, kappaV, j, J, h, H, a3N, a3] =
359 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
360
361 const ST membraneEnergy = 0.5 * thickness_ * epsV.dot(C * epsV);
362 const ST bendingEnergy = 0.5 * Dune::power(thickness_, 3) / 12.0 * kappaV.dot(C * kappaV);
363 energy += (membraneEnergy + bendingEnergy) * geo_->integrationElement(gp.position()) * gp.weight();
364 }
365
366 // External forces volume forces over the domain
367 energy += VolumeType::calculateScalarImpl(par, dx);
368
369 // line or surface loads, i.e., neumann boundary
370 energy += TractionType::calculateScalarImpl(par, dx);
371 return energy;
372 }

374 template <typename ST>
375 Eigen::Matrix<ST, 3, 3> kgBending(const Eigen::Matrix<ST, 2, 3>& jcur, const Eigen::Matrix3<ST>& h, const auto& dN,
376 const auto& ddN, const Eigen::Vector3<ST>& a3N, const Eigen::Vector3<ST>& a3,
377 const Eigen::Vector3<ST>& S, int I, int J) const {
378 Eigen::Matrix<ST, 3, 3> kg;
379 kg.setZero();
380
381 const auto& dN1i = dN(I, 0);
382 const auto& dN1j = dN(J, 0);
383 const auto& dN2i = dN(I, 1);
384 const auto& dN2j = dN(J, 1);
385
386 const Eigen::Matrix<ST, 3, 3> P =
387 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose());
388
389 const auto a1dxI = Eigen::Matrix<double, 3, 3>::Identity() *
390 dN1i; // the auto here allows the exploitation of the identity matrices,
391 // due to Eigen's expression templates
392 const auto a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN2i;
393 const auto a1dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN1j;
394 const auto a2dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN2j;
395 const auto a1 = jcur.row(0);
396 const auto a2 = jcur.row(1);
397 const Eigen::Matrix<ST, 3, 3> a3NdI = a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1);
398 const Eigen::Matrix<ST, 3, 3> a3NdJ = a1dxJ.colwise().cross(a2) - a2dxJ.colwise().cross(a1);
399 Eigen::Matrix<ST, 3, 3> a3dI = P * a3NdI;
400 Eigen::Matrix<ST, 3, 3> a3dJ = P * a3NdJ;
401 for (int i = 0; i < 3; ++i) {
402 const auto a_albe = h.row(i).transpose();
403 const auto& ddNI = ddN(I, i);
404 const auto& ddNJ = ddN(J, i);
405 Eigen::Vector3<ST> vecd = P * a_albe;
406
407 Eigen::Matrix<ST, 3, 3> a3Ndd =
408 1.0 / a3N.squaredNorm() *
409 ((3 * a3 * a3.transpose() - Eigen::Matrix<double, 3, 3>::Identity()) * (a3.dot(a_albe)) -
410 a_albe * a3.transpose() - a3 * a_albe.transpose());
411
412 Eigen::Matrix<ST, 3, 3> secondDerivativeDirectorIJ = skew(((dN2i * dN1j - dN1i * dN2j) * vecd).eval());
413 kg -= (a3NdI.transpose() * a3Ndd * a3NdJ + secondDerivativeDirectorIJ + (ddNI * a3dJ + ddNJ * a3dI.transpose())) *
414 S[i] * (i == 2 ? 2 : 1);
415 }
416
417 return kg;
418 }

420 template <typename ST>
421 Eigen::Matrix<ST, 3, 3> bopBending(const Eigen::Matrix<ST, 2, 3>& jcur, const Eigen::Matrix3<ST>& h, const auto& dN,
422 const auto& ddN, const int node, const Eigen::Vector3<ST>& a3N,
423 const Eigen::Vector3<ST>& a3) const {
424 const Eigen::Matrix<ST, 3, 3> a1dxI =
425 Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 0); // this should be double
426 // but the cross-product below complains otherwise
427 const Eigen::Matrix<ST, 3, 3> a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 1);
428 const auto a1 = jcur.row(0);
429 const auto a2 = jcur.row(1);
430 const Eigen::Matrix<ST, 3, 3> a3NdI =
431 a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1); // minus needed since order has
432 // to be swapped to get column-wise cross product working
433 const Eigen::Matrix<ST, 3, 3> a3d1 =
434 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose()) * a3NdI;
435
436 Eigen::Matrix<ST, 3, 3> bop = -(h * a3d1 + (a3 * ddN.row(node)).transpose());
437 bop.row(2) *= 2;
438
439 return bop;
440 }

442 Eigen::Matrix<double, 3, 3> materialTangent(const Eigen::Matrix<double, 3, 3>& Aconv) const {
443 const double lambda = emod_ * nu_ / ((1.0 + nu_) * (1.0 - 2.0 * nu_));
444 const double mu = emod_ / (2.0 * (1.0 + nu_));
445 const double lambdbar = 2.0 * lambda * mu / (lambda + 2.0 * mu);
446 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3, 3>> moduli;
447 const auto AconvT = tensorView(Aconv, std::array<Eigen::Index, 2>({3, 3}));
448 moduli = lambdbar * dyadic(AconvT, AconvT).eval() + 2.0 * mu * symmetricFourthOrder<double>(Aconv, Aconv);
449
450 auto C = toVoigt(moduli);
451 Eigen::Matrix<double, 3, 3> C33 = C({0, 1, 5}, {0, 1, 5});
452 return C33;
453 }
