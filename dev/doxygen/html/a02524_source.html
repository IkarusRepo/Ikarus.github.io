23template <class NonLinearElastic, class... options>
24void registerNonLinearElastic(pybind11::handle scope, pybind11::class_<NonLinearElastic, options...> cls) {
25 using pybind11::operator""_a;
26
27 using GlobalBasis = typename NonLinearElastic::Basis;
28 using FlatBasis = typename NonLinearElastic::FlatBasis;
29 using GridView = typename GlobalBasis::GridView;
30 using Element = typename NonLinearElastic::Element;
31 using Traits = typename NonLinearElastic::Traits;
32 using FERequirements = typename NonLinearElastic::FERequirementType;
33 using Material = typename NonLinearElastic::Material;
34 using FERequirements = typename NonLinearElastic::FERequirementType;
35 using ResultRequirementsType = typename NonLinearElastic::ResultRequirementsType;
36
37 cls.def(pybind11::init([](const GlobalBasis& basis, const Element& element, const Material& mat) {
38 return new NonLinearElastic(basis, element, mat);
39 }),
40 pybind11::keep_alive<1, 2>(), pybind11::keep_alive<1, 3>());
41
42 using LoadFunction = std::function<Eigen::Vector<double, Traits::worlddim>(
43 Dune::FieldVector<double, Traits::worlddim>, const double&)>;
44
45 cls.def(pybind11::init(
46 [](const GlobalBasis& basis, const Element& element, const Material& mat, const LoadFunction volumeLoad) {
47 return new NonLinearElastic(basis, element, mat, volumeLoad);
48 }),
49 pybind11::keep_alive<1, 2>(), pybind11::keep_alive<1, 3>());
50
51 cls.def(pybind11::init([](const GlobalBasis& basis, const Element& element, const Material& mat,
52 const LoadFunction volumeLoad, const BoundaryPatch<GridView>& bp,
53 const LoadFunction neumannBoundaryLoad) {
54 return new NonLinearElastic(basis, element, mat, volumeLoad, &bp, neumannBoundaryLoad);
55 }),
56 pybind11::keep_alive<1, 2>(), pybind11::keep_alive<1, 3>(), pybind11::keep_alive<1, 7>());
57
58 pybind11::module scopedf = pybind11::module::import("dune.functions");
59
60 typedef Dune::Python::LocalViewWrapper<FlatBasis> LocalViewWrapper;
61 auto includes = Dune::Python::IncludeFiles{"dune/python/functions/globalbasis.hh"};
62 auto lv = Dune::Python::insertClass<LocalViewWrapper>(
63 scopedf, "LocalViewWrapper",
64 Dune::Python::GenerateTypeName("Dune::Python::LocalViewWrapperWrapper", Dune::MetaType<FlatBasis>()),
65 includes)
66 .first;
67 lv.def("bind", &LocalViewWrapper::bind);
68 lv.def("unbind", &LocalViewWrapper::unbind);
69 lv.def("index", [](const LocalViewWrapper& localView, int index) { return localView.index(index); });
70 lv.def("__len__", [](LocalViewWrapper& self) -> int { return self.size(); });
71
72 Dune::Python::Functions::registerTree<typename LocalViewWrapper::Tree>(lv);
73 lv.def("tree", [](const LocalViewWrapper& view) { return view.tree(); });
74
75 auto basisName = Dune::className<FlatBasis>();
76 auto entry = Dune::Python::insertClass<FlatBasis>(scopedf, "DefaultGlobalBasis", pybind11::buffer_protocol(),
77 Dune::Python::GenerateTypeName(basisName),
78 Dune::Python::IncludeFiles{"dune/python/functions/globalbasis.hh"});
79 if (entry.second)
80 Dune::Python::registerGlobalBasis(scopedf, entry.first);
81
82 cls.def(
83 "localView",
84 [](NonLinearElastic& self) -> LocalViewWrapper {
85 auto lvWrapped = LocalViewWrapper(self.localView().globalBasis());
86 // this can be simplified when https://gitlab.dune-project.org/staging/dune-functions/-/merge_requests/418
87 // becomes available
88 pybind11::object obj = pybind11::cast(self.localView().element());
89 lvWrapped.bind(obj);
90 return lvWrapped;
91 },
92 pybind11::keep_alive<0, 1>());
93 cls.def("calculateScalar",
94 [](NonLinearElastic& self, const FERequirements& req) { return self.calculateScalar(req); });
95 cls.def("calculateVector", [](NonLinearElastic& self, const FERequirements& req, Eigen::Ref<Eigen::VectorXd> vec) {
96 return self.calculateVector(req, vec);
97 });
98 cls.def(
99 "calculateMatrix",
100 [](NonLinearElastic& self, const FERequirements& req, Eigen::Ref<Eigen::MatrixXd> mat) {
101 return self.calculateMatrix(req, mat);
102 },
103 pybind11::arg("FERequirements"), pybind11::arg("elementMatrix").noconvert());
104
105 cls.def(
106 "resultAt",
107 [](NonLinearElastic& self, const ResultRequirementsType& req,
108 const Dune::FieldVector<double, Traits::mydim>& local, ResultType resType = ResultType::noType) {
109 ResultTypeMap<double> resultTypeMap;
110 self.calculateAt(req, local, resultTypeMap);
111 if (resType == ResultType::noType)
112 return resultTypeMap.getSingleResult().second;
113 else
114 return resultTypeMap.getResult(resType);
115 },
116 pybind11::arg("resultRequirements"), pybind11::arg("local"), pybind11::arg("resultType") = ResultType::noType);
117}
