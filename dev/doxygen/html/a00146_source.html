72 template <typename ST>
73 auto calculateScalarImpl(const FERequirementType& par,
74 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const -> ST {
75 if (not volumeLoad_)
76 return 0.0;
77 ST energy = 0.0;
78 const auto uFunction = dbElement().displacementFunction(par, dx);
79 const auto geo = dbElement().geometry();
80 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
81
82 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
83 const auto uVal = uFunction.evaluate(gpIndex);
84 Eigen::Vector<double, worldDim> fext = volumeLoad_(geo.global(gp.position()), lambda);
85 energy -= uVal.dot(fext) * geo.integrationElement(gp.position()) * gp.weight();
86 }
87 return energy;
88 }

90 template <typename ST>
91 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ST> force,
92 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const {
93 if (not volumeLoad_)
94 return;
95 using namespace Dune::DerivativeDirections;
96 using namespace Dune;
97 const auto uFunction = dbElement().displacementFunction(par, dx);
98 const auto geo = dbElement().geometry();
99 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
100
101 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
102 const Eigen::Vector<double, worldDim> fext = volumeLoad_(geo.global(gp.position()), lambda);
103 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
104 for (size_t i = 0; i < dbElement().numberOfNodes(); ++i) {
105 const auto udCi = uFunction.evaluateDerivative(gpIndex, wrt(coeff(i)));
106 force.template segment<worldDim>(worldDim * i) -= udCi * fext * intElement;
107 }
108 }
109 }
