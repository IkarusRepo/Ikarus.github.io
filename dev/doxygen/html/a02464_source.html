185 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
186 const auto eps = strainFunction(par);
187 using namespace Dune::DerivativeDirections;
188 using namespace Dune;
189
190 const auto C = materialTangent();
191 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
192 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
193 for (size_t i = 0; i < numberOfNodes_; ++i) {
194 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
195 for (size_t j = 0; j < numberOfNodes_; ++j) {
196 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
197 K.template block<myDim, myDim>(i * myDim, j * myDim) += bopI.transpose() * C * bopJ * intElement;
198 }
199 }
200 }
201
202 // Update due to displacement-dependent external loads, e.g., follower loads
203 VolumeType::calculateMatrix(par, K);
204 TractionType::calculateMatrix(par, K);
205 }

216 template <ResultType resType>
217 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
218 static_assert(resType == ResultType::linearStress, "The requested result type is NOT implemented.");
219
220 if constexpr (resType == ResultType::linearStress) {
221 const auto eps = strainFunction(req);
222 const auto C = materialTangent();
223 auto epsVoigt = eps.evaluate(local, Dune::on(Dune::DerivativeDirections::gridElement));
224
225 return (C * epsVoigt).eval();
226 }
227 }

238 template <typename ScalarType>
239 auto calculateScalarImpl(const FERequirementType& par,
240 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const
241 -> ScalarType {
242 const auto uFunction = displacementFunction(par, dx);
243 const auto eps = strainFunction(par, dx);
244 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
245 using namespace Dune::DerivativeDirections;
246 using namespace Dune;
247
248 const auto C = materialTangent();
249
250 ScalarType energy = 0.0;
251 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
252 const auto EVoigt = eps.evaluate(gpIndex, on(gridElement));
253 energy += (0.5 * EVoigt.dot(C * EVoigt)) * geo_->integrationElement(gp.position()) * gp.weight();
254 }
255
256 // External forces volume forces over the domain
257 energy += VolumeType::calculateScalarImpl(par, dx);
258
259 // line or surface loads, i.e., neumann boundary
260 energy += TractionType::calculateScalarImpl(par, dx);
261 return energy;
262 }

264 template <typename ScalarType>
265 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
266 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
267 const auto eps = strainFunction(par, dx);
268 using namespace Dune::DerivativeDirections;
269 using namespace Dune;
270
271 const auto C = materialTangent();
272
273 // Internal forces
274 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
275 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
276 auto stresses = (C * eps.evaluate(gpIndex, on(gridElement))).eval();
277 for (size_t i = 0; i < numberOfNodes_; ++i) {
278 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
279 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
280 }
281 }
282
283 // External forces volume forces over the domain
284 VolumeType::calculateVectorImpl(par, force, dx);
285
286 // External forces, boundary forces, i.e., at the Neumann boundary
287 TractionType::calculateVectorImpl(par, force, dx);
288 }
