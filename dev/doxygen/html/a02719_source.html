88 template <typename F>
89 void fixBoundaryDOFs(F&& f) {
90 if constexpr (Concepts::IsFunctorWithArgs<F, BackendType, typename Basis::MultiIndex>) {
91 auto lambda = [&](auto&& indexGlobal) { f(dirichletFlagsBackend_, indexGlobal); };
92 Dune::Functions::forEachBoundaryDOF(basis_, lambda);
93 } else if constexpr (Concepts::IsFunctorWithArgs<F, BackendType, int, typename Basis::LocalView>) {
94 auto lambda = [&](auto&& localIndex, auto&& localView) { f(dirichletFlagsBackend_, localIndex, localView); };
95 Dune::Functions::forEachBoundaryDOF(basis_, lambda);
96 } else if constexpr (Concepts::IsFunctorWithArgs<F, BackendType, int, typename Basis::LocalView,
97 typename Basis::GridView::Intersection>) {
98 auto lambda = [&](auto&& localIndex, auto&& localView, auto&& intersection) {
99 f(dirichletFlagsBackend_, localIndex, localView, intersection);
100 };
101 Dune::Functions::forEachBoundaryDOF(basis_, lambda);
102 } else
103 DUNE_THROW(Dune::IOError, "Invalid callback function passed to fixBoundaryDOFs");
104 }

157 template <typename F>
158 void storeInhomogeneousBoundaryCondition(F&& f) {
159 auto derivativeLambda = [&](const auto& globalCoord, const double& lambda) {
160 autodiff::real lambdaDual = lambda;
161 lambdaDual[1] = 1; // Setting the derivative in lambda direction to 1
162 return derivative(f(globalCoord, lambdaDual));
163 };
164 dirichletFunctions_.push_back({f, derivativeLambda});
165 }

176 void evaluateInhomogeneousBoundaryCondition(Eigen::VectorXd& xIh, const double& lambda) {
177 inhomogeneousBoundaryVectorDummy_.setZero();
178 xIh.resizeLike(inhomogeneousBoundaryVectorDummy_);
179 xIh.setZero();
180 for (auto& f : dirichletFunctions_) {
181 interpolate(
182 basis_, inhomogeneousBoundaryVectorDummy_,
183 [&](const auto& globalCoord) { return f.value(globalCoord, lambda); }, dirichletFlagsBackend_);
184 xIh += inhomogeneousBoundaryVectorDummy_;
185 }
186 }
187
197 void evaluateInhomogeneousBoundaryConditionDerivative(Eigen::VectorXd& xIh, const double& lambda) {
198 inhomogeneousBoundaryVectorDummy_.setZero();
199 xIh.resizeLike(inhomogeneousBoundaryVectorDummy_);
200 xIh.setZero();
201 for (auto& f : dirichletFunctions_) {
202 interpolate(
203 basis_, inhomogeneousBoundaryVectorDummy_,
204 [&](const auto& globalCoord) { return f.derivative(globalCoord, lambda); }, dirichletFlagsBackend_);
205 xIh += inhomogeneousBoundaryVectorDummy_;
206 }
207 }
