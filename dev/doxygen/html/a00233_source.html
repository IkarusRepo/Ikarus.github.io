96 template <typename SolutionType = NoPredictor, typename SubsidiaryType>
97 requires std::is_same_v<SolutionType, NoPredictor> ||
98 std::is_convertible_v<SolutionType, std::remove_cvref_t<typename NonLinearOperatorImpl::ValueType>>
99 [[nodiscard(
100 "The solve method returns information of the solution process. You should store this information and check if "
101 "it was successful")]] NonLinearSolverInformation
102 solve(SubsidiaryType& subsidiaryFunction, SubsidiaryArgs& subsidiaryArgs,
103 const SolutionType& dx_predictor = NoPredictor{}) {
104 this->notify(NonLinearSolverMessages::INIT);
105
107 Ikarus::NonLinearSolverInformation solverInformation;
108 solverInformation.success = true;
109 auto& x = nonLinearOperator().firstParameter(); // x = D (Displacements)
110 if constexpr (not std::is_same_v<SolutionType, NoPredictor>)
111 updateFunction(x, dx_predictor);
112 auto& lambda = nonLinearOperator().lastParameter();
113
117
118 auto lambdaDummy = lambda;
119 auto DDummy = x;
120 x.setZero();
121 lambda = 1.0;
122 nonLinearOperator().template update<0>();
123 const auto Fext0 = (-nonLinearOperator().value()).eval(); // dRdlambda(lambda)
124 lambda = lambdaDummy;
125 x = DDummy;
126
127 Eigen::MatrixX2<double> residual2d, sol2d;
128 nonLinearOperator().updateAll();
129 const auto& rx = nonLinearOperator().value();
130 const auto& Ax = nonLinearOperator().derivative();
131
132 Eigen::VectorXd deltaD;
133 deltaD.resizeLike(rx);
134 deltaD.setZero();
135
136 subsidiaryArgs.dfdDD.resizeLike(Fext0);
137
138 subsidiaryFunction(subsidiaryArgs);
139 auto rNorm = sqrt(rx.dot(rx));
140 decltype(rNorm) dNorm;
141 int iter{0};
142 if constexpr (isLinearSolver)
143 linearSolver.analyzePattern(Ax);
144
146 while (rNorm > settings.tol && iter < settings.maxIter) {
147 this->notify(NonLinearSolverMessages::ITERATION_STARTED);
148
150 residual2d.resize(rx.rows(), 2);
151 residual2d << -rx, Fext0;
152 sol2d.resize(rx.rows(), 2);
153
154 if constexpr (isLinearSolver) {
155 linearSolver.factorize(Ax);
156 linearSolver.solve(sol2d, residual2d);
157 } else {
158 sol2d = linearSolver(residual2d, Ax);
159 }
160
161 subsidiaryFunction(subsidiaryArgs);
162
163 const double deltalambda = (-subsidiaryArgs.f - subsidiaryArgs.dfdDD.dot(sol2d.col(0))) /
164 (subsidiaryArgs.dfdDD.dot(sol2d.col(1)) + subsidiaryArgs.dfdDlambda);
165 deltaD = sol2d.col(0) + deltalambda * sol2d.col(1);
166
167 updateFunction(x, deltaD);
168 updateFunction(subsidiaryArgs.DD, deltaD);
169
170 lambda += deltalambda;
171 subsidiaryArgs.Dlambda += deltalambda;
172
173 dNorm = sqrt(deltaD.dot(deltaD) + deltalambda * deltalambda);
174 nonLinearOperator().updateAll();
175 rNorm = sqrt(rx.dot(rx) + subsidiaryArgs.f * subsidiaryArgs.f);
176
177 this->notify(NonLinearSolverMessages::SOLUTION_CHANGED, static_cast<double>(lambda));
178 this->notify(NonLinearSolverMessages::CORRECTIONNORM_UPDATED, static_cast<double>(dNorm));
179 this->notify(NonLinearSolverMessages::RESIDUALNORM_UPDATED, static_cast<double>(rNorm));
180 this->notify(NonLinearSolverMessages::ITERATION_ENDED);
181
182 ++iter;
183 }
184
185 if (iter == settings.maxIter)
186 solverInformation.success = false;
187 solverInformation.iterations = iter;
188 solverInformation.residualNorm = rNorm;
189 solverInformation.correctionNorm = dNorm;
190 if (solverInformation.success)
191 this->notify(NonLinearSolverMessages::FINISHED_SUCESSFULLY, iter);
192
193 return solverInformation;
194 }

200 auto& nonLinearOperator() { return nonLinearOperator_; }

204 LinearSolver linearSolver;

206 NewtonRaphsonWithSubsidiaryFunctionSettings settings;
207};
