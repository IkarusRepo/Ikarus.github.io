26 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> h) const {
27 if constexpr (requires { RealElement::calculateMatrix(par, h); } and not forceAutoDiff) {
28 RealElement::calculateMatrix(par, h);
29 } else if constexpr (requires {
30 this->template calculateVectorImpl<autodiff::dual>(
31 par, std::declval<typename Traits::template VectorType<autodiff::dual>>(),
32 std::declval<const Eigen::VectorXdual&>());
33 }) {
36 Eigen::VectorXdual dx(this->localView().size());
37 Eigen::VectorXdual g(this->localView().size());
38 dx.setZero();
39 auto f = [&](auto& x) -> auto& {
40 g.setZero();
41 this->template calculateVectorImpl<autodiff::dual>(par, g, x);
42 return g;
43 };
44 jacobian(f, autodiff::wrt(dx), at(dx), g, h);
45 } else if constexpr (requires {
46 this->template calculateScalarImpl<autodiff::dual2nd>(
47 par, std::declval<typename Traits::template VectorType<autodiff::dual2nd>>());
48 }) {
49 Eigen::VectorXdual2nd dx(this->localView().size());
50 Eigen::VectorXd g;
51 autodiff::dual2nd e;
52 dx.setZero();
53 auto f = [&](auto& x) { return this->template calculateScalarImpl<autodiff::dual2nd>(par, x); };
54 hessian(f, autodiff::wrt(dx), at(dx), e, g, h);
55 } else
56 static_assert(Ikarus::Std::DummyFalse<AutoDiffFE>::value,
57 "Appropriate calculateScalarImpl or calculateVectorImpl functions are not implemented for the "
58 "chosen element.");
59 }

61 inline void calculateVector(const FERequirementType& par, typename Traits::template VectorType<> g) const {
62 if constexpr (requires {
63 this->template calculateVectorImpl<double>(
64 par, std::declval<typename Traits::template VectorType<double>>(),
65 std::declval<const Eigen::VectorXd&>());
66 }
67 and not forceAutoDiff) {
68 return this->template calculateVectorImpl<double>(par, g);
69 } else if constexpr (requires {
70 this->template calculateScalarImpl<autodiff::dual>(
71 par, std::declval<const Eigen::VectorXdual&>());
72 }) {
73 Eigen::VectorXdual dx(this->localView().size());
74 dx.setZero();
75 autodiff::dual e;
76 auto f = [&](auto& x) { return this->template calculateScalarImpl<autodiff::dual>(par, x); };
77 gradient(f, autodiff::wrt(dx), at(dx), e, g);
78 } else
79 static_assert(Ikarus::Std::DummyFalse<AutoDiffFE>::value,
80 "Appropriate calculateScalarImpl function is not implemented for the "
81 "chosen element.");
82 }

84 void calculateLocalSystem(const FERequirementType& par, typename Traits::template MatrixType<> h,
85 typename Traits::template VectorType<> g) const {
86 Eigen::VectorXdual2nd dx(this->localView().size());
87 dx.setZero();
88 auto f = [&](auto& x) { return this->calculateScalarImpl(par, x); };
89 hessian(f, autodiff::wrt(dx), at(dx), g, h);
90 }

92 [[nodiscard]] double calculateScalar(const FERequirementType& par) const {
93 if constexpr (requires { RealElement::calculateScalar(par); }) {
94 return RealElement::calculateScalar(par);

96 return this->calculateScalarImpl(par);
97 } else
98 static_assert(Ikarus::Std::DummyFalse<AutoDiffFE>::value,
99 "Appropriate calculateScalar and calculateScalarImpl functions are not implemented for the "
100 "chosen element.");
101 }
