71 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
72 NonLinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, const Material& p_mat,
73 VolumeLoad p_volumeLoad = {}, const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
74 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
75 : BasePowerFE(globalBasis.flat(), element), neumannBoundary{p_neumannBoundary}, mat{p_mat} {
76 this->localView().bind(element);
77 auto& first_child = this->localView().tree().child(0);
78 const auto& fe = first_child.finiteElement();
79 numberOfNodes = fe.size();
80 dispAtNodes.resize(numberOfNodes);
81 order = 2 * (this->localView().tree().child(0).finiteElement().localBasis().order());
82 localBasis = Dune::CachedLocalBasis(this->localView().tree().child(0).finiteElement().localBasis());
83 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order); })
84 if (this->localView().element().impl().isTrimmed())
85 localBasis.bind(this->localView().element().impl().getQuadratureRule(order), Dune::bindDerivatives(0, 1));
86 else
87 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
88 Dune::bindDerivatives(0, 1));
89 else
90 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
91 Dune::bindDerivatives(0, 1));
92
93 if constexpr (!std::is_same_v<VolumeLoad, utils::LoadDefault>) volumeLoad = p_volumeLoad;
94 if constexpr (!std::is_same_v<NeumannBoundaryLoad, utils::LoadDefault>)
95 neumannBoundaryLoad = p_neumannBoundaryLoad;
96
97 assert(((not p_neumannBoundary and not neumannBoundaryLoad) or (p_neumannBoundary and neumannBoundaryLoad))
98 && "If you pass a Neumann boundary you should also pass the function for the Neumann load!");
99 }

109 template <typename ScalarType = double>
110 auto displacementFunction(const FERequirementType& par,
111 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
112 const auto& d = par.getGlobalSolution(Ikarus::FESolutions::displacement);
113
114 Dune::BlockVector<Dune::RealTuple<ScalarType, Traits::dimension>> disp(dispAtNodes.size());
115 // If optional displacement vector is provided, apply it
116 if (dx)
117 for (auto i = 0U; i < disp.size(); ++i)
118 for (auto k2 = 0U; k2 < myDim; ++k2)
119 disp[i][k2] = dx.value()[i * myDim + k2]
120 + d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
121 else
122 for (auto i = 0U; i < disp.size(); ++i)
123 for (auto k2 = 0U; k2 < myDim; ++k2)
124 disp[i][k2] = d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
125
126 auto geo = std::make_shared<const typename GridView::GridView::template Codim<0>::Entity::Geometry>(
127 this->localView().element().geometry());
128 Dune::StandardLocalFunction uFunction(localBasis, disp, geo);
129 return uFunction;
130 }

229 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
230 using namespace Dune::DerivativeDirections;
231 using namespace Dune;
232 const auto eps = strainFunction(par);
233 const auto geo = this->localView().element().geometry();
234
235 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
236 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
237 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
238 const auto C = getMaterialTangent(EVoigt);
239 const auto stresses = getStress(EVoigt);
240 for (size_t i = 0; i < numberOfNodes; ++i) {
241 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
242 for (size_t j = 0; j < numberOfNodes; ++j) {
243 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
244 const auto kgIJ = eps.evaluateDerivative(gpIndex, wrt(coeff(i, j)), along(stresses), on(gridElement));
245 K.template block<myDim, myDim>(i * myDim, j * myDim) += (bopI.transpose() * C * bopJ + kgIJ) * intElement;
246 }
247 }
248 }

258 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
259 ResultTypeMap<double>& result) const {
260 using namespace Dune::DerivativeDirections;
261 using namespace Dune;
262
263 const auto uFunction = displacementFunction(req.getFERequirements());
264 const auto H = uFunction.evaluateDerivative(local, Dune::wrt(spatialAll), Dune::on(gridElement));
265 const auto E = (0.5 * (H.transpose() + H + H.transpose() * H)).eval();
266 const auto EVoigt = toVoigt(E);
267 auto PK2 = mat.template stresses<StrainTags::greenLagrangian>(EVoigt);
268
269 if (req.isResultRequested(ResultType::PK2Stress))
270 result.insertOrAssignResult(ResultType::PK2Stress, PK2);
271 else
272 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
273 }

291 template <typename ScalarType>
292 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
293 = std::nullopt) const -> ScalarType {
294 using namespace Dune::DerivativeDirections;
295 using namespace Dune;
296 const auto uFunction = displacementFunction(par, dx);
297 const auto eps = strainFunction(par, dx);
298 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
299 const auto geo = this->localView().element().geometry();
300 ScalarType energy = 0.0;
301
302 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
303 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
304 const auto internalEnergy = getInternalEnergy(EVoigt);
305 energy += internalEnergy * geo.integrationElement(gp.position()) * gp.weight();
306 }
307
308 if (volumeLoad) {
309 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
310 const auto u = uFunction.evaluate(gpIndex);
311 const Eigen::Vector<double, Traits::worlddim> fExt = volumeLoad(toEigen(geo.global(gp.position())), lambda);
312 energy -= u.dot(fExt) * geo.integrationElement(gp.position()) * gp.weight();
313 }
314 }
315
316 // line or surface loads, i.e., neumann boundary
317 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
318
319 const auto& element = this->localView().element();
320 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
321 if (not neumannBoundary or not neumannBoundary->contains(intersection)) continue;
322
323 const auto& quadLine = Dune::QuadratureRules<double, Traits::mydim - 1>::rule(intersection.type(), order);
324
325 for (const auto& curQuad : quadLine) {
326 // Local position of the quadrature point
327 const Dune::FieldVector<double, Traits::mydim>& quadPos
328 = intersection.geometryInInside().global(curQuad.position());
329
330 const double intElement = intersection.geometry().integrationElement(curQuad.position());
331
332 // The value of the local function
333 const auto u = uFunction.evaluate(quadPos);
334
335 // Value of the Neumann data at the current position
336 const auto neumannValue
337 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
338 energy -= neumannValue.dot(u) * curQuad.weight() * intElement;
339 }
340 }
341 return energy;
342 }

344 template <typename ScalarType>
345 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
346 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
347 using namespace Dune::DerivativeDirections;
348 using namespace Dune;
349 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
350 const auto eps = strainFunction(par, dx);
351 const auto geo = this->localView().element().geometry();
352
353 // Internal forces
354 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
355 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
356 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
357 const auto stresses = getStress(EVoigt);
358 for (size_t i = 0; i < numberOfNodes; ++i) {
359 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
360 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
361 }
362 }
363
364 // External forces volume forces over the domain
365 if (volumeLoad) {
366 const auto u = displacementFunction(par, dx);
367 for (const auto& [gpIndex, gp] : u.viewOverIntegrationPoints()) {
368 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
369 const Eigen::Vector<double, Traits::worlddim> fExt = volumeLoad(toEigen(geo.global(gp.position())), lambda);
370 for (size_t i = 0; i < numberOfNodes; ++i) {
371 const auto udCi = u.evaluateDerivative(gpIndex, wrt(coeff(i)));
372 force.template segment<myDim>(myDim * i) -= udCi * fExt * intElement;
373 }
374 }
375 }
376
377 // External forces, boundary forces, i.e., at the Neumann boundary
378 if (not neumannBoundary and not neumannBoundaryLoad) return;
379
380 const auto u = displacementFunction(par, dx);
381 const auto& element = this->localView().element();
382 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
383 if (not neumannBoundary->contains(intersection)) continue;
384
385 // Integration rule along the boundary
386 const auto& quadLine = Dune::QuadratureRules<double, myDim - 1>::rule(intersection.type(), order);
387
388 for (const auto& curQuad : quadLine) {
389 const Dune::FieldVector<double, myDim>& quadPos = intersection.geometryInInside().global(curQuad.position());
390
391 const double intElement = intersection.geometry().integrationElement(curQuad.position());
392
393 // The value of the local function wrt the i-th coefficient
394 for (size_t i = 0; i < numberOfNodes; ++i) {
395 const auto udCi = u.evaluateDerivative(quadPos, wrt(coeff(i)));
396
397 // Value of the Neumann data at the current position
398 const auto neumannValue
399 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
400 force.template segment<myDim>(myDim * i) -= udCi * neumannValue * curQuad.weight() * intElement;
401 }
402 }
403 }
404 }
405 };
