29 template <typename Geometry>
30 Eigen::Matrix3d calcTransformationMatrix2D(const Geometry& geometry) {
31 const auto& referenceElement = Dune::ReferenceElements<double, 2>::general(geometry.type());
32 const auto quadPos0 = referenceElement.position(0, 0);
33
34 const auto jacobianinvT0 = toEigen(geometry.jacobianInverseTransposed(quadPos0));
35 const auto detJ0 = geometry.integrationElement(quadPos0);
36
37 auto jaco = (jacobianinvT0).inverse().eval();
38 auto J11 = jaco(0, 0);
39 auto J12 = jaco(0, 1);
40 auto J21 = jaco(1, 0);
41 auto J22 = jaco(1, 1);
42
43 Eigen::Matrix3d T0;
44 // clang-format off
45 T0 << J11 * J11, J12 * J12, J11 * J12,
46 J21 * J21, J22 * J22, J21 * J22,
47 2.0 * J11 * J21, 2.0 * J12 * J22, J21 * J12 + J11 * J22;
48 // clang-format on
49 return T0.inverse() * detJ0;
50 }

61 template <typename Geometry>
62 Eigen::Matrix<double, 6, 6> calcTransformationMatrix3D(const Geometry& geometry) {
63 const auto& referenceElement = Dune::ReferenceElements<double, 3>::general(geometry.type());
64 const auto quadPos0 = referenceElement.position(0, 0);
65
66 const auto jacobianinvT0 = toEigen(geometry.jacobianInverseTransposed(quadPos0));
67 const auto detJ0 = geometry.integrationElement(quadPos0);
68
69 auto jaco = (jacobianinvT0).inverse().eval();
70 auto J11 = jaco(0, 0);
71 auto J12 = jaco(0, 1);
72 auto J13 = jaco(0, 2);
73 auto J21 = jaco(1, 0);
74 auto J22 = jaco(1, 1);
75 auto J23 = jaco(1, 2);
76 auto J31 = jaco(2, 0);
77 auto J32 = jaco(2, 1);
78 auto J33 = jaco(2, 2);
79
80 Eigen::Matrix<double, 6, 6> T0;
81 // clang-format off
82 T0 << J11 * J11, J12 * J12, J13 * J13, J11 * J12, J11 * J13, J12 * J13,
83 J21 * J21, J22 * J22, J23 * J23, J21 * J22, J21 * J23, J22 * J23,
84 J31 * J31, J32 * J32, J33 * J33, J31 * J32, J31 * J33, J32 * J33,
85 2.0 * J11 * J21, 2.0 * J12 * J22, 2.0 * J13 * J23, J11 * J22 + J21 * J12, J11 * J23 + J21 * J13, J12 * J23 + J22 * J13,
86 2.0 * J11 * J31, 2.0 * J12 * J32, 2.0 * J13 * J33, J11 * J32 + J31 * J12, J11 * J33 + J31 * J13, J12 * J33 + J32 * J13,
87 2.0 * J31 * J21, 2.0 * J32 * J22, 2.0 * J33 * J23, J31 * J22 + J21 * J32, J31 * J23 + J21 * J33, J32 * J23 + J22 * J33;
88 // clang-format on
89
90 return T0.inverse() * detJ0;
91 }

102 template <typename Geometry>
103 struct EASQ1E4 {
104 static constexpr int strainSize = 3;
105 static constexpr int enhancedStrainSize = 4;
106 using MType = Eigen::Matrix<double, strainSize, enhancedStrainSize>;
107
108 EASQ1E4() = default;
109 explicit EASQ1E4(const Geometry& geometry_)
110 : geometry{std::make_shared<Geometry>(geometry_)},
111 T0InverseTransformed{calcTransformationMatrix2D(geometry_)} {}
112
113 auto calcM(const Dune::FieldVector<double, 2>& quadPos) const {
114 MType M;
115 M.setZero(strainSize, enhancedStrainSize);
116 const double xi = quadPos[0];
117 const double eta = quadPos[1];
118 M(0, 0) = 2 * xi - 1.0;
119 M(1, 1) = 2 * eta - 1.0;
120 M(2, 2) = 2 * xi - 1.0;
121 M(2, 3) = 2 * eta - 1.0;
122 const double detJ = geometry->integrationElement(quadPos);
123 M = T0InverseTransformed / detJ * M;
124 return M;
125 }
126
127 std::shared_ptr<Geometry> geometry;
128 Eigen::Matrix3d T0InverseTransformed;
129 };

460 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
461 ResultTypeMap<double>& result) const {
462 using namespace Dune::Indices;
463 using namespace Dune::DerivativeDirections;
464 using namespace Dune;
465
466 DisplacementBasedElement::calculateAt(req, local, result);
467
468 if (isDisplacementBased()) return;
469
470 const auto& d = req.getGlobalSolution(Ikarus::FESolutions::displacement);
471 const auto C = DisplacementBasedElement::getMaterialTangentFunction(req.getFERequirements());
472 const auto& numNodes = DisplacementBasedElement::numberOfNodes;
473
474 Eigen::VectorXd disp(localView().size());
475 for (auto i = 0U; i < numNodes; ++i)
476 for (auto k2 = 0U; k2 < Traits::mydim; ++k2)
477 disp[i * Traits::mydim + k2] = d[localView().index(localView().tree().child(k2).localIndex(i))[0]];
478
479 std::visit(
480 [&]<typename EAST>(const EAST& easFunction) {
481 if constexpr (not std::is_same_v<std::monostate, EAST>) {
482 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
483 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
484 calculateDAndLMatrix(easFunction, req.getFERequirements(), D, L);
485 const auto alpha = (-D.inverse() * L * disp).eval();
486 const auto M = easFunction.calcM(local);
487 const auto CEval = C(local);
488 auto easStress = (CEval * M * alpha).eval();
489 typename ResultTypeMap<double>::ResultArray resultVector;
490 if (req.isResultRequested(ResultType::linearStress)) {
491 resultVector.resize(3, 1);
492 resultVector = result.getResult(ResultType::linearStress) + easStress;
493 result.insertOrAssignResult(ResultType::linearStress, resultVector);
494 } else
495 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
496 }
497 },
498 easVariant_);
499 }

558 template <typename ScalarType>
559 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
560 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
561 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
562 if (isDisplacementBased()) return;
563 using namespace Dune;
564 const auto& d = par.getGlobalSolution(Ikarus::FESolutions::displacement);
565 auto strainFunction = DisplacementBasedElement::getStrainFunction(par, dx);
566 Eigen::VectorX<ScalarType> disp(localView().size());
567 const auto& numNodes = DisplacementBasedElement::numberOfNodes;
568
569 // FIXME this should not be needed in the future strainFunction should be able to hand out this vector
570 if (dx)
571 for (auto i = 0U; i < numNodes; ++i)
572 for (auto k2 = 0U; k2 < Traits::mydim; ++k2)
573 disp[i * Traits::mydim + k2] = dx.value()[i * Traits::mydim + k2]
574 + d[localView().index(localView().tree().child(k2).localIndex(i))[0]];
575 else
576 for (auto i = 0U; i < numNodes; ++i)
577 for (auto k2 = 0U; k2 < Traits::mydim; ++k2)
578 disp[i * Traits::mydim + k2] = d[localView().index(localView().tree().child(k2).localIndex(i))[0]];
579
580 using namespace Dune::DerivativeDirections;
581
582 auto C = DisplacementBasedElement::getMaterialTangentFunction(par);
583 const auto geo = localView().element().geometry();
584
585 // Internal forces from enhanced strains
586 std::visit(
587 [&]<typename EAST>(const EAST& easFunction) {
588 if constexpr (not std::is_same_v<std::monostate, EAST>) {
589 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
590 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
591 calculateDAndLMatrix(easFunction, par, D, L);
592
593 const auto alpha = (-D.inverse() * L * disp).eval();
594
595 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
596 const auto M = easFunction.calcM(gp.position());
597 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
598 const auto CEval = C(gpIndex);
599 auto stresses = (CEval * M * alpha).eval();
600 for (size_t i = 0; i < numNodes; ++i) {
601 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
602 force.template segment<Traits::mydim>(Traits::mydim * i) += bopI.transpose() * stresses * intElement;
603 }
604 }
605 }
606 },
607 easVariant_);
608 }

615 template <int enhancedStrainSize>
616 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
617 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
618 Eigen::MatrixXd& LMat) const {
619 using namespace Dune;
620 using namespace Dune::DerivativeDirections;
621
622 auto strainFunction = DisplacementBasedElement::getStrainFunction(par);
623 const auto C = DisplacementBasedElement::getMaterialTangentFunction(par);
624 const auto geo = localView().element().geometry();
625 const auto numNodes = DisplacementBasedElement::numberOfNodes;
626 DMat.setZero();
627 LMat.setZero(enhancedStrainSize, localView().size());
628 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
629 const auto M = easFunction.calcM(gp.position());
630 const auto CEval = C(gpIndex);
631 const double detJ = geo.integrationElement(gp.position());
632 DMat += M.transpose() * CEval * M * detJ * gp.weight();
633 for (size_t i = 0U; i < numNodes; ++i) {
634 const size_t I = Traits::worlddim * i;
635 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
636 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I)
637 += M.transpose() * CEval * Bi * detJ * gp.weight();
638 }
639 }
640 }
