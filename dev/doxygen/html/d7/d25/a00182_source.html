17 template <typename NonLinearSolver>
18 class LoadControl : public IObservable<ControlMessages> {
19 public:
20 static constexpr std::string_view name_ = "Load Control Method";
21
22 LoadControl(const std::shared_ptr<NonLinearSolver>& p_nonLinearSolver, int loadSteps,
23 const std::array<double, 2>& tbeginEnd)
24 : nonLinearSolver{p_nonLinearSolver},
25 loadSteps_{loadSteps},
26 parameterBegin_{tbeginEnd[0]},
27 parameterEnd_{tbeginEnd[1]},
28 stepSize_{(parameterEnd_ - parameterBegin_) / loadSteps_} {
29 static_assert(
30 requires {
31 nonLinearSolver->nonLinearOperator().lastParameter() = 0.0;
32 nonLinearSolver->nonLinearOperator().lastParameter() += 0.0;
33 },
34 "The last parameter (load factor) must be assignable and incrementable with a double!");
35 }
36
37 ControlInformation run() {
38 ControlInformation info({false});
39 auto& nonOp = nonLinearSolver->nonLinearOperator();
40 this->notify(ControlMessages::CONTROL_STARTED, static_cast<std::string>(name_));
41 auto& loadParameter = nonOp.lastParameter();
42
43 loadParameter = 0.0;
44 this->notify(ControlMessages::STEP_STARTED, 0, stepSize_);
45 auto solverInfo = nonLinearSolver->solve();
46 info.solverInfos.push_back(solverInfo);
47 info.totalIterations += solverInfo.iterations;
48 if (not solverInfo.success) return info;
49 this->notify(ControlMessages::SOLUTION_CHANGED);
50 this->notify(ControlMessages::STEP_ENDED);
51
52 for (int ls = 0; ls < loadSteps_; ++ls) {
53 this->notify(ControlMessages::STEP_STARTED, ls, stepSize_);
54 loadParameter += stepSize_;
55 solverInfo = nonLinearSolver->solve();
56 info.solverInfos.push_back(solverInfo);
57 info.totalIterations += solverInfo.iterations;
58 if (not solverInfo.success) return info;
59 this->notify(ControlMessages::SOLUTION_CHANGED);
60 this->notify(ControlMessages::STEP_ENDED);
61 }
62 this->notify(ControlMessages::CONTROL_ENDED, info.totalIterations, static_cast<std::string>(name_));
63 info.success = true;
64 return info;
65 }
66
67 private:
68 std::shared_ptr<NonLinearSolver> nonLinearSolver;
69 int loadSteps_;
70 double parameterBegin_;
71 double parameterEnd_;
72 double stepSize_;
73 };
