153template <typename SV = std::reference_wrapper<Eigen::VectorXd>, typename PM = std::reference_wrapper<double>>
154class FERequirements
155{
156public:
157 using SolutionVectorType = SV;
158 using SolutionVectorTypeRaw = typename Impl::DeduceRawVectorType<std::remove_cvref_t<SV>>::Type;
159 using ParameterType = PM;
160 using ParameterTypeRaw = typename PM::type;
161
171 template <FEAffordance Affordance>
172 FERequirements& addAffordance(Affordance&& affordance) {
173 if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, ScalarAffordances>)
174 affordances_.scalarAffordances = affordance;
175 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, VectorAffordances>)
176 affordances_.vectorAffordances = affordance;
177 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, MatrixAffordances>)
178 affordances_.matrixAffordances = affordance;
179 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, AffordanceCollectionImpl>)
180 affordances_ = affordance;
181 return *this;
182 }
183
193 FERequirements& insertParameter(const FEParameter& key, ParameterTypeRaw& val) {
194 parameter_.insert_or_assign(key, val);
195 return *this;
196 }
197
207 FERequirements& insertGlobalSolution(const FESolutions& key, SolutionVectorTypeRaw& sol) {
208 sols_.insert_or_assign(key, sol);
209 return *this;
210 }
211
222 const SolutionVectorTypeRaw& getGlobalSolution(const FESolutions& key) const {
223 try {
224 if constexpr (std::is_same_v<SolutionVectorType, std::reference_wrapper<Eigen::VectorXd>>)
225 return sols_.at(key).get();
226 else
227 return sols_.at(key);
228 } catch (std::out_of_range& oor) {
229 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
230 " in getGlobalSolution with key" + toString(key));
231 abort();
232 }
233 }
234
245 const ParameterTypeRaw& getParameter(FEParameter&& key) const {
246 try {
247 return parameter_.at(key).get();
248 } catch (std::out_of_range& oor) {
249 DUNE_THROW(Dune::RangeError, std::string("Out of Range error: ") + std::string(oor.what()) +
250 " in getParameter with key" + toString(key));
251 abort();
252 }
253 }
254
264 template <FEAffordance Affordance>
265 bool hasAffordance(Affordance&& affordance) const {
266 if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, ScalarAffordances>)
267 return affordances_.scalarAffordances == affordance;
268 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, VectorAffordances>)
269 return affordances_.vectorAffordances == affordance;
270 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, MatrixAffordances>)
271 return affordances_.matrixAffordances == affordance;
272 else if constexpr (std::is_same_v<std::remove_cvref_t<Affordance>, AffordanceCollectionImpl>)
273 return affordances_ == affordance;
274 }
275
276private:
277 std::map<FESolutions, SolutionVectorType> sols_;
278 std::map<FEParameter, ParameterType> parameter_;
279 AffordanceCollectionImpl affordances_;
280};
