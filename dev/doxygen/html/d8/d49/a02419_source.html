70 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
71 LinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, double emod, double nu,
72 VolumeLoad p_volumeLoad = {}, const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
73 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
74 : BaseDisp(globalBasis.flat(), element), neumannBoundary{p_neumannBoundary}, emod_{emod}, nu_{nu} {
75 this->localView().bind(element);
76 auto& first_child = this->localView().tree().child(0);
77 const auto& fe = first_child.finiteElement();
78 numberOfNodes = fe.size();
79 dispAtNodes.resize(numberOfNodes);
80 order = 2 * (this->localView().tree().child(0).finiteElement().localBasis().order());
81 localBasis = Dune::CachedLocalBasis(this->localView().tree().child(0).finiteElement().localBasis());
82 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order); })
83 if (this->localView().element().impl().isTrimmed())
84 localBasis.bind(this->localView().element().impl().getQuadratureRule(order), Dune::bindDerivatives(0, 1));
85 else
86 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
87 Dune::bindDerivatives(0, 1));
88 else
89 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
90 Dune::bindDerivatives(0, 1));
91
92 if constexpr (!std::is_same_v<VolumeLoad, utils::LoadDefault>) volumeLoad = p_volumeLoad;
93 if constexpr (!std::is_same_v<NeumannBoundaryLoad, utils::LoadDefault>)
94 neumannBoundaryLoad = p_neumannBoundaryLoad;
95
96 assert(((not p_neumannBoundary and not neumannBoundaryLoad) or (p_neumannBoundary and neumannBoundaryLoad))
97 && "If you pass a Neumann boundary you should also pass the function for the Neumann load!");
98 }
107 template <typename ScalarType>
108 auto getDisplacementFunction(const FERequirementType& par,
109 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
110 const auto& d = par.getGlobalSolution(Ikarus::FESolutions::displacement);
111
112 Dune::BlockVector<Dune::RealTuple<ScalarType, Traits::dimension>> disp(dispAtNodes.size());
113 if (dx) {
114 for (auto i = 0U; i < disp.size(); ++i)
115 for (auto k2 = 0U; k2 < myDim; ++k2)
116 disp[i][k2] = dx.value()[i * myDim + k2]
117 + d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
118 } else
119 for (auto i = 0U; i < disp.size(); ++i)
120 for (auto k2 = 0U; k2 < myDim; ++k2)
121 disp[i][k2] = d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
122
123 auto geo = std::make_shared<const typename GridView::GridView::template Codim<0>::Entity::Geometry>(
124 this->localView().element().geometry());
125 Dune::StandardLocalFunction uFunction(localBasis, disp, geo);
126
127 return uFunction;
128 }

189 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
190 const auto eps = getStrainFunction(par);
191 using namespace Dune::DerivativeDirections;
192 using namespace Dune;
193
194 const auto C = getMaterialTangent();
195 const auto geo = this->localView().element().geometry();
196 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
197 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
198 for (size_t i = 0; i < numberOfNodes; ++i) {
199 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
200 for (size_t j = 0; j < numberOfNodes; ++j) {
201 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
202 K.template block<myDim, myDim>(i * myDim, j * myDim) += bopI.transpose() * C * bopJ * intElement;
203 }
204 }
205 }

215 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
216 ResultTypeMap<double>& result) const {
217 using namespace Dune::Indices;
218 using namespace Dune::DerivativeDirections;
219 using namespace Dune;
220
221 const auto eps = getStrainFunction(req.getFERequirements());
222 const auto C = getMaterialTangent();
223 auto epsVoigt = eps.evaluate(local, on(gridElement));
224
225 auto linearStress = (C * epsVoigt).eval();
226
227 if (req.isResultRequested(ResultType::linearStress))
228 result.insertOrAssignResult(ResultType::linearStress, linearStress);
229 else
230 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
231 }

250 template <typename ScalarType>
251 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
252 = std::nullopt) const -> ScalarType {
253 const auto u = getDisplacementFunction(par, dx);
254 const auto eps = getStrainFunction(par, dx);
255 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
256 using namespace Dune::DerivativeDirections;
257 using namespace Dune;
258
259 const auto C = getMaterialTangent();
260
261 const auto geo = this->localView().element().geometry();
262 ScalarType energy = 0.0;
263 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
264 const auto EVoigt = eps.evaluate(gpIndex, on(gridElement));
265
266 energy += (0.5 * EVoigt.dot(C * EVoigt)) * geo.integrationElement(gp.position()) * gp.weight();
267 }
268
269 // External forces volume forces over the domain
270 if (volumeLoad) {
271 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
272 const auto uVal = u.evaluate(gpIndex);
273 Eigen::Vector<double, Traits::worlddim> fext = volumeLoad(toEigen(geo.global(gp.position())), lambda);
274 energy -= uVal.dot(fext) * geo.integrationElement(gp.position()) * gp.weight();
275 }
276 }
277
278 // line or surface loads, i.e., neumann boundary
279 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
280
281 auto element = this->localView().element();
282 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
283 if (not neumannBoundary->contains(intersection)) continue;
284
285 const auto& quadLine = Dune::QuadratureRules<double, myDim - 1>::rule(intersection.type(), order);
286
287 for (const auto& curQuad : quadLine) {
288 // Local position of the quadrature point
289 const Dune::FieldVector<double, myDim>& quadPos = intersection.geometryInInside().global(curQuad.position());
290
291 const double integrationElement = intersection.geometry().integrationElement(curQuad.position());
292
293 // The value of the local function
294 const auto uVal = u.evaluate(quadPos);
295
296 // Value of the Neumann data at the current position
297 auto neumannValue = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
298
299 energy -= neumannValue.dot(uVal) * curQuad.weight() * integrationElement;
300 }
301 }
302 return energy;
303 }

305 template <typename ScalarType>
306 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
307 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
308 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
309 const auto eps = getStrainFunction(par, dx);
310 using namespace Dune::DerivativeDirections;
311 using namespace Dune;
312
313 const auto C = getMaterialTangent();
314 const auto geo = this->localView().element().geometry();
315
316 // Internal forces
317 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
318 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
319 auto stresses = (C * eps.evaluate(gpIndex, on(gridElement))).eval();
320 for (size_t i = 0; i < numberOfNodes; ++i) {
321 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
322 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
323 }
324 }
325
326 // External forces volume forces over the domain
327 if (volumeLoad) {
328 const auto u = getDisplacementFunction(par, dx);
329 for (const auto& [gpIndex, gp] : u.viewOverIntegrationPoints()) {
330 Eigen::Vector<double, Traits::worlddim> fext = volumeLoad(toEigen(geo.global(gp.position())), lambda);
331 for (size_t i = 0; i < numberOfNodes; ++i) {
332 const auto udCi = u.evaluateDerivative(gpIndex, wrt(coeff(i)));
333 force.template segment<myDim>(myDim * i)
334 -= udCi * fext * geo.integrationElement(gp.position()) * gp.weight();
335 }
336 }
337 }
338
339 // External forces, boundary forces, i.e. at the Neumann boundary
340 if (not neumannBoundary) return;
341
342 const auto u = getDisplacementFunction(par, dx);
343 auto element = this->localView().element();
344 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
345 if (not neumannBoundary->contains(intersection)) continue;
346
347 // Integration rule along the boundary
348 const auto& quadLine = Dune::QuadratureRules<double, myDim - 1>::rule(intersection.type(), order);
349
350 for (const auto& curQuad : quadLine) {
351 const Dune::FieldVector<double, myDim>& quadPos = intersection.geometryInInside().global(curQuad.position());
352
353 const double integrationElement = intersection.geometry().integrationElement(curQuad.position());
354
355 // The value of the local function wrt the i-th coef
356 for (size_t i = 0; i < numberOfNodes; ++i) {
357 const auto udCi = u.evaluateDerivative(quadPos, wrt(coeff(i)));
358
359 // Value of the Neumann data at the current position
360 auto neumannValue
361 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
362 force.template segment<myDim>(myDim * i) -= udCi * neumannValue * curQuad.weight() * integrationElement;
363 }
364 }
365 }
366 }
367 };
