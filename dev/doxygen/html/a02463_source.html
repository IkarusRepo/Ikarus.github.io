80 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
81 LinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, double emod, double nu,
82 VolumeLoad p_volumeLoad = {}, const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
83 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
84 : BaseDisp(globalBasis, element),
85 VolumeType(p_volumeLoad),
86 TractionType(p_neumannBoundary, p_neumannBoundaryLoad),
87 emod_{emod},
88 nu_{nu} {
89 this->localView().bind(element);
90 auto& first_child = this->localView().tree().child(0);
91 const auto& fe = first_child.finiteElement();
92 geo_ = std::make_shared<const Geometry>(this->localView().element().geometry());
93 numberOfNodes_ = fe.size();
94 order_ = 2 * (fe.localBasis().order());
95 localBasis = Dune::CachedLocalBasis(fe.localBasis());
96 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order_); })
97 if (this->localView().element().impl().isTrimmed())
98 localBasis.bind(this->localView().element().impl().getQuadratureRule(order_), Dune::bindDerivatives(0, 1));
99 else
100 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
101 Dune::bindDerivatives(0, 1));
102 else
103 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
104 Dune::bindDerivatives(0, 1));

186 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
187 const auto eps = strainFunction(par);
188 using namespace Dune::DerivativeDirections;
189 using namespace Dune;
190
191 const auto C = materialTangent();
192 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
193 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
194 for (size_t i = 0; i < numberOfNodes_; ++i) {
195 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
196 for (size_t j = 0; j < numberOfNodes_; ++j) {
197 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
198 K.template block<myDim, myDim>(i * myDim, j * myDim) += bopI.transpose() * C * bopJ * intElement;
199 }
200 }
201 }
202
203 // Update due to displacement-dependent external loads, e.g., follower loads
204 VolumeType::calculateMatrix(par, K);
205 TractionType::calculateMatrix(par, K);
206 }

217 template <ResultType resType>
218 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
219 static_assert(resType == ResultType::linearStress, "The requested result type is NOT implemented.");
220
221 if constexpr (resType == ResultType::linearStress) {
222 const auto eps = strainFunction(req);
223 const auto C = materialTangent();
224 auto epsVoigt = eps.evaluate(local, Dune::on(Dune::DerivativeDirections::gridElement));
225
226 return (C * epsVoigt).eval();
227 }
228 }

239 template <typename ScalarType>
240 auto calculateScalarImpl(const FERequirementType& par,
241 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const
242 -> ScalarType {
243 const auto uFunction = displacementFunction(par, dx);
244 const auto eps = strainFunction(par, dx);
245 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
246 using namespace Dune::DerivativeDirections;
247 using namespace Dune;
248
249 const auto C = materialTangent();
250
251 ScalarType energy = 0.0;
252 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
253 const auto EVoigt = eps.evaluate(gpIndex, on(gridElement));
254 energy += (0.5 * EVoigt.dot(C * EVoigt)) * geo_->integrationElement(gp.position()) * gp.weight();
255 }
256
257 // External forces volume forces over the domain
258 energy += VolumeType::calculateScalarImpl(par, dx);
259
260 // line or surface loads, i.e., neumann boundary
261 energy += TractionType::calculateScalarImpl(par, dx);
262 return energy;
263 }

265 template <typename ScalarType>
266 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
267 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
268 const auto eps = strainFunction(par, dx);
269 using namespace Dune::DerivativeDirections;
270 using namespace Dune;
271
272 const auto C = materialTangent();
273
274 // Internal forces
275 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
276 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
277 auto stresses = (C * eps.evaluate(gpIndex, on(gridElement))).eval();
278 for (size_t i = 0; i < numberOfNodes_; ++i) {
279 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
280 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
281 }
282 }
283
284 // External forces volume forces over the domain
285 VolumeType::calculateVectorImpl(par, force, dx);
286
287 // External forces, boundary forces, i.e., at the Neumann boundary
288 TractionType::calculateVectorImpl(par, force, dx);
289 }
