31 // this factor denotes the differences between the returned stresses and moduli and the passed strain
32 // for neoHooke the inserted quantity is C the right Cauchy-Green tensor,
33 // the function relation between the energy and the stresses is S = 2*\partial \psi(C)/ \partial C.
34 // This factor is pre factor, which is the difference to the actual derivative is written here
35 static constexpr double derivativeFactor = 2;

37 template <typename Derived>
38 ScalarType storedEnergyImpl(const Eigen::MatrixBase<Derived>& C) const noexcept {
39 static_assert(Concepts::EigenMatrixOrVoigtNotation3<Derived>);
40 if constexpr (!Concepts::EigenVector<Derived>) {
41 const auto traceC = C.trace();
42 const auto logdetF = log(sqrt(C.determinant()));
43 return lambdaAndmu.mu / 2.0 * (traceC - 3 - 2 * logdetF) + lambdaAndmu.lambda / 2.0 * logdetF * logdetF;
44 } else
45 static_assert(!Concepts::EigenVector<Derived>,
46 "NeoHooke energy can only be called with matrix and not a vector in Voigt notation");
47 }
48
49 template <bool voigt, typename Derived>
50 auto stressesImpl(const Eigen::MatrixBase<Derived>& C) const {
51 static_assert(Concepts::EigenMatrixOrVoigtNotation3<Derived>);
52 if constexpr (!voigt) {
53 if constexpr (!Concepts::EigenVector<Derived>) {
54 const auto logdetF = log(sqrt(C.determinant()));
55 const auto invC = C.inverse().eval();
56 return (lambdaAndmu.mu * (StrainMatrix::Identity() - invC) + lambdaAndmu.lambda * logdetF * invC).eval();
57 } else
58 static_assert(!Concepts::EigenVector<Derived>,
59 "NeoHooke can only be called with matrix and not a vector in Voigt notation");
60 } else
61 static_assert(voigt == false, "NeoHooke does not support returning stresses in Voigt notation");
62 }
63
64 template <bool voigt, typename Derived>
65 auto tangentModuliImpl(const Eigen::MatrixBase<Derived>& C) const {
66 static_assert(Concepts::EigenMatrixOrVoigtNotation3<Derived>);
67 if constexpr (!voigt) {
68 const auto invC = C.inverse().eval();
69 const auto logdetF = log(sqrt(C.determinant()));
70 const auto CTinv = TensorCast(invC, std::array<Eigen::Index, 2>({3, 3}));
71 static_assert(Eigen::TensorFixedSize<ScalarType, Eigen::Sizes<3, 3>>::NumIndices == 2);
72 // see Javier Bonet, Richard D. Wood - Nonlinear Continuum Mechanics for Finite Element Analysis, 2nd Edition
73 // (2008) Eq. 6.30
74 // (first edition has an error)
75 Eigen::TensorFixedSize<ScalarType, Eigen::Sizes<3, 3, 3, 3>> moduli
76 = (lambdaAndmu.lambda * dyadic(CTinv, CTinv)
77 + 2 * (lambdaAndmu.mu - lambdaAndmu.lambda * logdetF) * symTwoSlots(fourthOrderIKJL(invC, invC), {2, 3}))
78 .eval();
79 return moduli;
80 } else
81 static_assert(voigt == false, "NeoHooke does not support returning tangentModuli in Voigt notation");
82 }

84 template <typename ScalarTypeOther>
85 auto rebind() const {
86 return NeoHookeT<ScalarTypeOther>(lambdaAndmu);
87 }
