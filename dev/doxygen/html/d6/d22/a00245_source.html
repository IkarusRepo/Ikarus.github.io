74 void operator()(SubsidiaryArgs& args) const {
75 if (psi) {
76 const auto root = sqrt(args.DD.squaredNorm() + psi.value() * psi.value() * args.Dlambda * args.Dlambda);
77 args.f = root - args.stepSize;
78 args.dfdDD = args.DD / root;
79 args.dfdDlambda = (psi.value() * psi.value() * args.Dlambda) / root;

82 "You have to call initialPrediction first. Otherwise psi is not defined");
83 }

97 template <typename NonLinearOperator>
98 void initialPrediction(NonLinearOperator& nonLinearOperator, SubsidiaryArgs& args) {
99 SolverTypeTag solverTag;
100 using JacobianType = std::remove_cvref_t<typename NonLinearOperator::DerivativeType>;
101 static_assert((traits::isSpecializationTypeAndNonTypes<Eigen::Matrix, JacobianType>::value)
102 or (traits::isSpecializationTypeNonTypeAndType<Eigen::SparseMatrix, JacobianType>::value),
103 "Linear solver not implemented for the chosen derivative type of the non-linear operator");
104
105 if constexpr (traits::isSpecializationTypeAndNonTypes<Eigen::Matrix, JacobianType>::value)
106 solverTag = SolverTypeTag::d_LDLT;
107 else
108 solverTag = SolverTypeTag::sd_SimplicialLDLT;
109
110 nonLinearOperator.lastParameter() = 1.0; // lambda =1.0
111 nonLinearOperator.template update<0>();
112 const auto& R = nonLinearOperator.value();
113 const auto& K = nonLinearOperator.derivative();
114
115 static constexpr bool isLinearSolver
116 = Ikarus::Concepts::LinearSolverCheck<decltype(LinearSolver(solverTag)),
117 typename NonLinearOperator::DerivativeType,
118 typename NonLinearOperator::ValueType>;
119 static_assert(isLinearSolver,
120 "Initial predictor step in the standard arc-length method doesn't have a linear solver");
121
122 auto linearSolver = LinearSolver(solverTag); // for the linear predictor step
123 linearSolver.analyzePattern(K);
124 linearSolver.factorize(K);
125 linearSolver.solve(args.DD, -R);
126
127 const auto DD2 = args.DD.squaredNorm();
128
129 psi = sqrt(DD2);
130 auto s = sqrt(psi.value() * psi.value() + DD2);
131
132 args.DD = args.DD * args.stepSize / s;
133 args.Dlambda = args.stepSize / s;
134
135 nonLinearOperator.firstParameter() = args.DD;
136 nonLinearOperator.lastParameter() = args.Dlambda;
137 }

151 nonLinearOperator.lastParameter() += args.Dlambda;
152 }

155 constexpr auto name() const { return std::string("Arc length"); }

157 private:
158 std::optional<double> psi;

232 struct DisplacementControl {

239 : controlledIndices{std::move(p_controlledIndices)} {}

248 void operator()(SubsidiaryArgs& args) const {
249 const auto controlledDOFsNorm = args.DD(controlledIndices).norm();
250 args.f = controlledDOFsNorm - args.stepSize;
