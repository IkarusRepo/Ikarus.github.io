163 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
164 ResultTypeMap<double>& result) const {
165 using namespace Dune::Indices;
166 using namespace Dune::DerivativeDirections;
167 using namespace Dune;
168
169 DisplacementBasedElement::calculateAt(req, local, result);
170
171 if (isDisplacementBased())
172 return;
173
174 const auto& par = req.getFERequirements();
175 const auto C = DisplacementBasedElement::materialTangentFunction(req.getFERequirements());
176 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
177 auto uFunction = DisplacementBasedElement::displacementFunction(par);
178 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
179
180 std::visit(
181 [&]<typename EAST>(const EAST& easFunction) {
182 if constexpr (not std::is_same_v<std::monostate, EAST>) {
183 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
184 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
185 calculateDAndLMatrix(easFunction, req.getFERequirements(), D, L);
186 const auto alpha = (-D.inverse() * L * disp).eval();
187 const auto M = easFunction.calcM(local);
188 const auto CEval = C(local);
189 auto easStress = (CEval * M * alpha).eval();
190 typename ResultTypeMap<double>::ResultArray resultVector;
191 if (req.isResultRequested(ResultType::linearStress)) {
192 resultVector.resize(3, 1);
193 resultVector = result.getResult(ResultType::linearStress) + easStress;
194 result.insertOrAssignResult(ResultType::linearStress, resultVector);
195 } else
196 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
197 }
198 },
199 easVariant_);
200 }

207 void setEASType(int numberOfEASParameters_) {
208 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
209 if (not((numberOfNodes == 4 and Traits::mydim == 2) or (numberOfNodes == 8 and Traits::mydim == 3)) and
210 (not isDisplacementBased()))
211 DUNE_THROW(Dune::NotImplemented, "EAS only supported for Q1 or H1 elements");
212
213 if constexpr (Traits::mydim == 2) {
214 switch (numberOfEASParameters_) {
215 case 0:
216 easVariant_ = std::monostate();
217 break;
218 case 4:
219 easVariant_ = EAS::Q1E4(localView().element().geometry());
220 break;

233 case 0:

259 template <typename ScalarType>
260 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
261 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
262 DisplacementBasedElement::calculateVectorImpl(par, force, dx);
263 if (isDisplacementBased())
264 return;
265 using namespace Dune;
266 const auto uFunction = DisplacementBasedElement::displacementFunction(par, dx);
267 auto strainFunction = DisplacementBasedElement::strainFunction(par, dx);
268 const auto& numberOfNodes = DisplacementBasedElement::numberOfNodes();
269 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
270
271 using namespace Dune::DerivativeDirections;
272
273 auto C = DisplacementBasedElement::materialTangentFunction(par);
274 const auto geo = localView().element().geometry();
275
276 // Internal forces from enhanced strains
277 std::visit(
278 [&]<typename EAST>(const EAST& easFunction) {
279 if constexpr (not std::is_same_v<std::monostate, EAST>) {
280 constexpr int enhancedStrainSize = EAST::enhancedStrainSize;
281 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize> D;
282 calculateDAndLMatrix(easFunction, par, D, L);
283
284 const auto alpha = (-D.inverse() * L * disp).eval();
285
286 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
287 const auto M = easFunction.calcM(gp.position());
288 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
289 const auto CEval = C(gpIndex);
290 auto stresses = (CEval * M * alpha).eval();
291 for (size_t i = 0; i < numberOfNodes; ++i) {
292 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
293 force.template segment<Traits::worlddim>(Traits::worlddim * i) +=
294 bopI.transpose() * stresses * intElement;
295 }
296 }
297 }
298 },
299 easVariant_);
300 }

306 template <int enhancedStrainSize>
307 void calculateDAndLMatrix(const auto& easFunction, const auto& par,
308 Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
309 Eigen::MatrixXd& LMat) const {
310 using namespace Dune;
311 using namespace Dune::DerivativeDirections;
312
313 auto strainFunction = DisplacementBasedElement::strainFunction(par);
314 const auto C = DisplacementBasedElement::materialTangentFunction(par);
315 const auto geo = localView().element().geometry();
316 const auto numberOfNodes = DisplacementBasedElement::numberOfNodes();
317 DMat.setZero();
318 LMat.setZero(enhancedStrainSize, localView().size());
319 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
320 const auto M = easFunction.calcM(gp.position());
321 const auto CEval = C(gpIndex);
322 const double detJ = geo.integrationElement(gp.position());
323 DMat += M.transpose() * CEval * M * detJ * gp.weight();
324 for (size_t i = 0U; i < numberOfNodes; ++i) {
325 const size_t I = Traits::worlddim * i;
326 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
327 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) +=
328 M.transpose() * CEval * Bi * detJ * gp.weight();
329 }
330 }
331 }
