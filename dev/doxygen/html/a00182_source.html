47 void calculateMatrix(const FERequirementType& req, typename Traits::template MatrixType<> h) const {
48 // real element implements calculateMatrix by itself, then we simply forward the call
49 if constexpr (requires { RealFE::calculateMatrix(req, h); } and not forceAutoDiff) {
50 RealFE::calculateMatrix(req, h);
51 } else if constexpr (requires {
52 this->template calculateVectorImpl<autodiff::dual>(
53 req, std::declval<typename Traits::template VectorType<autodiff::dual>>(),
54 std::declval<const Eigen::VectorXdual&>());
55 }) {
56 // real element implements calculateVector by itself, therefore we only need first order derivatives
57 Eigen::VectorXdual dx(this->localView().size());
58 Eigen::VectorXdual g(this->localView().size());
59 dx.setZero();
60 auto f = [&](auto& x) -> auto& {
61 g.setZero();
62 this->template calculateVectorImpl<autodiff::dual>(req, g, x);
63 return g;
64 };
65 jacobian(f, autodiff::wrt(dx), at(dx), g, h);
66 } else if constexpr (requires {
67 this->template calculateScalarImpl<autodiff::dual2nd>(
68 req, std::declval<typename Traits::template VectorType<autodiff::dual2nd>>());
69 }) {
70 // real element implements calculateScalar by itself, therefore we need second order derivatives
71 Eigen::VectorXdual2nd dx(this->localView().size());
72 Eigen::VectorXd g;
73 autodiff::dual2nd e;
74 dx.setZero();
75 auto f = [&](auto& x) { return this->template calculateScalarImpl<autodiff::dual2nd>(req, x); };
76 hessian(f, autodiff::wrt(dx), at(dx), e, g, h);
77 } else
78 static_assert(Dune::AlwaysFalse<AutoDiffFE>::value,
79 "Appropriate calculateScalarImpl or calculateVectorImpl functions are not implemented for the "
80 "chosen element.");
81 }

89 void calculateVector(const FERequirementType& req, typename Traits::template VectorType<> g) const {
90 // real element implements calculateVector by itself, then we simply forward the call
91 if constexpr (requires {
92 this->template calculateVectorImpl<double>(
93 req, std::declval<typename Traits::template VectorType<double>>(),
94 std::declval<const Eigen::VectorXd&>());
95 } and not forceAutoDiff) {
96 return this->template calculateVectorImpl<double>(req, g);
97 } else if constexpr (requires {
98 this->template calculateScalarImpl<autodiff::dual>(
99 req, std::declval<const Eigen::VectorXdual&>());
100 }) {
101 // real element implements calculateScalar by itself, therefore we need first order derivatives
102 Eigen::VectorXdual dx(this->localView().size());
103 dx.setZero();
104 autodiff::dual e;
105 auto f = [&](auto& x) { return this->template calculateScalarImpl<autodiff::dual>(req, x); };
106 gradient(f, autodiff::wrt(dx), at(dx), e, g);
107 } else
108 static_assert(Dune::AlwaysFalse<AutoDiffFE>::value,
109 "Appropriate calculateScalarImpl function is not implemented for the "
110 "chosen element.");
111 }
112
120 void calculateLocalSystem(const FERequirementType& req, typename Traits::template MatrixType<> h,
121 typename Traits::template VectorType<> g) const {
122 Eigen::VectorXdual2nd dx(this->localView().size());
123 dx.setZero();
124 auto f = [&](auto& x) { return this->calculateScalarImpl(req, x); };
125 hessian(f, autodiff::wrt(dx), at(dx), g, h);
126 }

134 [[nodiscard]] double calculateScalar(const FERequirementType& par) const {
135 // real element implements calculateScalar by itself, then we simply forward the call
136 if constexpr (requires { RealFE::calculateScalar(par); }) {
137 return RealFE::calculateScalar(par);
138 } else if constexpr (requires { this->calculateScalarImpl(par); }) {
139 // real element only implements the protected calculateScalarImpl by itself, thus we call that one.
140 return this->calculateScalarImpl(par);
141 } else {
142 static_assert(Dune::AlwaysFalse<AutoDiffFE>::value,
143 "Appropriate calculateScalar and calculateScalarImpl functions are not implemented for the "
144 "chosen element.");
145 }
146 }

153 const RealFE& realFE() const { return *this; }
