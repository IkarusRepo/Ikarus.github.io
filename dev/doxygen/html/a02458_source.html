232 auto computeMaterialAndStrains(const Dune::FieldVector<double, 2>& gpPos, int gpIndex, const Geometry& geo,
233 const auto& uFunction) const {
234 using ST = typename std::remove_cvref_t<decltype(uFunction)>::ctype;
235
236 KinematicVariables<ST> kin;
237 using namespace Dune;
238 using namespace Dune::DerivativeDirections;
239 const auto [X, Jd, Hd] = geo_->impl().zeroFirstAndSecondDerivativeOfPosition(gpPos);
240 kin.J = toEigen(Jd);
241 kin.H = toEigen(Hd);
242 const Eigen::Matrix<double, 2, 2> A = kin.J * kin.J.transpose();
243 Eigen::Matrix<double, 3, 3> G = Eigen::Matrix<double, 3, 3>::Zero();
244
245 G.block<2, 2>(0, 0) = A;
246 G(2, 2) = 1;
247 const Eigen::Matrix<double, 3, 3> GInv = G.inverse();
248 kin.C = materialTangent(GInv);
249
250 kin.epsV = membraneStrain_.value(gpPos, geo, uFunction);
251
252 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
253 const auto uasMatrix = Dune::viewAsEigenMatrixAsDynFixed(uFunction.coefficientsRef());
254 const auto hessianu = Ndd.transpose().template cast<ST>() * uasMatrix;
255 kin.h = kin.H + hessianu;
256 const Eigen::Matrix<ST, 3, 2> gradu = toEigen(uFunction.evaluateDerivative(
257 gpIndex, Dune::wrt(spatialAll), Dune::on(Dune::DerivativeDirections::referenceElement)));
258 kin.j = kin.J + gradu.transpose();
259 kin.a3N = (kin.j.row(0).cross(kin.j.row(1)));
260 kin.a3 = kin.a3N.normalized();
261 Eigen::Vector<ST, 3> bV = kin.h * kin.a3;
262 bV(2) *= 2; // Voigt notation requires the two here
263 const auto BV = toVoigt(toEigen(geo_->impl().secondFundamentalForm(gpPos)));
264 kin.kappaV = BV - bV;
265 return kin;
266 }

268 template <typename ST>
269 void calculateMatrixImpl(const FERequirementType& par, typename Traits::template MatrixType<ST> K,
270 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const {
271 K.setZero();
272 using namespace Dune::DerivativeDirections;
273 using namespace Dune;
274 const auto uFunction = displacementFunction(par, dx);
275 const auto& lambda = par.getParameter(FEParameter::loadfactor);
276
277 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
278 const auto intElement = geo_->integrationElement(gp.position()) * gp.weight();
279 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3] =
280 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
281 const Eigen::Vector<ST, membraneStrainSize> membraneForces = thickness_ * C * epsV;
282 const Eigen::Vector<ST, bendingStrainSize> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
283
284 const auto& Nd = localBasis_.evaluateJacobian(gpIndex);
285 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
286 for (size_t i = 0; i < numberOfNodes_; ++i) {
287 Eigen::Matrix<ST, membraneStrainSize, worldDim> bopIMembrane =
288 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, i);
289
290 Eigen::Matrix<ST, bendingStrainSize, worldDim> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
291 for (size_t j = i; j < numberOfNodes_; ++j) {
292 auto KBlock = K.template block<worldDim, worldDim>(worldDim * i, worldDim * j);
293 Eigen::Matrix<ST, membraneStrainSize, worldDim> bopJMembrane =
294 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, j);
295 Eigen::Matrix<ST, bendingStrainSize, worldDim> bopJBending = bopBending(jE, h, Nd, Ndd, j, a3N, a3);
296 KBlock += thickness_ * bopIMembrane.transpose() * C * bopJMembrane * intElement;
297 KBlock += Dune::power(thickness_, 3) / 12.0 * bopIBending.transpose() * C * bopJBending * intElement;
298
299 Eigen::Matrix<ST, worldDim, worldDim> kgMembraneIJ =
300 membraneStrain_.secondDerivative(gp.position(), Nd, *geo_, uFunction, localBasis_, membraneForces, i, j);
301 Eigen::Matrix<ST, worldDim, worldDim> kgBendingIJ = kgBending(jE, h, Nd, Ndd, a3N, a3, moments, i, j);
302 KBlock += kgMembraneIJ * intElement;
303 KBlock += kgBendingIJ * intElement;
304 }
305 }
306 }
307 K.template triangularView<Eigen::StrictlyLower>() = K.transpose();
308
309 // Update due to displacement-dependent external loads, e.g., follower loads
310 VolumeType::calculateMatrixImpl(par, K, dx);
311 TractionType::calculateMatrixImpl(par, K, dx);
312 }

314 template <typename ST>
315 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ST> force,
316 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const {
317 force.setZero();
318 using namespace Dune::DerivativeDirections;
319 using namespace Dune;
320 const auto uFunction = displacementFunction(par, dx);
321 const auto& lambda = par.getParameter(FEParameter::loadfactor);
322
323 // Internal forces
324 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
325 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3] =
326 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
327 const Eigen::Vector<ST, 3> membraneForces = thickness_ * C * epsV;
328 const Eigen::Vector<ST, 3> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
329
330 const auto& Nd = localBasis_.evaluateJacobian(gpIndex);
331 const auto& Ndd = localBasis_.evaluateSecondDerivatives(gpIndex);
332 for (size_t i = 0; i < numberOfNodes_; ++i) {
333 Eigen::Matrix<ST, 3, 3> bopIMembrane =
334 membraneStrain_.derivative(gp.position(), jE, Nd, *geo_, uFunction, localBasis_, i);
335 Eigen::Matrix<ST, 3, 3> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
336 force.template segment<3>(3 * i) +=
337 bopIMembrane.transpose() * membraneForces * geo_->integrationElement(gp.position()) * gp.weight();
338 force.template segment<3>(3 * i) +=
339 bopIBending.transpose() * moments * geo_->integrationElement(gp.position()) * gp.weight();
340 }
341 }
342
343 // External forces volume forces over the domain
344 VolumeType::calculateVectorImpl(par, force, dx);
345
346 // External forces, boundary forces, i.e., at the Neumann boundary
347 TractionType::calculateVectorImpl(par, force, dx);
348 }

350 template <typename ST>
351 auto calculateScalarImpl(const FERequirementType& par,
352 const std::optional<const Eigen::VectorX<ST>>& dx = std::nullopt) const -> ST {
353 using namespace Dune::DerivativeDirections;
354 using namespace Dune;
355 const auto uFunction = displacementFunction(par, dx);
356 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
357 ST energy = 0.0;
358
359 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
360 const auto [C, epsV, kappaV, j, J, h, H, a3N, a3] =
361 computeMaterialAndStrains(gp.position(), gpIndex, *geo_, uFunction);
362
363 const ST membraneEnergy = 0.5 * thickness_ * epsV.dot(C * epsV);
364 const ST bendingEnergy = 0.5 * Dune::power(thickness_, 3) / 12.0 * kappaV.dot(C * kappaV);
365 energy += (membraneEnergy + bendingEnergy) * geo_->integrationElement(gp.position()) * gp.weight();
366 }
367
368 // External forces volume forces over the domain
369 energy += VolumeType::calculateScalarImpl(par, dx);
370
371 // line or surface loads, i.e., neumann boundary
372 energy += TractionType::calculateScalarImpl(par, dx);
373 return energy;
374 }

376 template <typename ST>
377 Eigen::Matrix<ST, 3, 3> kgBending(const Eigen::Matrix<ST, 2, 3>& jcur, const Eigen::Matrix3<ST>& h, const auto& dN,
378 const auto& ddN, const Eigen::Vector3<ST>& a3N, const Eigen::Vector3<ST>& a3,
379 const Eigen::Vector3<ST>& S, int I, int J) const {
380 Eigen::Matrix<ST, 3, 3> kg;
381 kg.setZero();
382
383 const auto& dN1i = dN(I, 0);
384 const auto& dN1j = dN(J, 0);
385 const auto& dN2i = dN(I, 1);
386 const auto& dN2j = dN(J, 1);
387
388 const Eigen::Matrix<ST, 3, 3> P =
389 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose());
390
391 const auto a1dxI = Eigen::Matrix<double, 3, 3>::Identity() *
392 dN1i; // the auto here allows the exploitation of the identity matrices,
393 // due to Eigen's expression templates
394 const auto a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN2i;
395 const auto a1dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN1j;
396 const auto a2dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN2j;
397 const auto a1 = jcur.row(0);
398 const auto a2 = jcur.row(1);
399 const Eigen::Matrix<ST, 3, 3> a3NdI = a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1);
400 const Eigen::Matrix<ST, 3, 3> a3NdJ = a1dxJ.colwise().cross(a2) - a2dxJ.colwise().cross(a1);
401 Eigen::Matrix<ST, 3, 3> a3dI = P * a3NdI;
402 Eigen::Matrix<ST, 3, 3> a3dJ = P * a3NdJ;
403 for (int i = 0; i < 3; ++i) {
404 const auto a_albe = h.row(i).transpose();
405 const auto& ddNI = ddN(I, i);
406 const auto& ddNJ = ddN(J, i);
407 Eigen::Vector3<ST> vecd = P * a_albe;
408
409 Eigen::Matrix<ST, 3, 3> a3Ndd =
410 1.0 / a3N.squaredNorm() *
411 ((3 * a3 * a3.transpose() - Eigen::Matrix<double, 3, 3>::Identity()) * (a3.dot(a_albe)) -
412 a_albe * a3.transpose() - a3 * a_albe.transpose());
413
414 Eigen::Matrix<ST, 3, 3> secondDerivativeDirectorIJ = skew(((dN2i * dN1j - dN1i * dN2j) * vecd).eval());
415 kg -= (a3NdI.transpose() * a3Ndd * a3NdJ + secondDerivativeDirectorIJ + (ddNI * a3dJ + ddNJ * a3dI.transpose())) *
416 S[i] * (i == 2 ? 2 : 1);
417 }
418
419 return kg;
420 }

422 template <typename ST>
423 Eigen::Matrix<ST, 3, 3> bopBending(const Eigen::Matrix<ST, 2, 3>& jcur, const Eigen::Matrix3<ST>& h, const auto& dN,
424 const auto& ddN, const int node, const Eigen::Vector3<ST>& a3N,
425 const Eigen::Vector3<ST>& a3) const {
426 const Eigen::Matrix<ST, 3, 3> a1dxI =
427 Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 0); // this should be double
428 // but the cross-product below complains otherwise
429 const Eigen::Matrix<ST, 3, 3> a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 1);
430 const auto a1 = jcur.row(0);
431 const auto a2 = jcur.row(1);
432 const Eigen::Matrix<ST, 3, 3> a3NdI =
433 a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1); // minus needed since order has
434 // to be swapped to get column-wise cross product working
435 const Eigen::Matrix<ST, 3, 3> a3d1 =
436 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose()) * a3NdI;
437
438 Eigen::Matrix<ST, 3, 3> bop = -(h * a3d1 + (a3 * ddN.row(node)).transpose());
439 bop.row(2) *= 2;
440
441 return bop;
442 }

444 Eigen::Matrix<double, 3, 3> materialTangent(const Eigen::Matrix<double, 3, 3>& Aconv) const {
445 const double lambda = emod_ * nu_ / ((1.0 + nu_) * (1.0 - 2.0 * nu_));
446 const double mu = emod_ / (2.0 * (1.0 + nu_));
447 const double lambdbar = 2.0 * lambda * mu / (lambda + 2.0 * mu);
448 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3, 3>> moduli;
449 const auto AconvT = tensorView(Aconv, std::array<Eigen::Index, 2>({3, 3}));
450 moduli = lambdbar * dyadic(AconvT, AconvT).eval() + 2.0 * mu * symmetricFourthOrder<double>(Aconv, Aconv);
451
452 auto C = toVoigt(moduli);
453 Eigen::Matrix<double, 3, 3> C33 = C({0, 1, 5}, {0, 1, 5});
454 return C33;
455 }
