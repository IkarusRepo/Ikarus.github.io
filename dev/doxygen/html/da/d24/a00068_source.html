82 /* NonLinearOperator is a class taking linear algebra function and their arguments.
83 * The fcuntion are assumed to be derivatvies of each other. */
84 template <typename... DerivativeArgs, typename... ParameterArgs>
85 class NonLinearOperator<Impl::Functions<DerivativeArgs...>, Impl::Parameter<ParameterArgs...>> {
86 public:
87 using FunctionReturnValues = std::tuple<Ikarus::Std::ReturnType<DerivativeArgs, ParameterArgs&...>...>;
88 using ParameterValues = std::tuple<ParameterArgs...>;
89 template <int n>
90 using FunctionReturnType = std::tuple_element_t<n, FunctionReturnValues>;
91
92 template <int n>
93 using ParameterValue = std::remove_cvref_t<std::tuple_element_t<n, ParameterValues>>;
94
95 using ValueType = std::remove_cvref_t<std::tuple_element_t<0, FunctionReturnValues>>;
96 using DerivativeType = std::remove_cvref_t<std::tuple_element_t<1, FunctionReturnValues>>;
97
98 explicit NonLinearOperator(const Impl::Functions<DerivativeArgs...>& derivativesFunctions,
99 const Impl::Parameter<ParameterArgs...>& parameterI)
100 : derivatives_{derivativesFunctions.args},
101 args_{parameterI.args},
102 derivativesEvaluated_(initResults(derivativesFunctions, parameterI)) {}
103
104 /* Evaluates all functions. Usually called if the parameters changes */
105 void updateAll() {
106 Dune::Hybrid::forEach(
107 Dune::Hybrid::integralRange(Dune::index_constant<sizeof...(DerivativeArgs)>()), [&](const auto i) {
108 std::get<i>(derivativesEvaluated_) = Impl::applyAndRemoveReferenceWrapper(std::get<i>(derivatives_), args_);
109 });
110 }
111
112 /* Evaluates the n-th function */
113 template <int n>
114 void update() {
115 std::get<n>(derivativesEvaluated_) = Impl::applyAndRemoveReferenceWrapper(std::get<n>(derivatives_), args_);
116 }
117
118 /* Returns the value of the zeros function, e.g. the energy value as reference */
119 auto& value() requires(sizeof...(DerivativeArgs) > 0) { return nthDerivative<0>(); }
120
121 /* Returns the derivative value, e.g. the gradient of an energy */
122 auto& derivative() requires(sizeof...(DerivativeArgs) > 1) { return nthDerivative<1>(); }
123
124 /* Returns the second derivative value, e.g. the Hessian of an energy */
125 auto& secondDerivative() requires(sizeof...(DerivativeArgs) > 2) { return nthDerivative<2>(); }
126
127 /* Returns the n-th derivative value */
128 template <int n>
129 auto& nthDerivative() requires(sizeof...(DerivativeArgs) > n) {
130 if constexpr (requires { std::get<n>(derivativesEvaluated_).get(); })
131 return std::get<n>(derivativesEvaluated_).get();
132 else
133 return std::get<n>(derivativesEvaluated_);
134 }
135
136 /* Returns the last parameter value */
137 auto& lastParameter() { return nthParameter<sizeof...(ParameterArgs) - 1>(); }
138 /* Returns the first parameter value */
139 auto& firstParameter() requires(sizeof...(ParameterArgs) > 0) { return nthParameter<0>(); }
140 /* Returns the second parameter value */
141 auto& secondParameter() requires(sizeof...(ParameterArgs) > 1) { return nthParameter<1>(); }
142 /* Returns the n-th parameter value */
143 template <int n>
144 auto& nthParameter() requires(sizeof...(ParameterArgs) >= n) {
145 return std::get<n>(args_).get();
146 }
147
148 /* Returns a new NonLinearOperator from the given indices */
149 template <int... Derivatives>
150 auto subOperator() {
151 return Ikarus::NonLinearOperator(functions(std::get<Derivatives>(derivatives_)...),
152 Impl::applyAndRemoveReferenceWrapper(parameter<ParameterArgs...>, args_));
153 }
154
155 private:
156 using FunctionReturnValuesWrapper = std::tuple<std::conditional_t<
157 std::is_reference_v<Ikarus::Std::ReturnType<DerivativeArgs, ParameterArgs&...>>,
158 std::reference_wrapper<std::remove_cvref_t<Ikarus::Std::ReturnType<DerivativeArgs, ParameterArgs&...>>>,
159 std::remove_cvref_t<Ikarus::Std::ReturnType<DerivativeArgs, ParameterArgs&...>>>...>;
160 std::tuple<std::conditional_t<std::is_reference_v<DerivativeArgs>,
161 std::reference_wrapper<std::remove_cvref_t<DerivativeArgs>>,
162 std::remove_cvref_t<DerivativeArgs>>...>
163 derivatives_;
164 std::tuple<std::conditional_t<std::is_reference_v<ParameterArgs>,
165 std::reference_wrapper<std::remove_cvref_t<ParameterArgs>>,
166 std::remove_cvref_t<ParameterArgs>>...>
167 args_;
168 FunctionReturnValuesWrapper derivativesEvaluated_{};
169 };
