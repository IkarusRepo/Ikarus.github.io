27 template <class ScalarType>
28 ScalarType energyHelper(const Eigen::Vector<ScalarType, 3>& epsV, const auto& Aconv, double E, double nu) {
29 const double lambda = E * nu / ((1.0 + nu) * (1.0 - 2.0 * nu));
30 const double mu = E / (2.0 * (1.0 + nu));
31 const double lambdbar = 2.0 * lambda * mu / (lambda + 2.0 * mu);
32 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3, 3>> moduli;
33 const auto AconvT = TensorCast(Aconv, std::array<Eigen::Index, 2>({3, 3}));
34 moduli = lambdbar * dyadic(AconvT, AconvT).eval() + 2 * mu * symmetricFourthOrder<double>(Aconv, Aconv);
35
36 auto C = toVoigt(moduli);
37 auto C33 = C({0, 1, 5}, {0, 1, 5}).eval();
38
39 return 0.5 * epsV.dot(C33 * epsV);
40 }

58 template <typename VolumeLoad = LoadDefault, typename NeumannBoundaryLoad = LoadDefault>
59 KirchhoffLoveShell(const Basis& globalBasis, const typename LocalView::Element& element, double emod, double nu,
60 double thickness, VolumeLoad p_volumeLoad = {},
61 const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
62 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
63 : BasePowerFE(globalBasis.flat(), element),
64 neumannBoundary{p_neumannBoundary},
65 emod_{emod},
66 nu_{nu},
67 thickness_{thickness} {
68 this->localView().bind(element);
69 auto& first_child = this->localView().tree().child(0);
70 const auto& fe = first_child.finiteElement();
71 numberOfNodes = fe.size();
72 dispAtNodes.resize(fe.size());
73 order = 2 * (fe.localBasis().order());
74 localBasis = Dune::CachedLocalBasis(fe.localBasis());
75 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order); })
76 if (this->localView().element().impl().isTrimmed())
77 localBasis.bind(this->localView().element().impl().getQuadratureRule(order), Dune::bindDerivatives(0, 1, 2));
78 else
79 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
80 Dune::bindDerivatives(0, 1, 2));
81 else
82 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
83 Dune::bindDerivatives(0, 1, 2));
84
85 if constexpr (!std::is_same_v<VolumeLoad, LoadDefault>) volumeLoad = p_volumeLoad;
86 if constexpr (!std::is_same_v<NeumannBoundaryLoad, LoadDefault>) neumannBoundaryLoad = p_neumannBoundaryLoad;
87
88 assert(((not p_neumannBoundary and not neumannBoundaryLoad) or (p_neumannBoundary and neumannBoundaryLoad))
89 && "If you pass a Neumann boundary you should also pass the function for the Neumann load!");
90 }

93 template <typename ScalarType = double>
94 auto getDisplacementFunction(const FERequirementType& par,
95 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
96 const auto& d = par.getGlobalSolution(Ikarus::FESolutions::displacement);
97
98 Dune::BlockVector<Dune::RealTuple<ScalarType, Traits::worlddim>> disp(dispAtNodes.size());
99 if (dx)
100 for (auto i = 0U; i < disp.size(); ++i)
101 for (auto k2 = 0U; k2 < worlddim; ++k2)
102 disp[i][k2] = dx.value()[i * worlddim + k2]
103 + d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
104 else
105 for (auto i = 0U; i < disp.size(); ++i)
106 for (auto k2 = 0U; k2 < worlddim; ++k2)
107 disp[i][k2] = d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
108
109 auto geo = std::make_shared<const typename GridView::GridView::template Codim<0>::Entity::Geometry>(
110 this->localView().element().geometry());
111 Dune::StandardLocalFunction uFunction(localBasis, disp, geo);
112 return std::make_pair(uFunction, disp);
113 }

141 template <typename ScalarType>
142 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
143 = std::nullopt) const -> ScalarType {
144 using namespace Dune::DerivativeDirections;
145 using namespace Dune;
146 const auto [uFunction, uNodes] = getDisplacementFunction(par, dx);
147 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
148 const auto geo = this->localView().element().geometry();
149 ScalarType energy = 0.0;
150 const auto uasMatrix = Dune::viewAsEigenMatrixAsDynFixed(uNodes);
151
152 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
153 const auto [X, Jd, Hd] = geo.impl().zeroFirstAndSecondDerivativeOfPosition(gp.position());
154 const auto J = toEigen(Jd);
155 const auto H = toEigen(Hd);
156 const Eigen::Matrix<double, 2, 2> A = J * J.transpose();
157 const Eigen::Matrix<ScalarType, 3, 2> gradu = toEigen(
158 uFunction.evaluateDerivative(gpIndex, wrt(spatialAll), Dune::on(DerivativeDirections::referenceElement)));
159 const Eigen::Matrix<ScalarType, 2, 3> j = J + gradu.transpose();
160
161 const auto& Ndd = localBasis.evaluateSecondDerivatives(gpIndex);
162 const auto h = H + Ndd.transpose().template cast<ScalarType>() * uasMatrix;
163 const Eigen::Vector3<ScalarType> a3 = (j.row(0).cross(j.row(1))).normalized();
164 Eigen::Vector<ScalarType, 3> bV = h * a3;
165 bV(2) *= 2; // Voigt notation requires the two here
166
167 Eigen::Matrix<double, 3, 3> G;
168 G.setZero();
169 G.block<2, 2>(0, 0) = A;
170 G(2, 2) = 1;
171 const Eigen::Matrix<double, 3, 3> GInv = G.inverse();
172
173 const auto epsV = toVoigt((0.5 * (j * j.transpose() - A)).eval()).eval();
174 const auto BV = toVoigt(toEigen(geo.impl().secondFundamentalForm(gp.position())));
175 const auto kappaV = (BV - bV).eval();
176 const ScalarType membraneEnergy = thickness_ * energyHelper(epsV, GInv, emod_, nu_);
177 const ScalarType bendingEnergy = Dune::power(thickness_, 3) / 12.0 * energyHelper(kappaV, GInv, emod_, nu_);
178 energy += (membraneEnergy + bendingEnergy) * geo.integrationElement(gp.position()) * gp.weight();
179 }
180
181 if (volumeLoad) {
182 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
183 const auto u = uFunction.evaluate(gpIndex);
184 const Eigen::Vector<double, Traits::worlddim> fExt = volumeLoad(toEigen(geo.global(gp.position())), lambda);
185 energy -= u.dot(fExt) * geo.integrationElement(gp.position()) * gp.weight();
186 }
187 }
188
189 // line or surface loads, i.e., neumann boundary
190 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
191
192 const auto& element = this->localView().element();
193 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
194 if (not neumannBoundary or not neumannBoundary->contains(intersection)) continue;
195
196 const auto& quadLine = Dune::QuadratureRules<double, Traits::mydim - 1>::rule(intersection.type(), order);
197
198 for (const auto& curQuad : quadLine) {
199 // Local position of the quadrature point
200 const Dune::FieldVector<double, Traits::mydim>& quadPos
201 = intersection.geometryInInside().global(curQuad.position());
202
203 const double intElement = intersection.geometry().integrationElement(curQuad.position());
204
205 // The value of the local function
206 const auto u = uFunction.evaluate(quadPos);
207
208 // Value of the Neumann data at the current position
209 const auto neumannValue
210 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
211 energy -= neumannValue.dot(u) * curQuad.weight() * intElement;
212 }
213 }
214
215 return energy;
216 }
217 };
