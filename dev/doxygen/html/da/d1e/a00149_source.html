116 template <typename T, typename... Types>
117 auto makeNestedTupleFlatImpl() {
118 constexpr bool isTuple = isSpecialization<std::tuple, T>::value;
119 if constexpr (sizeof...(Types) > 0) {
120 if constexpr (isTuple)
121 return std::tuple_cat(makeNestedTupleFlat(T()), makeNestedTupleFlatImpl<Types...>());
122 else
123 return std::tuple_cat(std::make_tuple(T()), makeNestedTupleFlatImpl<Types...>());
124 } else {
125 if constexpr (isTuple)
126 return makeNestedTupleFlat(T());
127 else
128 return std::make_tuple(T());
129 }
130 }

132 template <typename T, typename... Types>
133 auto makeNestedTupleFlatAndStoreReferencesImpl(const std::tuple<T, Types...>& tup) {
134 constexpr bool isTuple = isSpecialization<std::tuple, std::remove_cvref_t<T>>::value;
135 if constexpr (sizeof...(Types) > 0) {
136 if constexpr (isTuple)
137 return std::tuple_cat(
138 makeNestedTupleFlatAndStoreReferencesImpl(std::get<0>(tup)),
139 std::apply(
140 [](const T&, const Types&... args) {
141 return makeNestedTupleFlatAndStoreReferencesImpl(std::make_tuple(std::cref(args)...));
142 },
143 tup));
144 else
145 return std::tuple_cat(
146 std::make_tuple(std::cref(std::get<0>(tup))),
147 std::apply(
148 [](const T&, const Types&... args) {
149 return makeNestedTupleFlatAndStoreReferencesImpl(std::make_tuple(std::cref(args)...));
150 },
151 tup));
152 } else {
153 if constexpr (isTuple)
154 return makeNestedTupleFlatAndStoreReferencesImpl(std::get<0>(tup));
155 else
156 return std::make_tuple(std::cref(std::get<0>(tup)));
157 }

191 template <typename Tuple, typename Predicate>
192 constexpr size_t find_if(Tuple&& tuple, Predicate pred) {
193 size_t index = std::tuple_size<std::remove_reference_t<Tuple>>::value;
194 size_t currentIndex = 0;
195 bool found = false;
196
197 Dune::Hybrid::forEach(tuple, [&](auto&& value) {
198 if (!found && pred(value)) {
199 index = currentIndex;
200 found = true;
201 }
202 ++currentIndex;
203 });
204 return index;
205 }
