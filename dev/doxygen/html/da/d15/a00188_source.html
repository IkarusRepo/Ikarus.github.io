34 struct StandardArcLength {
35 void evaluateSubsidiaryFunction(SubsidiaryArgs& args) const {
36 if (psi) {
37 const auto root = sqrt(args.DD.squaredNorm() + psi.value() * psi.value() * args.Dlambda * args.Dlambda);
38 args.f = root - args.stepSize;
39 args.dfdDD = args.DD / root;
40 args.dfdDlambda = (psi.value() * psi.value() * args.Dlambda) / root;
41 } else
42 DUNE_THROW(Dune::InvalidStateException,
43 "You have to call initialPrediction first. Otherwise psi is not defined");
44 }

46 template <typename NonLinearOperator>
47 void initialPrediction(NonLinearOperator& nonLinearOperator, SubsidiaryArgs& args) {
48 Ikarus::SolverTypeTag solverTag;
49 using JacobianType = std::remove_cvref_t<typename NonLinearOperator::DerivativeType>;
50 static_assert((Ikarus::Std::isSpecializationTypeAndNonTypes<Eigen::Matrix, JacobianType>::value)
51 or (Ikarus::Std::isSpecializationTypeNonTypeAndType<Eigen::SparseMatrix, JacobianType>::value),
52 "Linear solver not implemented for the chosen derivative type of the non-linear operator");
53
54 if constexpr (Ikarus::Std::isSpecializationTypeAndNonTypes<Eigen::Matrix, JacobianType>::value)
55 solverTag = Ikarus::SolverTypeTag::d_LDLT;
56 else
57 solverTag = Ikarus::SolverTypeTag::sd_SimplicialLDLT;
58
59 nonLinearOperator.lastParameter() = 1.0; // lambda =1.0
60
61 nonLinearOperator.template update<0>();
62 const auto& R = nonLinearOperator.value();
63 const auto& K = nonLinearOperator.derivative();
64
65 static constexpr bool isLinearSolver
66 = Ikarus::Concepts::LinearSolverCheck<decltype(Ikarus::LinearSolver(solverTag)),
67 typename NonLinearOperator::DerivativeType,
68 typename NonLinearOperator::ValueType>;
69 static_assert(isLinearSolver,
70 "Initial predictor step in the standard arc-length method doesn't have a linear solver");
71
72 auto linearSolver = Ikarus::LinearSolver(solverTag); // for the linear predictor step
73 linearSolver.analyzePattern(K);
74 linearSolver.factorize(K);
75 linearSolver.solve(args.DD, -R);
76
77 const auto DD2 = args.DD.squaredNorm();
78
79 psi = sqrt(DD2);
80 auto s = sqrt(psi.value() * psi.value() + DD2);
81
82 args.DD = args.DD * args.stepSize / s;
83 args.Dlambda = args.stepSize / s;
84
85 nonLinearOperator.firstParameter() = args.DD;
86 nonLinearOperator.lastParameter() = args.Dlambda;
87 }
88
89 template <typename NonLinearOperator>
90 void intermediatePrediction(NonLinearOperator& nonLinearOperator, SubsidiaryArgs& args) {
91 nonLinearOperator.firstParameter() += args.DD;
92 nonLinearOperator.lastParameter() += args.Dlambda;
93 }
94
95 std::string name = "Arc length";
96
97 private:
98 std::optional<double> psi;
99 };
