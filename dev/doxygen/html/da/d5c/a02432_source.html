246 auto computeMaterialAndStrains(const Dune::FieldVector<double, 2>& gpPos, int gpIndex, const Geometry& geo,
247 const auto& uFunction) const {
248 using ScalarType = typename std::remove_cvref_t<decltype(uFunction)>::ctype;
249
250 KinematicVariables<ScalarType> kin;
251 using namespace Dune;
252 using namespace Dune::DerivativeDirections;
253 const auto [X, Jd, Hd] = geo.impl().zeroFirstAndSecondDerivativeOfPosition(gpPos);
254 kin.J = toEigen(Jd);
255 kin.H = toEigen(Hd);
256 const Eigen::Matrix<double, 2, 2> A = kin.J * kin.J.transpose();
257 Eigen::Matrix<double, 3, 3> G = Eigen::Matrix<double, 3, 3>::Zero();
258
259 G.block<2, 2>(0, 0) = A;
260 G(2, 2) = 1;
261 const Eigen::Matrix<double, 3, 3> GInv = G.inverse();
262 kin.C = materialTangent(GInv);
263
264 kin.epsV = membraneStrain.value(gpPos, geo, uFunction);
265
266 const auto& Ndd = localBasis.evaluateSecondDerivatives(gpIndex);
267 const auto uasMatrix = Dune::viewAsEigenMatrixAsDynFixed(uFunction.coefficientsRef());
268 const auto hessianu = Ndd.transpose().template cast<ScalarType>() * uasMatrix;
269 kin.h = kin.H + hessianu;
270 const Eigen::Matrix<ScalarType, 3, 2> gradu = toEigen(uFunction.evaluateDerivative(
271 gpIndex, Dune::wrt(spatialAll), Dune::on(Dune::DerivativeDirections::referenceElement)));
272 kin.j = kin.J + gradu.transpose();
273 kin.a3N = (kin.j.row(0).cross(kin.j.row(1)));
274 kin.a3 = kin.a3N.normalized();
275 Eigen::Vector<ScalarType, 3> bV = kin.h * kin.a3;
276 bV(2) *= 2; // Voigt notation requires the two here
277 const auto BV = toVoigt(toEigen(geo.impl().secondFundamentalForm(gpPos)));
278 kin.kappaV = BV - bV;
279 return kin;
280 }

282 template <typename ScalarType>
283 void calculateMatrixImpl(const FERequirementType& par, typename Traits::template MatrixType<ScalarType> K,
284 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
285 K.setZero();
286 using namespace Dune::DerivativeDirections;
287 using namespace Dune;
288 const auto uFunction = getDisplacementFunction(par, dx);
289 const auto& lambda = par.getParameter(FEParameter::loadfactor);
290 const auto geo = this->localView().element().geometry();
291
292 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
293 const auto intElement = geo.integrationElement(gp.position()) * gp.weight();
294 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3]
295 = computeMaterialAndStrains(gp.position(), gpIndex, geo, uFunction);
296 const Eigen::Vector<ScalarType, membraneStrainSize> membraneForces = thickness_ * C * epsV;
297 const Eigen::Vector<ScalarType, bendingStrainSize> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
298
299 const auto& Nd = localBasis.evaluateJacobian(gpIndex);
300 const auto& Ndd = localBasis.evaluateSecondDerivatives(gpIndex);
301 for (size_t i = 0; i < numberOfNodes; ++i) {
302 Eigen::Matrix<ScalarType, membraneStrainSize, worlddim> bopIMembrane
303 = membraneStrain.derivative(gp.position(), jE, Nd, geo, uFunction, localBasis, i);
304
305 Eigen::Matrix<ScalarType, bendingStrainSize, worlddim> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
306 for (size_t j = i; j < numberOfNodes; ++j) {
307 auto KBlock = K.template block<worlddim, worlddim>(worlddim * i, worlddim * j);
308 Eigen::Matrix<ScalarType, membraneStrainSize, worlddim> bopJMembrane
309 = membraneStrain.derivative(gp.position(), jE, Nd, geo, uFunction, localBasis, j);
310 Eigen::Matrix<ScalarType, bendingStrainSize, worlddim> bopJBending = bopBending(jE, h, Nd, Ndd, j, a3N, a3);
311 KBlock += thickness_ * bopIMembrane.transpose() * C * bopJMembrane * intElement;
312 KBlock += Dune::power(thickness_, 3) / 12.0 * bopIBending.transpose() * C * bopJBending * intElement;
313
314 Eigen::Matrix<ScalarType, worlddim, worlddim> kgMembraneIJ
315 = membraneStrain.secondDerivative(gp.position(), Nd, geo, uFunction, localBasis, membraneForces, i, j);
316 Eigen::Matrix<ScalarType, worlddim, worlddim> kgBendingIJ
317 = kgBending(jE, h, Nd, Ndd, a3N, a3, moments, i, j);
318 KBlock += kgMembraneIJ * intElement;
319 KBlock += kgBendingIJ * intElement;
320 }
321 }
322 }
323 K.template triangularView<Eigen::StrictlyLower>() = K.transpose();
324 }

326 template <typename ScalarType>
327 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
328 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
329 force.setZero();
330 using namespace Dune::DerivativeDirections;
331 using namespace Dune;
332 const auto uFunction = getDisplacementFunction(par, dx);
333 const auto& lambda = par.getParameter(FEParameter::loadfactor);
334 const auto geo = this->localView().element().geometry();
335
336 // Internal forces
337 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
338 const auto [C, epsV, kappaV, jE, J, h, H, a3N, a3]
339 = computeMaterialAndStrains(gp.position(), gpIndex, geo, uFunction);
340 const Eigen::Vector<ScalarType, 3> membraneForces = thickness_ * C * epsV;
341 const Eigen::Vector<ScalarType, 3> moments = Dune::power(thickness_, 3) / 12.0 * C * kappaV;
342
343 const auto& Nd = localBasis.evaluateJacobian(gpIndex);
344 const auto& Ndd = localBasis.evaluateSecondDerivatives(gpIndex);
345 for (size_t i = 0; i < numberOfNodes; ++i) {
346 Eigen::Matrix<ScalarType, 3, 3> bopIMembrane
347 = membraneStrain.derivative(gp.position(), jE, Nd, geo, uFunction, localBasis, i);
348 Eigen::Matrix<ScalarType, 3, 3> bopIBending = bopBending(jE, h, Nd, Ndd, i, a3N, a3);
349 force.template segment<3>(3 * i)
350 += bopIMembrane.transpose() * membraneForces * geo.integrationElement(gp.position()) * gp.weight();
351 force.template segment<3>(3 * i)
352 += bopIBending.transpose() * moments * geo.integrationElement(gp.position()) * gp.weight();
353 }
354 }
355
356 // External forces volume forces over the domain
357 if (volumeLoad) {
358 const auto u = getDisplacementFunction(par, dx);
359 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
360 Eigen::Vector<double, Traits::worlddim> fext = volumeLoad(geo.global(gp.position()), lambda);
361 for (size_t i = 0; i < numberOfNodes; ++i) {
362 const auto udCi = uFunction.evaluateDerivative(gpIndex, wrt(coeff(i)));
363 force.template segment<worlddim>(worlddim * i)
364 -= udCi * fext * geo.integrationElement(gp.position()) * gp.weight();
365 }
366 }
367 }
368
369 if (not neumannBoundary and not neumannBoundaryLoad) return;
370
371 // External forces, boundary forces, i.e. at the Neumann boundary
372 for (auto&& intersection : intersections(neumannBoundary->gridView(), this->localView().element())) {
373 if (not neumannBoundary or not neumannBoundary->contains(intersection)) continue;
374
375 const auto& quadLine = QuadratureRules<double, Element::mydimension - 1>::rule(intersection.type(), order);
376
377 for (const auto& curQuad : quadLine) {
378 // Local position of the quadrature point
379 const FieldVector<double, Element::mydimension>& quadPos
380 = intersection.geometryInInside().global(curQuad.position());
381
382 const double intElement = intersection.geometry().integrationElement(curQuad.position()) * curQuad.weight();
383 for (size_t i = 0; i < numberOfNodes; ++i) {
384 const auto udCi = uFunction.evaluateDerivative(quadPos, wrt(coeff(i)));
385
386 // Value of the Neumann data at the current position
387 auto neumannValue = neumannBoundaryLoad(intersection.geometry().global(curQuad.position()), lambda);
388 force.template segment<worlddim>(worlddim * i) -= udCi * neumannValue * intElement;
389 }
390 }
391 }
392 }

394 template <typename ScalarType>
395 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
396 = std::nullopt) const -> ScalarType {
397 using namespace Dune::DerivativeDirections;
398 using namespace Dune;
399 const auto uFunction = getDisplacementFunction(par, dx);
400 const auto& lambda = par.getParameter(FEParameter::loadfactor);
401 const auto geo = this->localView().element().geometry();
402 ScalarType energy = 0.0;
403
404 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
405 const auto [C, epsV, kappaV, j, J, h, H, a3N, a3]
406 = computeMaterialAndStrains(gp.position(), gpIndex, geo, uFunction);
407
408 const ScalarType membraneEnergy = 0.5 * thickness_ * epsV.dot(C * epsV);
409 const ScalarType bendingEnergy = 0.5 * Dune::power(thickness_, 3) / 12.0 * kappaV.dot(C * kappaV);
410 energy += (membraneEnergy + bendingEnergy) * geo.integrationElement(gp.position()) * gp.weight();
411 }
412
413 if (volumeLoad) {
414 for (const auto& [gpIndex, gp] : uFunction.viewOverIntegrationPoints()) {
415 const auto u = uFunction.evaluate(gpIndex);
416 const Eigen::Vector<double, Traits::worlddim> fExt = volumeLoad(geo.global(gp.position()), lambda);
417 energy -= u.dot(fExt) * geo.integrationElement(gp.position()) * gp.weight();
418 }
419 }
420
421 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
422
423 // line or surface loads, i.e., neumann boundary
424 const auto& element = this->localView().element();
425 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
426 if (not neumannBoundary or not neumannBoundary->contains(intersection)) continue;
427
428 const auto& quadLine = QuadratureRules<double, Traits::mydim - 1>::rule(intersection.type(), order);
429
430 for (const auto& curQuad : quadLine) {
431 // Local position of the quadrature point
432 const FieldVector<double, Traits::mydim>& quadPos
433 = intersection.geometryInInside().global(curQuad.position());
434
435 const double intElement = intersection.geometry().integrationElement(curQuad.position());
436
437 // The value of the local function
438 const auto u = uFunction.evaluate(quadPos);
439
440 // Value of the Neumann data at the current position
441 const auto neumannValue = neumannBoundaryLoad(intersection.geometry().global(curQuad.position()), lambda);
442 energy -= neumannValue.dot(u) * curQuad.weight() * intElement;
443 }
444 }
445 return energy;
446 }

448 template <typename ScalarType>
449 Eigen::Matrix<ScalarType, 3, 3> kgBending(const Eigen::Matrix<ScalarType, 2, 3>& jcur,
450 const Eigen::Matrix3<ScalarType>& h, const auto& dN, const auto& ddN,
451 const Eigen::Vector3<ScalarType>& a3N,
452 const Eigen::Vector3<ScalarType>& a3, const Eigen::Vector3<ScalarType>& S,
453 int I, int J) const {
454 Eigen::Matrix<ScalarType, 3, 3> kg;
455 kg.setZero();
456
457 const auto& dN1i = dN(I, 0);
458 const auto& dN1j = dN(J, 0);
459 const auto& dN2i = dN(I, 1);
460 const auto& dN2j = dN(J, 1);
461
462 const Eigen::Matrix<ScalarType, 3, 3> P
463 = 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose());
464
465 const auto a1dxI = Eigen::Matrix<double, 3, 3>::Identity()
466 * dN1i; // the auto here allows the exploitation of the identity matrices,
467 // due to Eigen's expression templates
468 const auto a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN2i;
469 const auto a1dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN1j;
470 const auto a2dxJ = Eigen::Matrix<double, 3, 3>::Identity() * dN2j;
471 const auto a1 = jcur.row(0);
472 const auto a2 = jcur.row(1);
473 const Eigen::Matrix<ScalarType, 3, 3> a3NdI = a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1);
474 const Eigen::Matrix<ScalarType, 3, 3> a3NdJ = a1dxJ.colwise().cross(a2) - a2dxJ.colwise().cross(a1);
475 Eigen::Matrix<ScalarType, 3, 3> a3dI = P * a3NdI;
476 Eigen::Matrix<ScalarType, 3, 3> a3dJ = P * a3NdJ;
477 for (int i = 0; i < 3; ++i) {
478 const auto a_albe = h.row(i).transpose();
479 const auto& ddNI = ddN(I, i);
480 const auto& ddNJ = ddN(J, i);
481 Eigen::Vector3<ScalarType> vecd = P * a_albe;
482
483 Eigen::Matrix<ScalarType, 3, 3> a3Ndd
484 = 1.0 / a3N.squaredNorm()
485 * ((3 * a3 * a3.transpose() - Eigen::Matrix<double, 3, 3>::Identity()) * (a3.dot(a_albe))
486 - a_albe * a3.transpose() - a3 * a_albe.transpose());
487
488 Eigen::Matrix<ScalarType, 3, 3> secondDerivativeDirectorIJ = skew(((dN2i * dN1j - dN1i * dN2j) * vecd).eval());
489 kg -= (a3NdI.transpose() * a3Ndd * a3NdJ + secondDerivativeDirectorIJ + (ddNI * a3dJ + ddNJ * a3dI.transpose()))
490 * S[i] * (i == 2 ? 2 : 1);
491 }
492
493 return kg;
494 }

496 template <typename ScalarType>
497 Eigen::Matrix<ScalarType, 3, 3> bopBending(const Eigen::Matrix<ScalarType, 2, 3>& jcur,
498 const Eigen::Matrix3<ScalarType>& h, const auto& dN, const auto& ddN,
499 const int node, const Eigen::Vector3<ScalarType>& a3N,
500 const Eigen::Vector3<ScalarType>& a3) const {
501 const Eigen::Matrix<ScalarType, 3, 3> a1dxI
502 = Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 0); // this should be double
503 // but the cross-product below complains otherwise
504 const Eigen::Matrix<ScalarType, 3, 3> a2dxI = Eigen::Matrix<double, 3, 3>::Identity() * dN(node, 1);
505 const auto a1 = jcur.row(0);
506 const auto a2 = jcur.row(1);
507 const Eigen::Matrix<ScalarType, 3, 3> a3NdI
508 = a1dxI.colwise().cross(a2) - a2dxI.colwise().cross(a1); // minus needed since order has
509 // to be swapped to get column-wise cross product working
510 const Eigen::Matrix<ScalarType, 3, 3> a3d1
511 = 1.0 / a3N.norm() * (Eigen::Matrix<double, 3, 3>::Identity() - a3 * a3.transpose()) * a3NdI;
512
513 Eigen::Matrix<ScalarType, 3, 3> bop = -(h * a3d1 + (a3 * ddN.row(node)).transpose());
514 bop.row(2) *= 2;
515
516 return bop;
517 }

519 Eigen::Matrix<double, 3, 3> materialTangent(const Eigen::Matrix<double, 3, 3>& Aconv) const {
520 const double lambda = emod_ * nu_ / ((1.0 + nu_) * (1.0 - 2.0 * nu_));
521 const double mu = emod_ / (2.0 * (1.0 + nu_));
522 const double lambdbar = 2.0 * lambda * mu / (lambda + 2.0 * mu);
523 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3, 3>> moduli;
524 const auto AconvT = tensorView(Aconv, std::array<Eigen::Index, 2>({3, 3}));
525 moduli = lambdbar * dyadic(AconvT, AconvT).eval() + 2.0 * mu * symmetricFourthOrder<double>(Aconv, Aconv);
526
527 auto C = toVoigt(moduli);
528 Eigen::Matrix<double, 3, 3> C33 = C({0, 1, 5}, {0, 1, 5});
529 return C33;
530 }
