50 template <typename MatrixType, typename Rhs, typename Dest, typename Preconditioner>
51 void truncated_conjugate_gradient(const MatrixType& mat, const Rhs& rhs, Dest& x, const Preconditioner& precond,
52 Eigen::Index& iters, typename Dest::RealScalar& tol_error,
53 TCGInfo<typename Dest::RealScalar>& _info) {
54 using std::abs;
55 using std::sqrt;
56 typedef typename Dest::RealScalar RealScalar;
57 typedef typename Dest::Scalar Scalar;
58 typedef Matrix<Scalar, Dynamic, 1> VectorType;
59
60 RealScalar tol = tol_error;
61 Index maxIters = iters;
62
63 Index n = mat.cols();
64
65 VectorType residual = rhs - mat * x; // initial residual
66
67 RealScalar rhsNorm2 = rhs.norm();
68 const RealScalar considerAsZero = (std::numeric_limits<RealScalar>::min)();
69
70 if (rhsNorm2 <= considerAsZero) {
71 x.setZero();
72 iters = 0;
73 tol_error = 0;
74 return;
75 }
76 RealScalar threshold = numext::maxi(tol * tol * rhsNorm2 * rhsNorm2, considerAsZero);
77 RealScalar residualNorm2 = residual.norm();
78 if (residualNorm2 * residualNorm2 < threshold) {
79 iters = 0;
80 tol_error = (residualNorm2 / rhsNorm2);
81 return;
82 }
83
84 double e_Pd = 0.0;
85 double e_Pe_new = 0.0;
86 double e_Pe = x.squaredNorm();
87 double d_Pd;
88 double d_Hd;
89 VectorType p(n);
90 p = precond.solve(residual); // initial search direction
91 // bool coutflag=true;
92 _info.stop_tCG = TCGStopReason::maximumInnerIterations;
93 VectorType z(n), tmp(n);
94 RealScalar absNew = numext::real(residual.dot(p)); // the square of the absolute value of r scaled by invM
95 d_Pd = absNew;
96 Index i = 1;
97 while (i < maxIters) {
98 tmp.noalias() = mat * p; // the bottleneck of the algorithm
99 d_Hd = p.dot(tmp);
100 Scalar alpha = absNew / d_Hd; // the amount we travel on dir
101
102 e_Pe_new = e_Pe + 2.0 * alpha * e_Pd + alpha * alpha * d_Pd;
103
104 if (d_Hd <= 0 || e_Pe_new >= _info.Delta * _info.Delta) // negative curvature or execdet trustregion
105 {
106 double tau = (-e_Pd + sqrt(e_Pd * e_Pd + d_Pd * (_info.Delta * _info.Delta - e_Pe))) / d_Pd;
107
108 x += tau * p;
109 if (d_Hd <= 0)
110 _info.stop_tCG = TCGStopReason::negativeCurvature;
111 else
112 _info.stop_tCG = TCGStopReason::exceededTrustRegion;
113
114 break;
115 }
116 e_Pe = e_Pe_new;
117 x += alpha * p; // update solution
118 residual -= alpha * tmp; // update residual
119
120 residualNorm2 = residual.norm();
121
122 if (i >= _info.mininner
123 && residualNorm2 <= rhsNorm2 * std::min(rhsNorm2, _info.kappa)) // missing pow(rhsNorm2,_info.theta
124 {
125 // Residual is small enough to quit
126 if (_info.kappa < rhsNorm2)
127 _info.stop_tCG = TCGStopReason::reachedTargetResidualKappaLinear;
128 else
129 _info.stop_tCG = TCGStopReason::reachedTargetResidualThetaSuperLinear;
130 break;
131 }
132 if (residualNorm2 < threshold) break;
133
134 z = precond.solve(residual); // approximately solve for "A z = residual"
135
136 RealScalar absOld = absNew;
137 absNew = numext::real(residual.dot(z)); // update the absolute value of r
138 RealScalar beta = absNew / absOld; // calculate the Gram-Schmidt value used to create the new search direction
139
140 e_Pd = beta * (e_Pd + alpha * d_Pd);
141 d_Pd = absNew + beta * beta * d_Pd;
142
143 p = z + beta * p; // update search direction
144 i++;
145 }
146 tol_error = (residualNorm2 / rhsNorm2);
147 iters = i;
148 _info.numInnerIter = i;
149 }

214 template <typename Rhs, typename Dest>

216 typedef typename Base::MatrixWrapper MatrixWrapper;
217 typedef typename Base::ActualMatrixType ActualMatrixType;
218 enum {
219 TransposeInput = (!MatrixWrapper::MatrixFree) && (UpLo == (Lower | Upper)) && (!MatrixType::IsRowMajor)
220 && (!NumTraits<Scalar>::IsComplex)
221 };
222 typedef std::conditional_t<TransposeInput, Transpose<const ActualMatrixType>, ActualMatrixType const&>
223 RowMajorWrapper;
224 EIGEN_STATIC_ASSERT(internal::check_implication(MatrixWrapper::MatrixFree, UpLo == (Lower | Upper)),
225 MATRIX_FREE_CONJUGATE_GRADIENT_IS_COMPATIBLE_WITH_UPPER_UNION_LOWER_MODE_ONLY);
226 typedef std::conditional_t<UpLo == (Lower | Upper), RowMajorWrapper,
227 typename MatrixWrapper::template ConstSelfAdjointViewReturnType<UpLo>::Type>
228 SelfAdjointWrapper;
229
230 m_iterations = Base::maxIterations();
231 m_error = Base::m_tolerance;
232
233 RowMajorWrapper row_mat(matrix());
234 internal::truncated_conjugate_gradient(SelfAdjointWrapper(row_mat), b, x, Base::m_preconditioner, m_iterations,
235 m_error, algInfo);
236 m_info = m_error <= Base::m_tolerance ? Success : NoConvergence;
