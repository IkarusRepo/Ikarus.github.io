229 template <template <typename...> class MaterialToCheck, typename Material>
230 consteval bool isMaterial() {
231 if constexpr (Std::isSpecialization<MaterialToCheck, Material>::value) return true;
232
233 if constexpr (Std::isSpecializationNonTypeAndTypes<VanishingStress, Material>::value) {
234 if constexpr (Std::isSpecialization<MaterialToCheck, typename Material::Underlying>::value) {
235 return true;
236 } else {
237 return false;
238 }
239 } else
240 return false;
241 }
