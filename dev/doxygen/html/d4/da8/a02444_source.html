48 template <typename VolumeLoad = LoadDefault, typename NeumannBoundaryLoad = LoadDefault>
49 LinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, double emod, double nu,
50 VolumeLoad p_volumeLoad = {}, const BoundaryPatch<GridView>* p_neumannBoundary = nullptr,
51 NeumannBoundaryLoad p_neumannBoundaryLoad = {})
52 : BaseDisp(globalBasis.flat(), element), neumannBoundary{p_neumannBoundary}, emod_{emod}, nu_{nu} {
53 this->localView().bind(element);
54 auto& first_child = this->localView().tree().child(0);
55 const auto& fe = first_child.finiteElement();
56 numberOfNodes = fe.size();
57 dispAtNodes.resize(numberOfNodes);
58 order = 2 * (this->localView().tree().child(0).finiteElement().localBasis().order());
59 localBasis = Dune::CachedLocalBasis(this->localView().tree().child(0).finiteElement().localBasis());
60 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order); })
61 if (this->localView().element().impl().isTrimmed())
62 localBasis.bind(this->localView().element().impl().getQuadratureRule(order), Dune::bindDerivatives(0, 1));
63 else
64 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
65 Dune::bindDerivatives(0, 1));
66 else
67 localBasis.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order),
68 Dune::bindDerivatives(0, 1));
69
70 if constexpr (!std::is_same_v<VolumeLoad, LoadDefault>) volumeLoad = p_volumeLoad;
71 if constexpr (!std::is_same_v<NeumannBoundaryLoad, LoadDefault>) neumannBoundaryLoad = p_neumannBoundaryLoad;
72
73 assert(((not p_neumannBoundary and not neumannBoundaryLoad) or (p_neumannBoundary and neumannBoundaryLoad))
74 && "If you pass a Neumann boundary you should also pass the function for the Neumann load!");
75 }

78 template <typename ScalarType>
79 auto getDisplacementFunction(const FERequirementType& par,
80 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
81 const auto& d = par.getGlobalSolution(Ikarus::FESolutions::displacement);
82
83 Dune::BlockVector<Dune::RealTuple<ScalarType, Traits::dimension>> disp(dispAtNodes.size());
84 if (dx) {
85 for (auto i = 0U; i < disp.size(); ++i)
86 for (auto k2 = 0U; k2 < myDim; ++k2)
87 disp[i][k2] = dx.value()[i * myDim + k2]
88 + d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
89 } else
90 for (auto i = 0U; i < disp.size(); ++i)
91 for (auto k2 = 0U; k2 < myDim; ++k2)
92 disp[i][k2] = d[this->localView().index(this->localView().tree().child(k2).localIndex(i))[0]];
93
94 auto geo = std::make_shared<const typename GridView::GridView::template Codim<0>::Entity::Geometry>(
95 this->localView().element().geometry());
96 Dune::StandardLocalFunction uFunction(localBasis, disp, geo);
97
98 return uFunction;
99 }

124 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
125 const auto eps = getStrainFunction(par);
126 using namespace Dune::DerivativeDirections;
127 using namespace Dune;
128
129 const auto C = getMaterialTangent();
130 const auto geo = this->localView().element().geometry();
131
132 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
133 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
134 for (size_t i = 0; i < numberOfNodes; ++i) {
135 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
136 for (size_t j = 0; j < numberOfNodes; ++j) {
137 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
138 K.template block<myDim, myDim>(i * myDim, j * myDim) += bopI.transpose() * C * bopJ * intElement;
139 }
140 }
141 }

144 void calculateAt(const ResultRequirementsType& req, const Dune::FieldVector<double, Traits::mydim>& local,
145 ResultTypeMap<double>& result) const {
146 using namespace Dune::Indices;
147 using namespace Dune::DerivativeDirections;
148 using namespace Dune;
149
150 const auto eps = getStrainFunction(req.getFERequirements());
151 const auto C = getMaterialTangent();
152 auto epsVoigt = eps.evaluate(local, on(gridElement));
153
154 auto linearStress = (C * epsVoigt).eval();
155
156 if (req.isResultRequested(ResultType::linearStress))
157 result.insertOrAssignResult(ResultType::linearStress, linearStress);
158 else
159 DUNE_THROW(Dune::NotImplemented, "The requested result type is NOT implemented.");
160 }

179 template <typename ScalarType>
180 auto calculateScalarImpl(const FERequirementType& par, const std::optional<const Eigen::VectorX<ScalarType>>& dx
181 = std::nullopt) const -> ScalarType {
182 const auto u = getDisplacementFunction(par, dx);
183 const auto eps = getStrainFunction(par, dx);
184 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
185 using namespace Dune::DerivativeDirections;
186 using namespace Dune;
187
188 const auto C = getMaterialTangent();
189
190 const auto geo = this->localView().element().geometry();
191 ScalarType energy = 0.0;
192 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
193 const auto EVoigt = eps.evaluate(gpIndex, on(gridElement));
194
195 energy += (0.5 * EVoigt.dot(C * EVoigt)) * geo.integrationElement(gp.position()) * gp.weight();
196 }
197
198 // External forces volume forces over the domain
199 if (volumeLoad) {
200 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
201 const auto uVal = u.evaluate(gpIndex);
202 Eigen::Vector<double, Traits::worlddim> fext = volumeLoad(toEigen(geo.global(gp.position())), lambda);
203 energy -= uVal.dot(fext) * geo.integrationElement(gp.position()) * gp.weight();
204 }
205 }
206
207 // line or surface loads, i.e., neumann boundary
208 if (not neumannBoundary and not neumannBoundaryLoad) return energy;
209
210 auto element = this->localView().element();
211 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
212 if (not neumannBoundary->contains(intersection)) continue;
213
214 const auto& quadLine = Dune::QuadratureRules<double, myDim - 1>::rule(intersection.type(), order);
215
216 for (const auto& curQuad : quadLine) {
217 // Local position of the quadrature point
218 const Dune::FieldVector<double, myDim>& quadPos = intersection.geometryInInside().global(curQuad.position());
219
220 const double integrationElement = intersection.geometry().integrationElement(curQuad.position());
221
222 // The value of the local function
223 const auto uVal = u.evaluate(quadPos);
224
225 // Value of the Neumann data at the current position
226 auto neumannValue = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
227
228 energy -= neumannValue.dot(uVal) * curQuad.weight() * integrationElement;
229 }
230 }
231 return energy;
232 }

234 template <typename ScalarType>
235 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
236 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
237 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
238 const auto eps = getStrainFunction(par, dx);
239 using namespace Dune::DerivativeDirections;
240 using namespace Dune;
241
242 const auto C = getMaterialTangent();
243 const auto geo = this->localView().element().geometry();
244
245 // Internal forces
246 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
247 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
248 auto stresses = (C * eps.evaluate(gpIndex, on(gridElement))).eval();
249 for (size_t i = 0; i < numberOfNodes; ++i) {
250 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
251 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
252 }
253 }
254
255 // External forces volume forces over the domain
256 if (volumeLoad) {
257 const auto u = getDisplacementFunction(par, dx);
258 for (const auto& [gpIndex, gp] : u.viewOverIntegrationPoints()) {
259 Eigen::Vector<double, Traits::worlddim> fext = volumeLoad(toEigen(geo.global(gp.position())), lambda);
260 for (size_t i = 0; i < numberOfNodes; ++i) {
261 const auto udCi = u.evaluateDerivative(gpIndex, wrt(coeff(i)));
262 force.template segment<myDim>(myDim * i)
263 -= udCi * fext * geo.integrationElement(gp.position()) * gp.weight();
264 }
265 }
266 }
267
268 // External forces, boundary forces, i.e. at the Neumann boundary
269 if (not neumannBoundary) return;
270
271 const auto u = getDisplacementFunction(par, dx);
272 auto element = this->localView().element();
273 for (auto&& intersection : intersections(neumannBoundary->gridView(), element)) {
274 if (not neumannBoundary->contains(intersection)) continue;
275
276 // Integration rule along the boundary
277 const auto& quadLine = Dune::QuadratureRules<double, myDim - 1>::rule(intersection.type(), order);
278
279 for (const auto& curQuad : quadLine) {
280 const Dune::FieldVector<double, myDim>& quadPos = intersection.geometryInInside().global(curQuad.position());
281
282 const double integrationElement = intersection.geometry().integrationElement(curQuad.position());
283
284 // The value of the local function wrt the i-th coef
285 for (size_t i = 0; i < numberOfNodes; ++i) {
286 const auto udCi = u.evaluateDerivative(quadPos, wrt(coeff(i)));
287
288 // Value of the Neumann data at the current position
289 auto neumannValue
290 = neumannBoundaryLoad(toEigen(intersection.geometry().global(curQuad.position())), lambda);
291 force.template segment<myDim>(myDim * i) -= udCi * neumannValue * curQuad.weight() * integrationElement;
292 }
293 }
294 }
295 }
