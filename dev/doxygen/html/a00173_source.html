26[[deprecated(
27 "These are hard-coded function you should use the material library Ikarus::LinearElasticity")]] inline Eigen::
28 Matrix3d
29 planeStressLinearElasticMaterialTangent(double E, double nu) {
30 Eigen::Matrix3d C;
31 C.setZero();
32 C(0, 0) = C(1, 1) = 1;
33 C(0, 1) = C(1, 0) = nu;
34 C(2, 2) = (1 - nu) / 2;
35 C *= E / (1 - nu * nu);
36 return C;
37}

47[[deprecated(
48 "These are hard-coded function you should use the material library: Ikarus::LinearElasticity")]] inline Eigen::
49 Matrix<double, 6, 6>
50 linearElasticMaterialTangent3D(double E, double nu) {
51 Eigen::Matrix<double, 6, 6> C;
52 C.setZero();
53 C(0, 0) = C(1, 1) = C(2, 2) = 1 - nu;
54 C(0, 1) = C(1, 0) = C(2, 0) = C(0, 2) = C(1, 2) = C(2, 1) = nu;
55 C(3, 3) = C(4, 4) = C(5, 5) = (1 - 2 * nu) / 2;
56 C *= E / ((1 + nu) * (1 - 2 * nu));
57 return C;
58}

173template <typename ValuePair>
174struct ConvertLameConstants
175{
176 constexpr double toLamesFirstParameter()
177 requires(!std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter> and
178 !std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter> and
179 !std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>)
180 {
181 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
182 const auto& E = vp.emodul;
183 const auto& nu = vp.nu;
184 return Dune::FloatCmp::eq(nu, 0.5) ? std::numeric_limits<double>::infinity()
185 : E * nu / ((1.0 + nu) * (1.0 - 2.0 * nu));
186 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
187 const auto& E = vp.emodul;
188 const auto& mu = vp.mu;
189 return mu * (E - 2.0 * mu) / (3.0 * mu - E);
190 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
191 const auto& E = vp.emodul;
192 const auto& K = vp.K;
193 return 3.0 * K * (3.0 * K - E) / (9.0 * K - E);
194 } else
195 assert(false && "Your LameParameter request is not implemented");
196 }
197
198 constexpr double toBulkModulus()
199 requires(!std::is_same_v<ValuePair, YoungsModulusAndBulkModulus> and
200 !std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>)
201 {
202 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
203 const auto& E = vp.emodul;
204 const auto& nu = vp.nu;
205 return E / (3.0 * (1.0 - 2.0 * nu));
206 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
207 const auto& E = vp.emodul;
208 const auto& mu = vp.mu;
209 return E * mu / (3.0 * (3.0 * mu - E));
210 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
211 const auto& E = vp.emodul;
212 const auto& lambda = vp.lambda;
213 return (E + 3.0 * lambda + calcR(vp)) / 6.0;
214 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
215 const auto& lambda = vp.lambda;
216 const auto& mu = vp.mu;
217 return lambda + 2.0 * mu / 3.0;
218 } else
219 assert(false && "Your LameParameter request is not implemented");
220 }
221
222 constexpr double toShearModulus()
223 requires(!std::is_same_v<ValuePair, YoungsModulusAndShearModulus> and
224 !std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>)
225 {
226 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
227 const auto& E = vp.emodul;
228 const auto& nu = vp.nu;
229 return E / (2.0 * (1.0 + nu));
230 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
231 const auto& E = vp.emodul;
232 const auto& K = vp.K;
233 return 3.0 * K * E / (9.0 * K - E);
234 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
235 const auto& E = vp.emodul;
236 const auto& lambda = vp.lambda;
237 return (E - 3.0 * lambda + calcR(vp)) / 4.0;
238 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
239 const auto& K = vp.K;
240 const auto& lambda = vp.lambda;
241 return 3.0 * (K - lambda) / 2.0;
242 } else
243 assert(false && "Your LameParameter request is not implemented");
244 }
245
246 constexpr double toPWaveModulus() {
247 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>) {
248 const auto& E = vp.emodul;
249 const auto& nu = vp.nu;
250 return E * (1.0 - nu) / ((1.0 + nu) * (1.0 - 2.0 * nu));
251 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
252 const auto& E = vp.emodul;
253 const auto& mu = vp.mu;
254 return mu * (4.0 * mu - E) / (3.0 * mu - E);
255 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
256 const auto& E = vp.emodul;
257 const auto& K = vp.K;
258 return 3.0 * K * (3.0 * K + E) / (9.0 * K - E);
259 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
260 const auto& E = vp.emodul;
261 const auto& lambda = vp.lambda;
262 return (E - lambda + calcR(vp)) / 2.0;
263 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
264 const auto& K = vp.K;
265 const auto& lambda = vp.lambda;
266 return 3.0 * K - 2.0 * lambda;
267 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
268 const auto& lambda = vp.lambda;
269 const auto& mu = vp.mu;
270 return lambda + 2.0 * mu;
271 } else
272 assert(false && "Your LameParameter request is not implemented");
273 }
274
275 constexpr double toPoissonsRatio()
276 requires(!std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio>)
277 {
278 if constexpr (std::is_same_v<ValuePair, YoungsModulusAndShearModulus>) {
279 const auto& E = vp.emodul;
280 const auto& mu = vp.mu;
281 return E / (2.0 * mu) - 1.0;
282 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndBulkModulus>) {
283 const auto& E = vp.emodul;
284 const auto& K = vp.K;
285 return (3.0 * K - E) / (6.0 * K);
286 } else if constexpr (std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>) {
287 const auto& E = vp.emodul;
288 const auto& lambda = vp.lambda;
289 return 2.0 * lambda / (E + lambda + calcR(vp));
290 } else if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
291 const auto& K = vp.K;
292 const auto& lambda = vp.lambda;
293 return lambda / (3 * K - lambda);
294 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
295 const auto& lambda = vp.lambda;
296 const auto& mu = vp.mu;
297 return lambda / (2.0 * (lambda + mu));
298 } else
299 assert(false && "Your LameParameter request is not implemented");
300 }
301
302 constexpr double toYoungsModulus()
303 requires(!std::is_same_v<ValuePair, YoungsModulusAndPoissonsRatio> and
304 !std::is_same_v<ValuePair, YoungsModulusAndShearModulus> and
305 !std::is_same_v<ValuePair, YoungsModulusAndBulkModulus> and
306 !std::is_same_v<ValuePair, YoungsModulusAndLamesFirstParameter>)
307 {
308 if constexpr (std::is_same_v<ValuePair, BulkModulusAndLamesFirstParameter>) {
309 return 9.0 * vp.K * (vp.K - vp.lambda) / (3.0 * vp.K - vp.lambda);
310 } else if constexpr (std::is_same_v<ValuePair, LamesFirstParameterAndShearModulus>) {
311 const auto& lambda = vp.lambda;
312 const auto& mu = vp.mu;
313 return mu * (3.0 * lambda + 2.0 * mu) / (lambda + mu);
314 } else
315 assert(false && "Your LameParameter request is not implemented");
316 }

337 double calcR(const YoungsModulusAndLamesFirstParameter& vp_) {
338 const auto& E = vp_.emodul;
339 const auto& lambda = vp_.lambda;
340 return std::sqrt(E * E + 9 * lambda * lambda + 2 * E * lambda);
341 }

370inline auto toLamesFirstParameterAndShearModulus(const YoungsModulusAndPoissonsRatio& matParameter) {
371 auto lambda = convertLameConstants(matParameter).toLamesFirstParameter();
372 auto mu = convertLameConstants(matParameter).toShearModulus();
373
374 return LamesFirstParameterAndShearModulus{.lambda = lambda, .mu = mu};
