173 template <typename SolutionType = NoPredictor>
174 requires std::is_same_v<SolutionType, NoPredictor> || std::is_convertible_v<SolutionType, CorrectionType>
175 NonLinearSolverInformation solve(const SolutionType& dxPredictor = NoPredictor{}) {
176 this->notify(NonLinearSolverMessages::INIT);
177 stats_ = Stats{};
178 info_ = AlgoInfo{};
179
180 NonLinearSolverInformation solverInformation;
181 nonLinearOperator().updateAll();
182 stats_.energy = energy();
183 auto& x = nonLinearOperator().firstParameter();
184 xOld_ = x;
185 stats_.gradNorm = norm(gradient());
186 if constexpr (not std::is_same_v<SolutionType, NoPredictor>)
187 updateFunction(x, dxPredictor);
188 truncatedConjugateGradient_.analyzePattern(hessian());
189
190 innerInfo_.Delta = options_.Delta0;
191 spdlog::info(
192 " | iter | inner_i | rho | energy | energy_p | energy_inc | norm(g) | Delta | norm(corr) | "
193 "InnerBreakReason");
194 spdlog::info("{:-^143}", "-");
195 while (not stoppingCriterion()) {
196 this->notify(NonLinearSolverMessages::ITERATION_STARTED);
197 if (options_.useRand) {
198 if (stats_.outerIter == 0) {
199 eta_.setRandom();
200 while (eta_.dot(hessian() * eta_) > innerInfo_.Delta * innerInfo_.Delta)
201 eta_ *= eps_; // eps is sqrt(sqrt(maschine-precision))
202 } else
203 eta_.setZero();
204 } else
205 eta_.setZero();
206
207 solveInnerProblem();
208 stats_.innerIterSum += innerInfo_.numInnerIter;
209
210 info_.stopReasonString = tcg_stop_reason_[static_cast<int>(innerInfo_.stop_tCG)];
211 Heta_ = hessian() * eta_;
212 if (options_.useRand and stats_.outerIter == 0) {
213 info_.used_cauchy = false;
214 info_.randomPredictionString = " Used Random correction predictor";
215 info_.cauchystr = " ";
216 double tauC;
217 // Check the curvature
218 const Eigen::VectorXd Hg = hessian() * gradient();
219 const auto g_Hg = (gradient().dot(Hg));
220 if (g_Hg <= 0)
221 tauC = 1;
222 else
223 tauC = std::min(Dune::power(stats_.gradNorm, 3) / (innerInfo_.Delta * g_Hg), 1.0);
224
225 // generate the Cauchy point.
226 const Eigen::VectorXd etaC = -tauC * innerInfo_.Delta / stats_.gradNorm * gradient();
227 const Eigen::VectorXd HetaC = -tauC * innerInfo_.Delta / stats_.gradNorm * Hg;
228
229 const double mdle = stats_.energy + gradient().dot(eta_) + .5 * Heta_.dot(eta_);
230 const double mdlec = stats_.energy + gradient().dot(etaC) + .5 * HetaC.dot(etaC);
231 if (mdlec < mdle && stats_.outerIter == 0) {
232 eta_ = etaC;
233 Heta_ = HetaC;
234 info_.used_cauchy = true;
235
236 info_.cauchystr = " USED CAUCHY POINT!";
237 }
238 } else
239 info_.cauchystr = " ";
240
241 stats_.etaNorm = eta_.norm();
242
243 updateFunction_(x, eta_);
244
245 // Calculate energy of our proposed update step
246 nonLinearOperator().template update<0>();
247 stats_.energyProposal = energy();
248
249 // Will we accept the proposal or not?
250 // Check the performance of the quadratic model against the actual energy.
251 auto rhonum = stats_.energy - stats_.energyProposal;
252 auto rhoden = -eta_.dot(gradient() + 0.5 * Heta_);
253
254 /* Close to convergence the proposed energy and the real energy almost coincide.
255 * Therefore, the performance check of our model becomes ill-conditioned
256 * The regularisation fixes this */
257 const auto rhoReg = std::max(1.0, abs(stats_.energy)) * eps_ * options_.rho_reg;
258 rhonum = rhonum + rhoReg;
259 rhoden = rhoden + rhoReg;
260
261 const bool modelDecreased = rhoden > 0.0;
262
263 if (!modelDecreased)
264 info_.stopReasonString.append(", model did not decrease");
265
266 stats_.rho = rhonum / rhoden;
267 stats_.rho = stats_.rho < 0.0 ? -1.0 : stats_.rho; // move rho to the domain [-1.0,inf]
268
269 info_.trstr = " ";
270
271 // measure if energy decreased
272 const bool energyDecreased = Dune::FloatCmp::ge(stats_.energy - stats_.energyProposal, -1e-12);
273
274 // If the model behaves badly or if the energy increased we reduce the trust region size
275 if (stats_.rho < 1e-4 || not modelDecreased || std::isnan(stats_.rho) || not energyDecreased) {
276 info_.trstr = "TR-";
277 innerInfo_.Delta /= 4.0;
278 info_.consecutive_TRplus = 0;
279 info_.consecutive_TRminus++;
280 if (info_.consecutive_TRminus >= 5 && options_.verbosity >= 1) {
281 info_.consecutive_TRminus = -std::numeric_limits<int>::infinity();
282 spdlog::info(" +++ Detected many consecutive TR- (radius decreases).");
283 spdlog::info(" +++ Consider decreasing options.Delta_bar by an order of magnitude.");
284 }
285
286 } else if (stats_.rho > 0.99 && (innerInfo_.stop_tCG == Eigen::TCGStopReason::negativeCurvature ||
287 innerInfo_.stop_tCG == Eigen::TCGStopReason::exceededTrustRegion)) {
288 info_.trstr = "TR+";
289 innerInfo_.Delta = std::min(3.5 * innerInfo_.Delta, options_.Delta_bar);
290 info_.consecutive_TRminus = 0;
291 info_.consecutive_TRplus++;
292 if (info_.consecutive_TRplus >= 5 && options_.verbosity >= 1) {
293 info_.consecutive_TRplus = -std::numeric_limits<int>::infinity();
294 spdlog::info(" +++ Detected many consecutive TR+ (radius increases)");
295 spdlog::info(" +++ Consider increasing options.Delta_bar by an order of magnitude");
296
297 } else {
298 info_.consecutive_TRplus = 0;
299 info_.consecutive_TRminus = 0;
300 }
301 }
302
303 if (modelDecreased && stats_.rho > options_.rho_prime && energyDecreased) {
304 if (stats_.energyProposal > stats_.energy)
305 spdlog::info(
306 "Energy function increased by {} (step size: {}). Since this is small we accept the step and hope for "
307 "convergence of the gradient norm.",
308 stats_.energyProposal - stats_.energy, stats_.etaNorm);
309
310 info_.acceptProposal = true;
311 info_.accstr = "acc";
312 info_.consecutive_Rejected = 0;
313 } else {
314 info_.acceptProposal = false;
315 info_.accstr = "REJ";
316
317 if (info_.consecutive_Rejected >= 5)
318 innerInfo_.Delta /= 2;
319 else
320 innerInfo_.Delta = std::min(innerInfo_.Delta, stats_.etaNorm / 2.0);
321 ++info_.consecutive_Rejected;
322 }
323
324 stats_.outerIter++;
325
326 if (options_.verbosity == 1)
327 logState();
328
329 info_.randomPredictionString = "";
330
331 if (info_.acceptProposal) {
332 stats_.energy = stats_.energyProposal;
333 nonLinearOperator_.updateAll();
334 xOld_ = x;
335 this->notify(NonLinearSolverMessages::CORRECTIONNORM_UPDATED, stats_.etaNorm);
336 this->notify(NonLinearSolverMessages::RESIDUALNORM_UPDATED, stats_.gradNorm);
337 this->notify(NonLinearSolverMessages::SOLUTION_CHANGED);
338 } else {
339 x = xOld_;
340 eta_.setZero();
341 }
342 nonLinearOperator_.updateAll();
343 stats_.gradNorm = gradient().norm();
344 this->notify(NonLinearSolverMessages::ITERATION_ENDED);
345 }
346 spdlog::info("{}", info_.reasonString);
347 spdlog::info("Total iterations: {} Total CG Iterations: {}", stats_.outerIter, stats_.innerIterSum);
348
349 solverInformation.success =
350 (info_.stop == StopReason::correctionNormTolReached) or (info_.stop == StopReason::gradientNormTolReached);
351
352 solverInformation.iterations = stats_.outerIter;
353 solverInformation.residualNorm = stats_.gradNorm;
354 if (solverInformation.success)
355 this->notify(NonLinearSolverMessages::FINISHED_SUCESSFULLY, solverInformation.iterations);
356 return solverInformation;
357 }

384 bool stoppingCriterion() {
385 std::ostringstream stream;
387 if (stats_.gradNorm < options_.grad_tol && stats_.outerIter != 0) {
388 logFinalState();
389 spdlog::info("CONVERGENCE: Energy: {:1.16e} norm(gradient): {:1.16e}", nonLinearOperator().value(),
390 stats_.gradNorm);
391 stream << "Gradient norm tolerance reached; options.tolerance = " << options_.grad_tol;
392
393 info_.reasonString = stream.str();
394
395 info_.stop = StopReason::gradientNormTolReached;
396 return true;
397 } else if (stats_.etaNorm < options_.corr_tol && stats_.outerIter != 0) {
398 logFinalState();
399 spdlog::info("CONVERGENCE: Energy: {:1.16e} norm(correction): {:1.16e}", nonLinearOperator().value(),
400 stats_.etaNorm);
401 stream << "Displacement norm tolerance reached; = " << options_.corr_tol << "." << std::endl;
402
403 info_.reasonString = stream.str();
404 info_.stop = StopReason::correctionNormTolReached;
405 return true;
406 }
407
409 if (stats_.time >= options_.maxtime) {
410 logFinalState();
411 stream << "Max time exceeded; options.maxtime = " << options_.maxtime << ".";
412 info_.reasonString = stream.str();
413 info_.stop = StopReason::maximumTimeReached;
414 return true;
415 }
416
418 if (stats_.outerIter >= options_.maxiter) {
419 logFinalState();
420 stream << "Max iteration count reached; options.maxiter = " << options_.maxiter << ".";
421 info_.reasonString = stream.str();
422 info_.stop = StopReason::maximumIterationsReached;
423 return true;
424 }
425 return false;
426 }

428 void solveInnerProblem() {
429 truncatedConjugateGradient_.setInfo(innerInfo_);
430 int attempts = 0;
431 truncatedConjugateGradient_.factorize(hessian());
432 // If the preconditioner is IncompleteCholesky the factorization may fail if we have negative diagonal entries and
433 // the initial shift is too small. Therefore, if the factorization fails we increase the initial shift by a factor
434 // of 5.
435 if constexpr (preConditioner == PreConditioner::IncompleteCholesky) {
436 while (truncatedConjugateGradient_.info() != Eigen::Success) {
437 choleskyInitialShift_ *= 5;
438 truncatedConjugateGradient_.preconditioner().setInitialShift(choleskyInitialShift_);
439 truncatedConjugateGradient_.factorize(hessian());
440 if (attempts > 5)
441 DUNE_THROW(Dune::MathError, "Factorization of preconditioner failed!");
442 ++attempts;
443 }
444 if (truncatedConjugateGradient_.info() == Eigen::Success)
445 choleskyInitialShift_ = 1e-3;
446 }
447 eta_ = truncatedConjugateGradient_.solveWithGuess(-gradient(), eta_);
448 innerInfo_ = truncatedConjugateGradient_.getInfo();
449 }
