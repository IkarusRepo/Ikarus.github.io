35 void updateImpl(ControlMessages message) final {
36 assert(isFieldInfoSet && "You need to call setFieldInfo first!");
37 switch (message) {
38 case ControlMessages::SOLUTION_CHANGED: {
39 auto disp = Dune::Functions::makeDiscreteGlobalBasisFunction<Dune::FieldVector<double, components>>(
40 *basis, *solution);
41 vtkWriter.addVertexData(disp, fieldInfo);
42 vtkWriter.write(prefixString + std::to_string(step++));
43 } break;
44 default:
45 break; // default: do nothing when notified
46 }
47 }
