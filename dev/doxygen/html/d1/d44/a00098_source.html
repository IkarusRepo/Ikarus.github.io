27 template <typename ScalarType = double, int dim = 3>
28 auto symmetricIdentityFourthOrder() {
29 Eigen::TensorFixedSize<double, Eigen::Sizes<dim, dim, dim, dim>> idTensor;
30 for (int i = 0; i < dim; ++i)
31 for (int j = 0; j < dim; ++j)
32 for (int k = 0; k < dim; ++k)
33 for (int l = 0; l < dim; ++l)
34 idTensor(i, j, k, l) = 0.5 * ((i == k) * (j == l) + (i == l) * (j == k));
35 return idTensor;
36 }

60 template <typename AType, typename BType>
61 auto fourthOrderIKJL(const Eigen::MatrixBase<AType>& A, const Eigen::MatrixBase<BType>& B) {
62 static_assert(AType::RowsAtCompileTime == BType::RowsAtCompileTime);
63 static_assert(AType::ColsAtCompileTime == BType::ColsAtCompileTime);
64 using ScalarResultType =
65 typename Dune::PromotionTraits<typename AType::Scalar, typename BType::Scalar>::PromotedType;
66 constexpr int dim = AType::RowsAtCompileTime;
67 Eigen::TensorFixedSize<ScalarResultType, Eigen::Sizes<dim, dim, dim, dim>> res;
68 for (int i = 0; i < dim; ++i)
69 for (int j = 0; j < dim; ++j)
70 for (int k = 0; k < dim; ++k)
71 for (int l = 0; l < dim; ++l)
72 res(i, j, k, l) = A(i, k) * B(j, l);
73 return res;

76 template <typename ScalarType, long int dim>
77 auto symTwoSlots(const Eigen::TensorFixedSize<ScalarType, Eigen::Sizes<dim, dim, dim, dim>>& t,
78 const std::array<size_t, 2>& slots) {
79 std::array<size_t, 4> shuffleSlot;
80 std::iota(shuffleSlot.begin(), shuffleSlot.end(), 0); // create 0,1,2,3 array
81 std::swap(shuffleSlot[slots[0]], shuffleSlot[slots[1]]); // swap the given slots
82 return (0.5 * (t + t.shuffle(shuffleSlot))).eval();
83 }

85 constexpr Eigen::Index toVoigt(Eigen::Index i, Eigen::Index j) noexcept {
86 if (i == j) // _00 -> 0, _11 -> 1, _22 -> 2
87 return i;
88 else if ((i == 1 and j == 2) or (i == 2 and j == 1)) // _12 and _21 --> 3
89 return 3;
90 else if ((i == 0 and j == 2) or (i == 2 and j == 0)) // _02 and _20 --> 4
91 return 4;
92 else if ((i == 0 and j == 1) or (i == 1 and j == 0)) // _01 and _10 --> 5
93 return 5;
94 else {
95 assert(i < 3 and j < 3 && "For Voigt notation the indices need to be 0,1 or 2.");
96 __builtin_unreachable();
97 }
98 }

111 template <typename ST, int size, int Options, int MaxRows>
112 requires(size > 0 and size <= 3) auto toVoigt(const Eigen::Matrix<ST, size, size, Options, MaxRows, MaxRows>& E,
113 bool isStrain = true) {
114 Eigen::Vector<ST, (size * (size + 1)) / 2> EVoigt;
115 EVoigt.template segment<size>(0) = E.diagonal();
116
117 const ST possibleStrainFactor = isStrain ? 2.0 : 1.0;
118 if constexpr (size == 2)
119 EVoigt(2) = E(0, 1) * possibleStrainFactor;
120 else if constexpr (size == 3) {
121 EVoigt(size) = E(1, 2) * possibleStrainFactor;
122 EVoigt(size + 1) = E(0, 2) * possibleStrainFactor;
123 EVoigt(size + 2) = E(0, 1) * possibleStrainFactor;
124 }
125 return EVoigt;
126 }

128 constexpr std::array<size_t, 2> fromVoigt(size_t i) {
129 if (i < 3) // _00 -> 0, _11 -> 1, _22 -> 2
130 return {i, i};
131 else if (i == 3)
132 return {1, 2};
133 else if (i == 4)
134 return {0, 2};
135 else if (i == 5)
136 return {0, 1};
137 else {
138 assert(i < 6 && "For Voigt notation the indices need to be 0 and 5.");
139 __builtin_unreachable();
140 }

143 template <typename ST, int size>
144 requires(size == 1 or size == 3 or size == 6) auto fromVoigt(const Eigen::Vector<ST, size>& EVoigt,
145 bool isStrain = true) {
146 constexpr int matrixSize = (-1 + Ikarus::ct_sqrt(1 + 8 * size)) / 2;
147 Eigen::Matrix<ST, matrixSize, matrixSize> E;
148 E.diagonal() = EVoigt.template segment<3>(0);
149
150 const ST possibleStrainFactor = isStrain ? 0.5 : 1.0;
151 if constexpr (matrixSize == 2)
152 E(0, 1) = E(1, 0) = EVoigt(2) * possibleStrainFactor;
153 else if constexpr (matrixSize == 3) {
154 E(2, 1) = E(1, 2) = EVoigt(matrixSize) * possibleStrainFactor;
155 E(2, 0) = E(0, 2) = EVoigt(matrixSize + 1) * possibleStrainFactor;
156 E(1, 0) = E(0, 1) = EVoigt(matrixSize + 2) * possibleStrainFactor;
157 }
158 return E;
159 }

175 template <typename Derived, size_t sizeOfCondensedIndices>
176 auto staticCondensation(const Eigen::MatrixBase<Derived>& E,
177 const std::array<size_t, sizeOfCondensedIndices>& indices) {
178 constexpr size_t colsFull = std::remove_cvref_t<Derived>::ColsAtCompileTime;
179 constexpr size_t rowsFull = std::remove_cvref_t<Derived>::RowsAtCompileTime;
180 static_assert(colsFull == rowsFull, "staticCondensation only allowed for square matrices");
181 std::array<size_t, rowsFull - sizeOfCondensedIndices> remainingIndices{};
182 std::ranges::set_difference(std::ranges::iota_view(size_t(0), size_t(colsFull)), indices, remainingIndices.begin());
183
184 auto K11 = E(remainingIndices, remainingIndices);
185 auto K12 = E(indices, remainingIndices);
186 auto K22 = E(indices, indices);
187
188 return (K11 - K12.transpose() * K22.inverse() * K12).eval();
189 }

191 template <typename Derived, size_t sizeOfRemovedCols>
192 auto removeCol(const Eigen::MatrixBase<Derived>& E, const std::array<size_t, sizeOfRemovedCols>& indices) {
193 constexpr size_t colsFull = std::remove_cvref_t<Derived>::ColsAtCompileTime;
194 constexpr size_t rowsFull = std::remove_cvref_t<Derived>::RowsAtCompileTime;
195 static_assert(colsFull == 1);
196
197 std::array<size_t, rowsFull - sizeOfRemovedCols> remainingIndices{};
198 std::ranges::set_difference(std::ranges::iota_view(size_t(0), size_t(rowsFull)), indices, remainingIndices.begin());
199
200 return (E(remainingIndices)).eval();
201 }

204 auto toVoigtAndMaybeReduce(const Eigen::Matrix<ST, 3, 3>& E, const MaterialImpl&, bool isStrain = true) {

218 template <typename Material, typename Derived>
219 decltype(auto) enlargeIfReduced(const Eigen::MatrixBase<Derived>& E) {
220 if constexpr (!Concepts::EigenVector6<Derived> and Concepts::EigenVector<Derived>) {
221 static_assert(Material::isReduced, "You should only end up here, if your material is reduced");
222
223 auto freeindices = Material::MaterialImpl::freeVoigtIndices;
224 auto p_E = Eigen::Vector<typename Material::MaterialImpl::ScalarType, 6>::Zero().eval();
225 for (int ri = 0; auto i : freeindices)
226 p_E(i) = E(ri++);
227 return p_E;
228
229 } else if constexpr (Concepts::EigenMatrix66<
230 Derived> or Concepts::EigenMatrix33<Derived> or Concepts::EigenVector6<Derived>) {
231 return E.derived();
232 } else {
233 static_assert(Material::isReduced, "You should only end up here, if your material is reduced");
234
235 auto freeindices = Material::MaterialImpl::freeVoigtIndices;
236 auto p_E = Eigen::Matrix<typename Material::MaterialImpl::ScalarType, 6, 6>::Zero().eval();
237 for (int ri = 0; auto i : freeindices) {
238 for (int rj = 0; auto j : freeindices)
239 p_E(i, j) = E(ri, rj++);
240 ++ri;
241 }
242 return p_E;
