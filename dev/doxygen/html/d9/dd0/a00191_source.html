41 ControlInformation run() {
42 ControlInformation info;
43 auto& nonOp = nonLinearSolver->nonLinearOperator();
44 this->notify(ControlMessages::CONTROL_STARTED, pathFollowingType_.name);
45
46 SubsidiaryArgs subsidiaryArgs;
47
48 info.totalIterations = 0;
49 subsidiaryArgs.stepSize = stepSize_;
50 subsidiaryArgs.DD.resizeLike(nonOp.firstParameter());
51 subsidiaryArgs.DD.setZero();
52
54 this->notify(ControlMessages::STEP_STARTED, 0, subsidiaryArgs.stepSize);
55 auto subsidiaryFunctionPtr = [&](auto&& args) { return pathFollowingType_.evaluateSubsidiaryFunction(args); };
56 pathFollowingType_.initialPrediction(nonOp, subsidiaryArgs);
57 auto solverInfo = nonLinearSolver->solve(subsidiaryFunctionPtr, subsidiaryArgs);
58 info.solverInfos.push_back(solverInfo);
59 info.totalIterations += solverInfo.iterations;
60 if (not solverInfo.success) return info;
61 this->notify(ControlMessages::SOLUTION_CHANGED);
62 this->notify(ControlMessages::STEP_ENDED);
63
65 for (int ls = 1; ls < loadSteps_; ++ls) {
66 subsidiaryArgs.currentStep = ls;
67
68 adaptiveStepSizing(solverInfo, subsidiaryArgs, nonOp);
69
70 this->notify(ControlMessages::STEP_STARTED, subsidiaryArgs.currentStep, subsidiaryArgs.stepSize);
71
72 pathFollowingType_.intermediatePrediction(nonOp, subsidiaryArgs);
73
74 solverInfo = nonLinearSolver->solve(subsidiaryFunctionPtr, subsidiaryArgs);
75
76 info.solverInfos.push_back(solverInfo);
77 info.totalIterations += solverInfo.iterations;
78 if (not solverInfo.success) return info;
79 this->notify(ControlMessages::SOLUTION_CHANGED);
80 this->notify(ControlMessages::STEP_ENDED);
81 }
82
83 this->notify(ControlMessages::CONTROL_ENDED, info.totalIterations, pathFollowingType_.name);
84 info.success = true;
85 return info;
86 }
