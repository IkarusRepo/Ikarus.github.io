22 template <typename Derived>
23 auto orthonormalizeMatrixColumns(const Eigen::MatrixBase<Derived>& A) {
24 // Gram Schmidt Ortho
25 auto Q = A.eval();
26
27 Q.col(0).normalize();
28
29 for (int colIndex = 1; colIndex < Q.cols(); colIndex++) {
30 Q.col(colIndex) -= Q.leftCols(colIndex) * (Q.leftCols(colIndex).transpose() * A.col(colIndex));
31 Q.col(colIndex).normalize();
32 }
33
34 return Q;
35 }

132 template <typename... Types, typename Derived>

134 using namespace Dune::Indices;
135 size_t posStart = 0;
136 Dune::Hybrid::forEach(Dune::Hybrid::integralRange(Dune::index_constant<sizeof...(Types)>()), [&](const auto i) {
137 const size_t size = correctionSize(a[i]);

226 template <typename Derived>
227 auto eval(const Eigen::EigenBase<Derived>& A) {
228 if constexpr (static_cast<bool>(
229 Eigen::internal::is_diagonal<Derived>::ret)) // workaround needed since Eigen::DiagonalWrapper
230 // does not has a eval function
231 {
232 using Scalar = typename Derived::Scalar;
233 using namespace Eigen;
234 constexpr int diag_size = EIGEN_SIZE_MIN_PREFER_DYNAMIC(Derived::RowsAtCompileTime, Derived::ColsAtCompileTime);
235 constexpr int max_diag_size
236 = EIGEN_SIZE_MIN_PREFER_FIXED(Derived::MaxRowsAtCompileTime, Derived::MaxColsAtCompileTime);
237
238 return Eigen::DiagonalMatrix<Scalar, diag_size, max_diag_size>(A.derived().diagonal());
239 } else
240 return A.derived().eval();
241 }

339 template <typename FieldVectorT>
340 auto createRandomVector(typename FieldVectorT::value_type lower = -1, typename FieldVectorT::value_type upper = 1) {
341 std::random_device rd;
342 std::mt19937 mt(rd());
343 std::uniform_real_distribution<typename FieldVectorT::value_type> dist(lower, upper);
344 auto rand = [&dist, &mt]() { return dist(mt); };
345 FieldVectorT vec;
346 std::generate(vec.begin(), vec.end(), rand);
347 return vec;
348 }
