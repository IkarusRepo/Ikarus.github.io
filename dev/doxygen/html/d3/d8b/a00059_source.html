18 struct VonMises {
19 template <typename ElementType, typename FERequirements, int size, typename ScalarType>
20 double operator()(const ElementType &fe, const Ikarus::ResultRequirements<FERequirements> &req,
21 const Dune::FieldVector<ScalarType, size> &pos, [[maybe_unused]] int comp) const
22 requires(size == 2) {
23 Ikarus::ResultTypeMap<ScalarType> res_;
24 fe.calculateAt(req, pos, res_);
25
26 const auto &[resultType, sigma] = res_.getSingleResult();
27 assert(resultType == ResultType::cauchyStress or resultType == ResultType::PK2Stress);
28 const auto s_x = sigma(0, 0);
29 const auto s_y = sigma(1, 0);
30 const auto s_xy = sigma(2, 0);
31
32 return std::sqrt(std::pow(s_x, 2) + Dune::power(s_y, 2) - s_x * s_y + 3 * Dune::power(s_xy, 2));
33 }
34 static std::string name() { return "VonMises"; }
35 static int ncomps() { return 1; }
36 };

38 struct PrincipalStress {
39 template <typename ElementType, typename FERequirements, int size, typename ScalarType>
40 double operator()(const ElementType &fe, const Ikarus::ResultRequirements<FERequirements> &req,
41 const Dune::FieldVector<ScalarType, size> &pos, int comp) const requires(size == 2) {
42 Ikarus::ResultTypeMap<ScalarType> res_;
43 fe.calculateAt(req, pos, res_);
44
45 const auto &[resultType, sigma] = res_.getSingleResult();
46 assert(resultType == ResultType::cauchyStress or resultType == ResultType::PK2Stress);
47 const auto s_x = sigma(0, 0);
48 const auto s_y = sigma(1, 0);
49 const auto s_xy = sigma(2, 0);
50
51 auto t1 = (s_x + s_y) / 2;
52 auto t2 = std::sqrt(Dune::power((s_x - s_y) / 2, 2) + Dune::power(s_xy, 2));
53
54 return comp == 0 ? t1 + t2 : t1 - t2;
55 }
56 static std::string name() { return "PrincipalStress"; }
57 static int ncomps() { return 2; }
