169 template <FEAffordance Affordance>
170 bool hasAffordance(Affordance &&affordance) const {
171 if constexpr (std::is_same_v<Affordance, ScalarAffordances>)
172 return affordances.scalarAffordances == affordance;
173 else if constexpr (std::is_same_v<Affordance, VectorAffordances>)
174 return affordances.vectorAffordances == affordance;
175 else if constexpr (std::is_same_v<Affordance, MatrixAffordances>)
176 return affordances.matrixAffordances == affordance;
177 else if constexpr (std::is_same_v<Affordance, AffordanceCollectionImpl>)
178 return affordances == affordance;
179 }

181 private:
182 std::map<FESolutions, SolutionVectorType> sols;
183 std::map<FEParameter, ParameterType> parameter;
184 AffordanceCollectionImpl affordances;
