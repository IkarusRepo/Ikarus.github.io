25template <typename Geometry>
26struct Q1E4
27{
28 static constexpr int strainSize = 3;
29 static constexpr int enhancedStrainSize = 4;
30 using MType = Eigen::Matrix<double, strainSize, enhancedStrainSize>;
31
32 Q1E4() = default;
33 explicit Q1E4(const Geometry& geometry_)
34 : geometry{std::make_shared<Geometry>(geometry_)},
35 T0InverseTransformed{calcTransformationMatrix2D(geometry_)} {}
36
37 auto calcM(const Dune::FieldVector<double, 2>& quadPos) const {
38 MType M;
39 M.setZero(strainSize, enhancedStrainSize);
40 const double xi = quadPos[0];
41 const double eta = quadPos[1];
42 M(0, 0) = 2 * xi - 1.0;
43 M(1, 1) = 2 * eta - 1.0;
44 M(2, 2) = 2 * xi - 1.0;
45 M(2, 3) = 2 * eta - 1.0;
46 const double detJ = geometry->integrationElement(quadPos);
47 M = T0InverseTransformed / detJ * M;
48 return M;
49 }
50
51 std::shared_ptr<Geometry> geometry;
52 Eigen::Matrix3d T0InverseTransformed;
53};
