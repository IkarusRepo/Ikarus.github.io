78 template <typename VolumeLoad = utils::LoadDefault, typename NeumannBoundaryLoad = utils::LoadDefault>
79 LinearElastic(const Basis& globalBasis, const typename LocalView::Element& element, double emod, double nu,
80 VolumeLoad volumeLoad = {}, const BoundaryPatch<GridView>* neumannBoundary = nullptr,
81 NeumannBoundaryLoad neumannBoundaryLoad = {})
82 : BaseDisp(globalBasis, element),
83 VolumeType(volumeLoad),
84 TractionType(neumannBoundary, neumannBoundaryLoad),
85 emod_{emod},
86 nu_{nu} {
87 this->localView().bind(element);
88 auto& firstChild = this->localView().tree().child(0);
89 const auto& fe = firstChild.finiteElement();
90 geo_ = std::make_shared<const Geometry>(this->localView().element().geometry());
91 numberOfNodes_ = fe.size();
92 order_ = 2 * (fe.localBasis().order());
93 localBasis_ = Dune::CachedLocalBasis(fe.localBasis());
94 if constexpr (requires { this->localView().element().impl().getQuadratureRule(order_); })
95 if (this->localView().element().impl().isTrimmed())
96 localBasis_.bind(this->localView().element().impl().getQuadratureRule(order_), Dune::bindDerivatives(0, 1));
97 else
98 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
99 Dune::bindDerivatives(0, 1));
100 else
101 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(this->localView().element().type(), order_),
102 Dune::bindDerivatives(0, 1));
103 }

184 void calculateMatrix(const FERequirementType& par, typename Traits::template MatrixType<> K) const {
185 const auto eps = strainFunction(par);
186 using namespace Dune::DerivativeDirections;
187 using namespace Dune;
188
189 const auto C = materialTangent();
190 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
191 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
192 for (size_t i = 0; i < numberOfNodes_; ++i) {
193 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
194 for (size_t j = 0; j < numberOfNodes_; ++j) {
195 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
196 K.template block<myDim, myDim>(i * myDim, j * myDim) += bopI.transpose() * C * bopJ * intElement;
197 }
198 }
199 }
200
201 // Update due to displacement-dependent external loads, e.g., follower loads
202 VolumeType::calculateMatrix(par, K);
203 TractionType::calculateMatrix(par, K);
204 }

215 template <ResultType resType>
216 auto calculateAt(const FERequirementType& req, const Dune::FieldVector<double, Traits::mydim>& local) const {
217 static_assert(resType == ResultType::linearStress, "The requested result type is NOT implemented.");
218
219 if constexpr (resType == ResultType::linearStress) {
220 const auto eps = strainFunction(req);
221 const auto C = materialTangent();
222 auto epsVoigt = eps.evaluate(local, Dune::on(Dune::DerivativeDirections::gridElement));
223
224 return (C * epsVoigt).eval();
225 }
226 }

237 template <typename ScalarType>
238 auto calculateScalarImpl(const FERequirementType& par,
239 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const
240 -> ScalarType {
241 const auto uFunction = displacementFunction(par, dx);
242 const auto eps = strainFunction(par, dx);
243 const auto& lambda = par.getParameter(Ikarus::FEParameter::loadfactor);
244 using namespace Dune::DerivativeDirections;
245 using namespace Dune;
246
247 const auto C = materialTangent();
248
249 ScalarType energy = 0.0;
250 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
251 const auto EVoigt = eps.evaluate(gpIndex, on(gridElement));
252 energy += (0.5 * EVoigt.dot(C * EVoigt)) * geo_->integrationElement(gp.position()) * gp.weight();
253 }
254
255 // External forces volume forces over the domain
256 energy += VolumeType::calculateScalarImpl(par, dx);
257
258 // line or surface loads, i.e., neumann boundary
259 energy += TractionType::calculateScalarImpl(par, dx);
260 return energy;
261 }

263 template <typename ScalarType>
264 void calculateVectorImpl(const FERequirementType& par, typename Traits::template VectorType<ScalarType> force,
265 const std::optional<const Eigen::VectorX<ScalarType>>& dx = std::nullopt) const {
266 const auto eps = strainFunction(par, dx);
267 using namespace Dune::DerivativeDirections;
268 using namespace Dune;
269
270 const auto C = materialTangent();
271
272 // Internal forces
273 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
274 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
275 auto stresses = (C * eps.evaluate(gpIndex, on(gridElement))).eval();
276 for (size_t i = 0; i < numberOfNodes_; ++i) {
277 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
278 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
279 }
280 }
281
282 // External forces volume forces over the domain
283 VolumeType::calculateVectorImpl(par, force, dx);
284
285 // External forces, boundary forces, i.e., at the Neumann boundary
286 TractionType::calculateVectorImpl(par, force, dx);
287 }
